21#[instrument(skip_all)]
22#[allow(clippy::multiple_bound_locations, clippy::type_repetition_in_bounds)] // cfg not supported in where clauses?
23pub fn prove_with_preprocessed<
24    SC,
25    #[cfg(debug_assertions)] A: for<'a> Air<crate::check_constraints::DebugConstraintBuilder<'a, Val<SC>>>,
26    #[cfg(not(debug_assertions))] A,
27>(
28    config: &SC,
29    air: &A,
30    trace: RowMajorMatrix<Val<SC>>,
31    public_values: &[Val<SC>],
32    preprocessed: Option<&PreprocessedProverData<SC>>,
33) -> Proof<SC>
34where
35    SC: StarkGenericConfig,
36    A: Air<SymbolicAirBuilder<Val<SC>>> + for<'a> Air<ProverConstraintFolder<'a, SC>>,
37{
38    #[cfg(debug_assertions)]
39    crate::check_constraints::check_constraints(air, &trace, public_values);
40
41    // Compute the height `N = 2^n` and `log_2(height)`, `n`, of the trace.
42    let degree = trace.height();
43    let log_degree = log2_strict_usize(degree);
44    let log_ext_degree = log_degree + config.is_zk();
45
46    // Get preprocessed width for symbolic constraint evaluation.
47    //
48    // - If reusable preprocessed prover data is provided, trust its width and degree_bits
49    //   (and enforce consistency).
50    // - Otherwise, if the AIR defines preprocessed columns, we treat it as an error:
51    //   callers must use `setup_preprocessed` and pass the resulting data in.
52    let preprocessed_width = preprocessed.map_or_else(
53        || {
54            if let Some(preprocessed_trace) = air.preprocessed_trace() {
55                let width = preprocessed_trace.width();
56                if width > 0 {
57                    panic!(
58                        "AIR defines preprocessed columns (width = {}), \
59                         but no PreprocessedProverData was provided. \
60                         Call `setup_preprocessed` and pass it to `prove_with_preprocessed`.",
61                        width
62                    );
63                }
64            }
65            0
66        },
67        |pp| {
68            // Preprocessed columns are currently only supported in non-ZK mode.
69            assert_eq!(
70                config.is_zk(),
71                0,
72                "preprocessed columns are not supported in zk mode"
73            );
74            assert_eq!(
75                pp.degree_bits, log_ext_degree,
76                "PreprocessedProverData degree_bits does not match trace degree_bits"
77            );
78            pp.width
79        },
80    );
81
82    // Compute the constraint polynomials as vectors of symbolic expressions.
83    let symbolic_constraints =
84        get_symbolic_constraints(air, preprocessed_width, public_values.len());
85
86    // Count the number of constraints that we have.
87    let constraint_count = symbolic_constraints.len();
88
89    // Each constraint polynomial looks like `C_j(X_1, ..., X_w, Y_1, ..., Y_w, Z_1, ..., Z_j)`.
90    // When evaluated on a given row, the X_i's will be the `i`'th element of the that row, the
91    // Y_i's will be the `i`'th element of the next row and the Z_i's will be evaluations of
92    // selector polynomials on the given row index.
93    //
94    // When we convert to working with polynomials, the `X_i`'s and `Y_i`'s will be replaced by the
95    // degree `N - 1` polynomials `T_i(x)` and `T_i(hx)` respectively. The selector polynomials are
96    //  a little more complicated however.
97    //
98    // In our our case, the selector polynomials are `S_1(x) = is_first_row`, `S_2(x) = is_last_row`
99    // and `S_3(x) = is_transition`. Both `S_1(x)` and `S_2(x)` are polynomials of degree `N - 1`
100    // as they must be non zero only at a single location in the initial domain. However, `is_transition`
101    // is a polynomial of degree `1` as it simply need to be `0` on the last row.
102    //
103    // The constraint degree (`deg(C)`) is the linear factor of `N` in the constraint polynomial. In other
104    // words, it is roughly the total degree of `C` however, we treat `Z_3` as a constant term which does
105    // not contribute to the degree.
106    //
107    // E.g. `C_j = Z_1 * (X_1^3 - X_2 * X_3 * X_4)` would have degree `4`.
108    //      `C_j = Z_3 * (X_1^3 - X_2 * X_3 * X_4)` would have degree `3`.
109    //
110    // The point of all this is that, defining:
111    //          C(x) = C(T_1(x), ..., T_w(x), T_1(hx), ... T_w(hx), S_1(x), S_2(x), S_3(x))
112    // We get the constraint bound:
113    //          deg(C(x)) <= deg(C) * (N - 1) + 1
114    // The `+1` is due to the `is_transition` selector which is not accounted for in `deg(C)`. Note
115    // that S_i^2 should never appear in a constraint as it should just be replaced by `S_i`.
116    //
117    // For now in comments we assume that `deg(C) = 3` meaning `deg(C(x)) <= 3N - 2`
118
119    // From the degree of the constraint polynomial, compute the number
120    // of quotient polynomials we will split Q(x) into. This is chosen to
121    // always be a power of 2.
122    let log_num_quotient_chunks = get_log_num_quotient_chunks::<Val<SC>, A>(
123        air,
124        preprocessed_width,
125        public_values.len(),
126        config.is_zk(),
127    );
128
129    let num_quotient_chunks = 1 << (log_num_quotient_chunks + config.is_zk());
130
131    // Initialize the PCS and the Challenger.
132    let pcs = config.pcs();
133    let mut challenger = config.initialise_challenger();
134
135    // Get the subgroup `H` of size `N`. We treat each column `T_i` of
136    // the trace as an evaluation vector of polynomials `T_i(x)` over `H`.
137    // (In the Circle STARK case `H` is instead a standard position twin coset of size `N`)
138    let trace_domain = pcs.natural_domain_for_degree(degree);
139
140    // When ZK is enabled, we need to use an extended domain of size `2N` as we will
141    // add random values to the trace.
142    let ext_trace_domain = pcs.natural_domain_for_degree(degree * (config.is_zk() + 1));
143
144    // Let `g` denote a generator of the multiplicative group of `F` and `H'` the unique
145    // subgroup of `F` of size `N << (pcs.config.log_blowup + config.is_zk())`.
146    // If `zk` is enabled, we double the trace length by adding random values.
147    //
148    // For each trace column `T_i`, we compute the evaluation vector of `T_i(x)` over `H'`. This
149    // new extended trace `ET` is hashed into a Merkle tree with its rows bit-reversed.
150    //      trace_commit contains the root of the tree
151    //      trace_data contains the entire tree.
152    //          - trace_data.leaves is the matrix containing `ET`.
153    let (trace_commit, trace_data) =
154        info_span!("commit to trace data").in_scope(|| pcs.commit([(ext_trace_domain, trace)]));
155
156    // Preprocessed commitment and prover data (if any).
157    let (preprocessed_commit, preprocessed_data_ref) = preprocessed
158        .map(|pp| (pp.commitment.clone(), &pp.prover_data))
159        .unzip();
160
161    // Observe the instance.
162    // degree < 2^255 so we can safely cast log_degree to a u8.
163    challenger.observe(Val::<SC>::from_u8(log_ext_degree as u8));
164    challenger.observe(Val::<SC>::from_u8(log_degree as u8));
165    challenger.observe(Val::<SC>::from_usize(preprocessed_width));
166    // TODO: Might be best practice to include other instance data here; see verifier comment.
167
168    // Observe the Merkle root of the trace commitment.
169    challenger.observe(trace_commit.clone());
170    if preprocessed_width > 0 {
171        challenger.observe(preprocessed_commit.as_ref().unwrap().clone());
172    }
173
174    // Observe the public input values.
175    challenger.observe_slice(public_values);
176
177    // Get the first Fiat Shamir challenge which will be used to combine all constraint polynomials
178    // into a single polynomial.
179    //
180    // Soundness Error:
181    // If a prover is malicious, we can find a row `i` such that some of the constraints
182    // C_0, ..., C_n are non 0 on this row. The malicious prover "wins" if the random challenge
183    // alpha is such that:
184    // (1): C_0(i) + alpha * C_1(i) + ... + alpha^n * C_n(i) = 0
185    // This is a polynomial of degree n, so it has at most n roots. Thus the probability of this
186    // occurring for a given trace and set of constraints is n/|EF|.
187    //
188    // Currently, we do not observe data about the constraint polynomials directly. In particular
189    // a prover could take a trace and fiddle around with the AIR it claims to satisfy without
190    // changing this sample alpha.
191    //
192    // In particular this means that a malicious prover could create a custom AIR for a given trace
193    // such that equation (1) holds. However, such AIRs would need to be very specific and
194    // so such tampering should be obvious to spot. The verifier needs to check the AIR anyway to
195    // confirm that satisfying it indeed proves what the prover claims. Hence this should not be
196    // a soundness issue.
197    let alpha: SC::Challenge = challenger.sample_algebra_element();
198
199    // A domain large enough to uniquely identify the quotient polynomial.
200    // This domain must be contained in the domain over which `trace_data` is defined.
201    // Explicitly it should be equal to `gK` for some subgroup `K` contained in `H'`.
202    let quotient_domain =
203        ext_trace_domain.create_disjoint_domain(1 << (log_ext_degree + log_num_quotient_chunks));
204
205    // Return a the subset of the extended trace `ET` corresponding to the rows giving evaluations
206    // over the quotient domain.
207    //
208    // This only works if the trace domain is `gH'` and the quotient domain is `gK` for some subgroup `K` contained in `H'`.
209    // TODO: Make this explicit in `get_evaluations_on_domain` or otherwise fix this.
210    let trace_on_quotient_domain = pcs.get_evaluations_on_domain(&trace_data, 0, quotient_domain);
211    let preprocessed_on_quotient_domain =
212        preprocessed_data_ref.map(|data| pcs.get_evaluations_on_domain(data, 0, quotient_domain));
213
214    // Compute the quotient polynomial `Q(x)` by evaluating
215    //          `C(T_1(x), ..., T_w(x), T_1(hx), ..., T_w(hx), selectors(x)) / Z_H(x)`
216    // at every point in the quotient domain. The degree of `Q(x)` is `<= deg(C(x)) - N = 2N - 2` in the case
217    // where `deg(C) = 3`. (See the discussion above constraint_degree for more details.)
218    let quotient_values = quotient_values(
219        air,
220        public_values,
221        trace_domain,
222        quotient_domain,
223        &trace_on_quotient_domain,
224        preprocessed_on_quotient_domain.as_ref(),
225        alpha,
226        constraint_count,
227    );
228
229    // Due to `alpha`, evaluations of `Q` all lie in the extension field `E`.
230    // We flatten this into a matrix of `F` values by treating `E` as an `F`
231    // vector space and so separating each element of `E` into `e + 1 = [E: F]` elements of `F`.
232    //
233    // This is valid to do because our domain lies in the base field `F`. Hence we can split
234    // `Q(x)` into `e + 1` polynomials `Q_0(x), ... , Q_e(x)` each contained in `F`.
235    // such that `Q(x) = [Q_0(x), ... ,Q_e(x)]` holds for all `x` in `F`.
236    let quotient_flat = RowMajorMatrix::new_col(quotient_values).flatten_to_base();
237
238    // Currently each polynomial `Q_i(x)` is of degree `<= 2(N - 1)` and
239    // we have it's evaluations over a the coset `gK of size `2N`. Let `k` be the chosen
240    // generator of `K` which satisfies `k^2 = h`.
241    //
242    // We can split this coset into the sub-cosets `gH` and `gkH` each of size `N`.
243    // Define:  L_g(x)    = (x^N - (gk)^N)/(g^N - (gk)^N) = (x^N + g^N)/2g^N
244    //          L_{gk}(x) = (x^N - g^N)/(g^N - (gk)^N)    = -(x^N - g^N)/2g^N.
245    // Then `L_g` is equal to `1` on `gH` and `0` on `gkH` and `L_{gk}` is equal to `1` on `gkH` and `0` on `gH`.
246    //
247    // Thus we can decompose `Q_i(x) = L_{g}(x)q_{i0}(x) + L_{gk}(x)q_{i1}(x)` (Or an randomized version of this in the zk case)
248    // where `q_{i0}(x)` and `q_{i1}(x)` are polynomials of degree `<= N - 1`.
249    // Moreover the evaluations of `q_{i0}(x), q_{i1}(x)` on `gH` and `gkH` respectively are
250    // exactly the evaluations of `Q_i(x)` on `gH` and `gkH`.
251    // For each polynomial `q_{ij}`, compute the evaluation vector of `q_{ij}(x)` over `gH'`. We bit
252    // reverse the rows and hash the resulting matrix into a merkle tree.
253    //      quotient_commit contains the root of the tree
254    //      quotient_data contains the entire tree.
255    //          - quotient_data.leaves is a pair of matrices containing the `q_i0(x)` and `q_i1(x)`.
256    let (quotient_commit, quotient_data) = info_span!("commit to quotient poly chunks")
257        .in_scope(|| pcs.commit_quotient(quotient_domain, quotient_flat, num_quotient_chunks));
258    challenger.observe(quotient_commit.clone());
259
260    // If zk is enabled, we generate random extension field values of the size of the randomized trace. If `n` is the degree of the initial trace,
261    // then the randomized trace has degree `2n`. To randomize the FRI batch polynomial, we then need an extension field random polynomial of degree `2n -1`.
262    // So we can generate a random polynomial  of degree `2n`, and provide it to `open` as is.
263    // Then the method will add `(R(X) - R(z)) / (X - z)` (which is of the desired degree `2n - 1`), to the batch of polynomials.
264    // Since we need a random polynomial defined over the extension field, and the `commit` method is over the base field,
265    // we actually need to commit to `SC::CHallenge::D` base field random polynomials.
266    // This is similar to what is done for the quotient polynomials.
267    // TODO: This approach is only statistically zk. To make it perfectly zk, `R` would have to truly be an extension field polynomial.
268    let (opt_r_commit, opt_r_data) = if SC::Pcs::ZK {
269        let (r_commit, r_data) = pcs
270            .get_opt_randomization_poly_commitment([ext_trace_domain])
271            .expect("ZK is enabled, so we should have randomization commitments");
272        (Some(r_commit), Some(r_data))
273    } else {
274        (None, None)
275    };
276
277    // Combine our commitments to the trace and quotient polynomials into a single object which
278    // will be passed to the verifier.
279    let commitments = Commitments {
280        trace: trace_commit,
281        quotient_chunks: quotient_commit,
282        random: opt_r_commit.clone(),
283    };
284
285    if let Some(r_commit) = opt_r_commit {
286        challenger.observe(r_commit);
287    }
288
289    // Get an out-of-domain point to open our values at.
290    //
291    // Soundness Error:
292    // This sample will be used to check the equality: `C(X) = ZH(X)Q(X)`. If a prover is malicious
293    // and this equality is false, the probability that it is true at the point `zeta` will be
294    // deg(C(X))/|EF| = dN/|EF| where `N` is the trace length and our constraints have degree `d`.
295    //
296    // Completeness Error:
297    // If zeta happens to lie in the domain `gK`, then when opening at zeta we will run into division
298    // by zero errors. This doesn't lead to a soundness issue as the verifier will just reject in those
299    // cases but it is a completeness issue and contributes a completeness error of |gK| = 2N/|EF|.
300    let zeta: SC::Challenge = challenger.sample_algebra_element();
301    let zeta_next = trace_domain
302        .next_point(zeta)
303        .expect("domain should support next_point operation");
304
305    let is_random = opt_r_data.is_some();
306    let (opened_values, opening_proof) = info_span!("open").in_scope(|| {
307        let round0 = opt_r_data.as_ref().map(|r_data| (r_data, vec![vec![zeta]]));
308        let round1 = (&trace_data, vec![vec![zeta, zeta_next]]);
309        let round2 = (&quotient_data, vec![vec![zeta]; num_quotient_chunks]); // open every chunk at zeta
310        let round3 = preprocessed_data_ref.map(|data| (data, vec![vec![zeta, zeta_next]]));
311
312        let rounds = round0
313            .into_iter()
314            .chain([round1, round2])
315            .chain(round3)
316            .collect();
317
318        pcs.open(rounds, &mut challenger)
319    });
320    let trace_idx = SC::Pcs::TRACE_IDX;
321    let quotient_idx = SC::Pcs::QUOTIENT_IDX;
322    let trace_local = opened_values[trace_idx][0][0].clone();
323    let trace_next = opened_values[trace_idx][0][1].clone();
324    let quotient_chunks = opened_values[quotient_idx]
325        .iter()
326        .map(|v| v[0].clone())
327        .collect_vec();
328    let random = if is_random {
329        Some(opened_values[0][0][0].clone())
330    } else {
331        None
332    };
333    let (preprocessed_local, preprocessed_next) = if preprocessed_width > 0 {
334        (
335            Some(opened_values[SC::Pcs::PREPROCESSED_TRACE_IDX][0][0].clone()),
336            Some(opened_values[SC::Pcs::PREPROCESSED_TRACE_IDX][0][1].clone()),
337        )
338    } else {
339        (None, None)
340    };
341    let opened_values = OpenedValues {
342        trace_local,
343        trace_next,
344        preprocessed_local,
345        preprocessed_next,
346        quotient_chunks,
347        random,
348    };
349    Proof {
350        commitments,
351        opened_values,
352        opening_proof,
353        aux_finals: vec![],
354        degree_bits: log_ext_degree,
355    }
356}

360pub fn prove<
361    SC,
362    #[cfg(debug_assertions)] A: for<'a> Air<crate::check_constraints::DebugConstraintBuilder<'a, Val<SC>>>,
363    #[cfg(not(debug_assertions))] A,
364>(
365    config: &SC,
366    air: &A,
367    trace: RowMajorMatrix<Val<SC>>,
368    public_values: &[Val<SC>],
369) -> Proof<SC>
370where
371    SC: StarkGenericConfig,
372    A: Air<SymbolicAirBuilder<Val<SC>>> + for<'a> Air<ProverConstraintFolder<'a, SC>>,
373{
374    prove_with_preprocessed::<SC, A>(config, air, trace, public_values, None)
375}

377#[instrument(skip_all)]
378// TODO: Group some arguments to remove the `allow`?
379#[allow(clippy::too_many_arguments)]
380pub fn quotient_values<SC, A, Mat>(
381    air: &A,
382    public_values: &[Val<SC>],
383    trace_domain: Domain<SC>,
384    quotient_domain: Domain<SC>,
385    trace_on_quotient_domain: &Mat,
386    preprocessed_on_quotient_domain: Option<&Mat>,
387    alpha: SC::Challenge,
388    constraint_count: usize,
389) -> Vec<SC::Challenge>
390where
391    SC: StarkGenericConfig,
392    A: for<'a> Air<ProverConstraintFolder<'a, SC>>,
393    Mat: Matrix<Val<SC>> + Sync,
394{
395    let quotient_size = quotient_domain.size();
396    let width = trace_on_quotient_domain.width();
397    let mut sels = debug_span!("Compute Selectors")
398        .in_scope(|| trace_domain.selectors_on_coset(quotient_domain));
399
400    let qdb = log2_strict_usize(quotient_domain.size()) - log2_strict_usize(trace_domain.size());
401    let next_step = 1 << qdb;
402
403    // We take PackedVal::<SC>::WIDTH worth of values at a time from a quotient_size slice, so we need to
404    // pad with default values in the case where quotient_size is smaller than PackedVal::<SC>::WIDTH.
405    for _ in quotient_size..PackedVal::<SC>::WIDTH {
406        sels.is_first_row.push(Val::<SC>::default());
407        sels.is_last_row.push(Val::<SC>::default());
408        sels.is_transition.push(Val::<SC>::default());
409        sels.inv_vanishing.push(Val::<SC>::default());
410    }
411
412    let mut alpha_powers = alpha.powers().collect_n(constraint_count);
413    alpha_powers.reverse();
414    // alpha powers looks like Vec<EF> ~ Vec<[F; D]>
415    // It's useful to also have access to the transpose of this of form [Vec<F>; D].
416    let decomposed_alpha_powers: Vec<_> = (0..SC::Challenge::DIMENSION)
417        .map(|i| {
418            alpha_powers
419                .iter()
420                .map(|x| x.as_basis_coefficients_slice()[i])
421                .collect()
422        })
423        .collect();
424    (0..quotient_size)
425        .into_par_iter()
426        .step_by(PackedVal::<SC>::WIDTH)
427        .flat_map_iter(|i_start| {
428            let i_range = i_start..i_start + PackedVal::<SC>::WIDTH;
429
430            let is_first_row = *PackedVal::<SC>::from_slice(&sels.is_first_row[i_range.clone()]);
431            let is_last_row = *PackedVal::<SC>::from_slice(&sels.is_last_row[i_range.clone()]);
432            let is_transition = *PackedVal::<SC>::from_slice(&sels.is_transition[i_range.clone()]);
433            let inv_vanishing = *PackedVal::<SC>::from_slice(&sels.inv_vanishing[i_range]);
434
435            let main = RowMajorMatrix::new(
436                trace_on_quotient_domain.vertically_packed_row_pair(i_start, next_step),
437                width,
438            );
439
440            let preprocessed = preprocessed_on_quotient_domain.map(|preprocessed| {
441                let preprocessed_width = preprocessed.width();
442                RowMajorMatrix::new(
443                    preprocessed.vertically_packed_row_pair(i_start, next_step),
444                    preprocessed_width,
445                )
446            });
447
448            let accumulator = PackedChallenge::<SC>::ZERO;
449            let mut folder = ProverConstraintFolder {
450                main: main.as_view(),
451                preprocessed: preprocessed.as_ref().map(|m| m.as_view()),
452                public_values,
453                is_first_row,
454                is_last_row,
455                is_transition,
456                alpha_powers: &alpha_powers,
457                decomposed_alpha_powers: &decomposed_alpha_powers,
458                accumulator,
459                constraint_index: 0,
460            };
461            air.eval(&mut folder);
462
463            // quotient(x) = constraints(x) / Z_H(x)
464            let quotient = folder.accumulator * inv_vanishing;
465
466            // "Transpose" D packed base coefficients into WIDTH scalar extension coefficients.
467            (0..core::cmp::min(quotient_size, PackedVal::<SC>::WIDTH)).map(move |idx_in_packing| {
468                SC::Challenge::from_basis_coefficients_fn(|coeff_idx| {
469                    quotient.as_basis_coefficients_slice()[coeff_idx].as_slice()[idx_in_packing]
470                })
471            })
472        })
473        .collect()
474}
