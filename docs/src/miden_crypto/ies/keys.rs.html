33/// Generates seal_bytes_with_associated_data method implementation
34macro_rules! impl_seal_bytes_with_associated_data {
35    ($($variant:path => $crypto_box:ty, $ephemeral_variant:path;)*) => {
36        /// Seals the provided plaintext (represented as bytes) and associated data with this
37        /// sealing key.
38        ///
39        /// The returned message can be unsealed with the [UnsealingKey] associated with this
40        /// sealing key.
41        pub fn seal_bytes_with_associated_data<R: CryptoRng + RngCore>(
42            &self,
43            rng: &mut R,
44            plaintext: &[u8],
45            associated_data: &[u8],
46        ) -> Result<SealedMessage, IesError> {
47            match self {
48                $(
49                    $variant(key) => {
50                        let (ciphertext, ephemeral) = <$crypto_box>::seal_bytes_with_associated_data(
51                            rng,
52                            key,
53                            plaintext,
54                            associated_data,
55                        )?;
56
57                        Ok(SealedMessage {
58                            ephemeral_key: $ephemeral_variant(ephemeral),
59                            ciphertext,
60                        })
61                    }
62                )*
63            }
64        }
65    };
66}

103/// Generates unseal_bytes_with_associated_data method implementation
104macro_rules! impl_unseal_bytes_with_associated_data {
105    ($($variant:path => $crypto_box:ty, $ephemeral_variant:path;)*) => {
106        /// Unseals the provided message using this unsealing key and returns the plaintext as bytes.
107        ///
108        /// # Errors
109        /// Returns an error if:
110        /// - The message was not sealed as bytes (i.e., if it was sealed using `seal_elements()`
111        ///   or `seal_elements_with_associated_data()`)
112        /// - The scheme used to seal the message does not match this unsealing key's scheme
113        /// - Decryption or authentication fails
114        pub fn unseal_bytes_with_associated_data(
115            &self,
116            sealed_message: SealedMessage,
117            associated_data: &[u8],
118        ) -> Result<Vec<u8>, IesError> {
119            // Check scheme compatibility using constant-time comparison
120            let self_algo = self.scheme() as u8;
121            let msg_algo = sealed_message.ephemeral_key.scheme() as u8;
122
123            let compatible = self_algo == msg_algo;
124            if !compatible {
125                return Err(IesError::SchemeMismatch);
126            }
127
128            let SealedMessage { ephemeral_key, ciphertext } = sealed_message;
129
130            match (self, ephemeral_key) {
131                $(
132                    ($variant(key), $ephemeral_variant(ephemeral)) => {
133                        <$crypto_box>::unseal_bytes_with_associated_data(key, &ephemeral, &ciphertext, associated_data)
134                    }
135                )*
136                _ => Err(IesError::SchemeMismatch),
137            }
138        }
139    };
140}

394/// Ephemeral public key, part of sealed messages
395#[derive(Debug, Clone, PartialEq, Eq)]
396pub(super) enum EphemeralPublicKey {
397    K256XChaCha20Poly1305(crate::ecdh::k256::EphemeralPublicKey),
398    X25519XChaCha20Poly1305(crate::ecdh::x25519::EphemeralPublicKey),
399    K256AeadPoseidon2(crate::ecdh::k256::EphemeralPublicKey),
400    X25519AeadPoseidon2(crate::ecdh::x25519::EphemeralPublicKey),
401}
402
403impl EphemeralPublicKey {
404    /// Get scheme identifier for this ephemeral key
405    pub fn scheme(&self) -> IesScheme {
406        match self {
407            EphemeralPublicKey::K256XChaCha20Poly1305(_) => IesScheme::K256XChaCha20Poly1305,
408            EphemeralPublicKey::X25519XChaCha20Poly1305(_) => IesScheme::X25519XChaCha20Poly1305,
409            EphemeralPublicKey::K256AeadPoseidon2(_) => IesScheme::K256AeadPoseidon2,
410            EphemeralPublicKey::X25519AeadPoseidon2(_) => IesScheme::X25519AeadPoseidon2,
411        }
412    }
413
414    /// Serialize to bytes
415    pub fn to_bytes(&self) -> Vec<u8> {
416        match self {
417            EphemeralPublicKey::K256XChaCha20Poly1305(key) => key.to_bytes(),
418            EphemeralPublicKey::X25519XChaCha20Poly1305(key) => key.to_bytes(),
419            EphemeralPublicKey::K256AeadPoseidon2(key) => key.to_bytes(),
420            EphemeralPublicKey::X25519AeadPoseidon2(key) => key.to_bytes(),
421        }
422    }
423
424    /// Deserialize from bytes with explicit scheme
425    pub fn from_bytes(scheme: IesScheme, bytes: &[u8]) -> Result<Self, IesError> {
426        let expected_len = match scheme {
427            IesScheme::K256XChaCha20Poly1305 | IesScheme::K256AeadPoseidon2 => {
428                K256_PUBLIC_KEY_BYTES
429            },
430            IesScheme::X25519XChaCha20Poly1305 | IesScheme::X25519AeadPoseidon2 => {
431                X25519_PUBLIC_KEY_BYTES
432            },
433        };
434
435        if bytes.len() != expected_len {
436            return Err(IesError::EphemeralPublicKeyDeserializationFailed);
437        }
438
439        match scheme {
440            IesScheme::K256XChaCha20Poly1305 => {
441                let key =
442                    <K256 as KeyAgreementScheme>::EphemeralPublicKey::read_from_bytes_with_budget(
443                        bytes,
444                        expected_len,
445                    )
446                    .map_err(|_| IesError::EphemeralPublicKeyDeserializationFailed)?;
447                Ok(EphemeralPublicKey::K256XChaCha20Poly1305(key))
448            },
449            IesScheme::K256AeadPoseidon2 => {
450                let key =
451                    <K256 as KeyAgreementScheme>::EphemeralPublicKey::read_from_bytes_with_budget(
452                        bytes,
453                        expected_len,
454                    )
455                    .map_err(|_| IesError::EphemeralPublicKeyDeserializationFailed)?;
456                Ok(EphemeralPublicKey::K256AeadPoseidon2(key))
457            },
458            IesScheme::X25519XChaCha20Poly1305 => {
459                let key =
460                    <X25519 as KeyAgreementScheme>::EphemeralPublicKey::read_from_bytes_with_budget(
461                        bytes,
462                        expected_len,
463                    )
464                        .map_err(|_| IesError::EphemeralPublicKeyDeserializationFailed)?;
465                Ok(EphemeralPublicKey::X25519XChaCha20Poly1305(key))
466            },
467            IesScheme::X25519AeadPoseidon2 => {
468                let key =
469                    <X25519 as KeyAgreementScheme>::EphemeralPublicKey::read_from_bytes_with_budget(
470                        bytes,
471                        expected_len,
472                    )
473                        .map_err(|_| IesError::EphemeralPublicKeyDeserializationFailed)?;
474                Ok(EphemeralPublicKey::X25519AeadPoseidon2(key))
475            },
476        }
477    }
