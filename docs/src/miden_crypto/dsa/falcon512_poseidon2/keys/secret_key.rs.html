35/// Represents the secret key for Falcon DSA.
36///
37/// The secret key is a quadruple [[g, -f], [G, -F]] of polynomials with integer coefficients. Each
38/// polynomial is of degree at most N = 512 and computations with these polynomials is done modulo
39/// the monic irreducible polynomial ϕ = x^N + 1. The secret key is a basis for a lattice and has
40/// the property of being short with respect to a certain norm and an upper bound appropriate for
41/// a given security parameter. The public key on the other hand is another basis for the same
42/// lattice and can be described by a single polynomial h with integer coefficients modulo ϕ.
43/// The two keys are related by the following relation:
44///
45/// 1. h = g /f [mod ϕ][mod p]
46/// 2. f.G - g.F = p [mod ϕ]
47///
48/// where p = 12289 is the Falcon prime. Equation 2 is called the NTRU equation.
49/// The secret key is generated by first sampling a random pair (f, g) of polynomials using
50/// an appropriate distribution that yields short but not too short polynomials with integer
51/// coefficients modulo ϕ. The NTRU equation is then used to find a matching pair (F, G).
52/// The public key is then derived from the secret key using equation 1.
53///
54/// To allow for fast signature generation, the secret key is pre-processed into a more suitable
55/// form, called the LDL tree, and this allows for fast sampling of short vectors in the lattice
56/// using Fast Fourier sampling during signature generation (ffSampling algorithm 11 in [1]).
57///
58/// [1]: https://falcon-sign.info/falcon.pdf
59#[derive(Clone, SilentDebug, SilentDisplay)]
60pub struct SecretKey {
61    secret_key: ShortLatticeBasis,
62    tree: LdlTree,
63}
64

100    /// Given a short basis [[g, -f], [G, -F]], computes the normalized LDL tree i.e., Falcon tree.
101    pub(crate) fn from_short_lattice_basis(basis: ShortLatticeBasis) -> SecretKey {
102        // FFT each polynomial of the short basis.
103        let basis_fft = to_complex_fft(&basis);
104        // compute the Gram matrix.
105        let gram_fft = gram(basis_fft);
106        // construct the LDL tree of the Gram matrix.
107        let mut tree = ffldl(gram_fft);
108        // normalize the leaves of the LDL tree.
109        normalize_tree(&mut tree, SIGMA);
110        Self { secret_key: basis, tree }
111    }

134    /// Signs a message with this secret key.
135    pub fn sign(&self, message: crate::Word) -> Signature {
136        use rand::SeedableRng;
137        use rand_chacha::ChaCha20Rng;
138
139        let mut seed = self.generate_seed(&message);
140        let mut rng = ChaCha20Rng::from_seed(seed);
141        let signature = self.sign_with_rng(message, &mut rng);
142
143        // Zeroize the seed to prevent leakage
144        seed.zeroize();
145
146        signature
147    }

149    /// Signs a message with the secret key relying on the provided randomness generator.
150    pub fn sign_with_rng<R: Rng>(&self, message: Word, rng: &mut R) -> Signature {
151        let nonce = Nonce::deterministic();
152
153        let h = self.compute_pub_key_poly();
154        let c = hash_to_point_poseidon2(message, &nonce);
155        let s2 = self.sign_helper(c, rng);
156
157        Signature::new(nonce, h, s2)
158    }

192    /// Derives the public key corresponding to this secret key using h = g /f [mod ϕ][mod p].
193    fn compute_pub_key_poly(&self) -> PublicKey {
194        let g: Polynomial<FalconFelt> = self.secret_key[0].clone().into();
195        let g_fft = g.fft();
196        let minus_f: Polynomial<FalconFelt> = self.secret_key[1].clone().into();
197        let f = -minus_f;
198        let f_fft = f.fft();
199        let h_fft = g_fft.hadamard_div(&f_fft);
200        h_fft.ifft().into()
201    }
202
203    /// Signs a message polynomial with the secret key.
204    ///
205    /// Takes a randomness generator implementing `Rng` and message polynomial representing `c`
206    /// the hash-to-point of the message to be signed. It outputs a signature polynomial `s2`.
207    fn sign_helper<R: Rng>(&self, c: Polynomial<FalconFelt>, rng: &mut R) -> SignaturePoly {
208        let one_over_q = 1.0 / (MODULUS as f64);
209        let c_over_q_fft = c.map(|cc| Complex::new(one_over_q * cc.value() as f64, 0.0)).fft();
210
211        // B = [[FFT(g), -FFT(f)], [FFT(G), -FFT(F)]]
212        let [g_fft, minus_f_fft, big_g_fft, minus_big_f_fft] = to_complex_fft(&self.secret_key);
213        let t0 = c_over_q_fft.hadamard_mul(&minus_big_f_fft);
214        let t1 = -c_over_q_fft.hadamard_mul(&minus_f_fft);
215
216        loop {
217            let bold_s = loop {
218                let z = ffsampling(&(t0.clone(), t1.clone()), &self.tree, rng);
219                let t0_min_z0 = t0.clone() - z.0;
220                let t1_min_z1 = t1.clone() - z.1;
221
222                // s = (t-z) * B
223                let s0 = t0_min_z0.hadamard_mul(&g_fft) + t1_min_z1.hadamard_mul(&big_g_fft);
224                let s1 =
225                    t0_min_z0.hadamard_mul(&minus_f_fft) + t1_min_z1.hadamard_mul(&minus_big_f_fft);
226
227                // compute the norm of (s0||s1) and note that they are in FFT representation
228                let length_squared: f64 =
229                    (s0.coefficients.iter().map(|a| (a * a.conj()).re).sum::<f64>()
230                        + s1.coefficients.iter().map(|a| (a * a.conj()).re).sum::<f64>())
231                        / (N as f64);
232
233                if length_squared > (SIG_L2_BOUND as f64) {
234                    continue;
235                }
236
237                break [-s0, s1];
238            };
239
240            let s2 = bold_s[1].ifft();
241            let s2_coef: [i16; N] = s2
242                .coefficients
243                .iter()
244                .map(|a| a.re.round() as i16)
245                .collect::<Vec<i16>>()
246                .try_into()
247                .expect("The number of coefficients should be equal to N");
248
249            if let Ok(s2) = SignaturePoly::try_from(&s2_coef) {
250                return s2;
251            }
252        }
253    }
254
255    /// Deterministically generates a seed for seeding the PRNG used in the trapdoor sampling
256    /// algorithm used during signature generation.
257    ///
258    /// This uses the argument described in [RFC 6979](https://datatracker.ietf.org/doc/html/rfc6979#section-3.5)
259    /// § 3.5 where the concatenation of the private key and the hashed message, i.e., sk || H(m),
260    /// is used in order to construct the initial seed of a PRNG. See also [1].
261    ///
262    ///
263    /// Note that we hash in also a `log_2(N)` where `N = 512` in order to domain separate between
264    /// different versions of the Falcon DSA, see [1] Section 3.4.1.
265    ///
266    /// [1]: https://github.com/algorand/falcon/blob/main/falcon-det.pdf
267    fn generate_seed(&self, message: &Word) -> [u8; 32] {
268        let mut buffer = Vec::with_capacity(1 + SK_LEN + Word::SERIALIZED_SIZE);
269        buffer.push(LOG_N);
270        buffer.extend_from_slice(&self.to_bytes());
271        buffer.extend_from_slice(&message.to_bytes());
272
273        let digest = Blake3_256::hash(&buffer);
274
275        // Zeroize the buffer as it contains secret key material
276        buffer.zeroize();
277
278        digest.into()
279    }

294impl Serializable for SecretKey {
295    fn write_into<W: ByteWriter>(&self, target: &mut W) {
296        let basis = &self.secret_key;
297
298        // header
299        let n = basis[0].coefficients.len();
300        let l = n.checked_ilog2().unwrap() as u8;
301        let header: u8 = (5 << 4) | l;
302
303        let neg_f = &basis[1];
304        let g = &basis[0];
305        let neg_big_f = &basis[3];
306
307        let mut buffer = Vec::with_capacity(1281);
308        buffer.push(header);
309
310        let mut f_i8: Vec<i8> = neg_f
311            .coefficients
312            .iter()
313            .map(|&a| FalconFelt::new(-a).balanced_value() as i8)
314            .collect();
315        let f_i8_encoded = encode_i8(&f_i8, WIDTH_SMALL_POLY_COEFFICIENT).unwrap();
316        buffer.extend_from_slice(&f_i8_encoded);
317        f_i8.zeroize();
318
319        let mut g_i8: Vec<i8> = g
320            .coefficients
321            .iter()
322            .map(|&a| FalconFelt::new(a).balanced_value() as i8)
323            .collect();
324        let g_i8_encoded = encode_i8(&g_i8, WIDTH_SMALL_POLY_COEFFICIENT).unwrap();
325        buffer.extend_from_slice(&g_i8_encoded);
326        g_i8.zeroize();
327
328        let mut big_f_i8: Vec<i8> = neg_big_f
329            .coefficients
330            .iter()
331            .map(|&a| FalconFelt::new(-a).balanced_value() as i8)
332            .collect();
333        let big_f_i8_encoded = encode_i8(&big_f_i8, WIDTH_BIG_POLY_COEFFICIENT).unwrap();
334        buffer.extend_from_slice(&big_f_i8_encoded);
335        big_f_i8.zeroize();
336
337        target.write_bytes(&buffer);
338        // Note: buffer is not zeroized here as it's being passed to write_bytes which consumes it
339        // The caller should ensure proper handling of the written bytes
340    }
341}

343impl Deserializable for SecretKey {
344    fn read_from<R: ByteReader>(source: &mut R) -> Result<Self, DeserializationError> {
345        let byte_vector: [u8; SK_LEN] = source.read_array()?;
346
347        // read fields
348        let header = byte_vector[0];
349
350        // check fixed bits in header
351        if (header >> 4) != 5 {
352            return Err(DeserializationError::InvalidValue("Invalid header format".to_string()));
353        }
354
355        // check log n
356        let logn = (header & 15) as usize;
357        let n = 1 << logn;
358
359        // match against const variant generic parameter
360        if n != N {
361            return Err(DeserializationError::InvalidValue(
362                "Unsupported Falcon DSA variant".to_string(),
363            ));
364        }
365
366        let chunk_size_f = ((n * WIDTH_SMALL_POLY_COEFFICIENT) + 7) >> 3;
367        let chunk_size_g = ((n * WIDTH_SMALL_POLY_COEFFICIENT) + 7) >> 3;
368        let chunk_size_big_f = ((n * WIDTH_BIG_POLY_COEFFICIENT) + 7) >> 3;
369
370        let f = decode_i8(&byte_vector[1..chunk_size_f + 1], WIDTH_SMALL_POLY_COEFFICIENT).ok_or(
371            DeserializationError::InvalidValue("Failed to decode f coefficients".to_string()),
372        )?;
373        let g = decode_i8(
374            &byte_vector[chunk_size_f + 1..(chunk_size_f + chunk_size_g + 1)],
375            WIDTH_SMALL_POLY_COEFFICIENT,
376        )
377        .unwrap();
378        let big_f = decode_i8(
379            &byte_vector[(chunk_size_f + chunk_size_g + 1)
380                ..(chunk_size_f + chunk_size_g + chunk_size_big_f + 1)],
381            WIDTH_BIG_POLY_COEFFICIENT,
382        )
383        .unwrap();
384
385        let f = Polynomial::new(f.iter().map(|&c| FalconFelt::new(c.into())).collect());
386        let g = Polynomial::new(g.iter().map(|&c| FalconFelt::new(c.into())).collect());
387        let big_f = Polynomial::new(big_f.iter().map(|&c| FalconFelt::new(c.into())).collect());
388
389        // big_g * f - g * big_f = p (mod X^n + 1)
390        let big_g = g.fft().hadamard_div(&f.fft()).hadamard_mul(&big_f.fft()).ifft();
391        let basis = [
392            g.map(|f| f.balanced_value()),
393            -f.map(|f| f.balanced_value()),
394            big_g.map(|f| f.balanced_value()),
395            -big_f.map(|f| f.balanced_value()),
396        ];
397        Ok(Self::from_short_lattice_basis(basis))
398    }
399}

404/// Computes the complex FFT of the secret key polynomials.
405fn to_complex_fft(basis: &[Polynomial<i16>; 4]) -> [Polynomial<Complex<f64>>; 4] {
406    let [g, f, big_g, big_f] = basis.clone();
407    let g_fft = g.map(|cc| Complex64::new(*cc as f64, 0.0)).fft();
408    let minus_f_fft = f.map(|cc| -Complex64::new(*cc as f64, 0.0)).fft();
409    let big_g_fft = big_g.map(|cc| Complex64::new(*cc as f64, 0.0)).fft();
410    let minus_big_f_fft = big_f.map(|cc| -Complex64::new(*cc as f64, 0.0)).fft();
411    [g_fft, minus_f_fft, big_g_fft, minus_big_f_fft]
412}

414/// Encodes a sequence of signed integers such that each integer x satisfies |x| < 2^(bits-1)
415/// for a given parameter bits. bits can take either the value 6 or 8.
416pub fn encode_i8(x: &[i8], bits: usize) -> Option<Vec<u8>> {
417    let maxv = (1 << (bits - 1)) - 1_usize;
418    let maxv = maxv as i8;
419    let minv = -maxv;
420
421    for &c in x {
422        if c > maxv || c < minv {
423            return None;
424        }
425    }
426
427    let out_len = ((N * bits) + 7) >> 3;
428    let mut buf = vec![0_u8; out_len];
429
430    let mut acc = 0_u32;
431    let mut acc_len = 0;
432    let mask = ((1_u16 << bits) - 1) as u8;
433
434    let mut input_pos = 0;
435    for &c in x {
436        acc = (acc << bits) | (c as u8 & mask) as u32;
437        acc_len += bits;
438        while acc_len >= 8 {
439            acc_len -= 8;
440            buf[input_pos] = (acc >> acc_len) as u8;
441            input_pos += 1;
442        }
443    }
444    if acc_len > 0 {
445        buf[input_pos] = (acc >> (8 - acc_len)) as u8;
446    }
447
448    Some(buf)
449}

451/// Decodes a sequence of bytes into a sequence of signed integers such that each integer x
452/// satisfies |x| < 2^(bits-1) for a given parameter bits. bits can take either the value 6 or 8.
453pub fn decode_i8(buf: &[u8], bits: usize) -> Option<Vec<i8>> {
454    let mut x = [0_i8; N];
455
456    let mut i = 0;
457    let mut j = 0;
458    let mut acc = 0_u32;
459    let mut acc_len = 0;
460    let mask = (1_u32 << bits) - 1;
461    let a = (1 << bits) as u8;
462    let b = ((1 << (bits - 1)) - 1) as u8;
463
464    while i < N {
465        acc = (acc << 8) | (buf[j] as u32);
466        j += 1;
467        acc_len += 8;
468
469        while acc_len >= bits && i < N {
470            acc_len -= bits;
471            let w = (acc >> acc_len) & mask;
472
473            let w = w as u8;
474
475            let z = if w > b { w as i8 - a as i8 } else { w as i8 };
476
477            x[i] = z;
478            i += 1;
479        }
480    }
481
482    if (acc & ((1u32 << acc_len) - 1)) == 0 {
483        Some(x.to_vec())
484    } else {
485        None
486    }
