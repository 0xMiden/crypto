12/// Implements Cyclotomic FFT without bitreversing the outputs, and using precomputed powers of the
13/// 2n-th primitive root of unity.

16    fn fft_inplace(&mut self);

36    Self: Sized
37        + Copy
38        + One
39        + Zero
40        + Add<Output = Self>
41        + Sub<Output = Self>
42        + Mul<Output = Self>
43        + MulAssign

46    /// Gets the inverse of 2^n.
47    #[allow(dead_code)]
48    fn power_of_two_inverse(n: usize) -> Self {
49        let mut a = Self::one() + Self::one();
50        for _ in 0..n {
51            a *= a;
52        }
53        Self::inverse_or_zero(a)
54    }
55
56    /// Gets a primitive nth (with n a power of 2) root of unity.
57    #[allow(dead_code)]
58    fn primitive_root_of_unity(n: usize) -> Self;
59
60    /// Computes the integer whose n-bit binary expansion is the reverse of that of the argument.
61    fn bitreverse_index(arg: usize, n: usize) -> usize {
62        assert!(n > 0);
63        assert_eq!(n & (n - 1), 0);
64        let mut rev = 0;
65        let mut m = n >> 1;
66        let mut k = 1;
67        while m > 0 {
68            rev |= (((arg & m) != 0) as usize) * k;
69            k <<= 1;
70            m >>= 1;
71        }
72        rev
73    }
74
75    /// Computes the first n powers of the 2nd root of unity, and put them in bit-reversed order.
76    #[allow(dead_code)]
77    fn bitreversed_powers(n: usize) -> Vec<Self> {
78        let psi = Self::primitive_root_of_unity(2 * n);
79        let mut array = vec![Self::zero(); n];
80        let mut alpha = Self::one();
81        for a in array.iter_mut() {
82            *a = alpha;
83            alpha *= psi;
84        }
85        Self::bitreverse_array(&mut array);
86        array
87    }
88
89    /// Computes the first n powers of the 2nd root of unity, invert them, and put them in
90    /// bit-reversed order.
91    #[allow(dead_code)]
92    fn bitreversed_powers_inverse(n: usize) -> Vec<Self> {
93        let psi = Self::primitive_root_of_unity(2 * n).inverse_or_zero();
94        let mut array = vec![Self::zero(); n];
95        let mut alpha = Self::one();
96        for a in array.iter_mut() {
97            *a = alpha;
98            alpha *= psi;
99        }
100        Self::bitreverse_array(&mut array);
101        array
102    }

104    /// Reorders the given elements in the array by reversing the binary expansions of their
105    /// indices.
106    fn bitreverse_array<T>(array: &mut [T]) {
107        let n = array.len();
108        for i in 0..n {
109            let j = Self::bitreverse_index(i, n);
110            if i < j {
111                array.swap(i, j);
112            }
113        }
114    }

116    /// Computes the evaluations of the polynomial on the roots of the polynomial X^n + 1 using a
117    /// fast Fourier transform. Algorithm 1 from https://eprint.iacr.org/2016/504.pdf.
118    ///
119    /// Arguments:
120    ///
121    ///  - a : &mut [Self] (a reference to) a mutable array of field elements which is to be
122    ///    transformed under the FFT. The transformation happens in- place.
123    ///
124    ///  - psi_rev: &[Self] (a reference to) an array of powers of psi, from 0 to n-1, but ordered
125    ///    by bit-reversed index. Here psi is a primitive root of order 2n. You can use
126    ///    `Self::bitreversed_powers(psi, n)` for this purpose, but this trait implementation is not
127    ///    const. For the performance benefit you want a precompiled array, which you can get if you
128    ///    can get by implementing the same method and marking it "const".
129    fn fft(a: &mut [Self], psi_rev: &[Self]) {
130        let n = a.len();
131        let mut t = n;
132        let mut m = 1;
133        while m < n {
134            t >>= 1;
135            for i in 0..m {
136                let j1 = 2 * i * t;
137                let j2 = j1 + t - 1;
138                let s = psi_rev[m + i];
139                for j in j1..=j2 {
140                    let u = a[j];
141                    let v = a[j + t] * s;
142                    a[j] = u + v;
143                    a[j + t] = u - v;
144                }
145            }
146            m <<= 1;
147        }
148    }
149
150    /// Computes the coefficients of the polynomial with the given evaluations on the roots of
151    /// X^n + 1 using an inverse fast Fourier transform.
152    /// Algorithm 2 from https://eprint.iacr.org/2016/504.pdf.
153    ///
154    /// Arguments:
155    ///
156    ///  - a : &mut [Self] (a reference to) a mutable array of field elements which is to be
157    ///    transformed under the IFFT. The transformation happens in- place.
158    ///
159    ///  - psi_inv_rev: &[Self] (a reference to) an array of powers of psi^-1, from 0 to n-1, but
160    ///    ordered by bit-reversed index. Here psi is a primitive root of order 2n. You can use
161    ///    `Self::bitreversed_powers(Self::inverse_or_zero(psi), n)` for this purpose, but this
162    ///    trait implementation is not const. For the performance benefit you want a precompiled
163    ///    array, which you can get if you can get by implementing the same methods and marking them
164    ///    "const".
165    fn ifft(a: &mut [Self], psi_inv_rev: &[Self], ninv: Self) {
166        let n = a.len();
167        let mut t = 1;
168        let mut m = n;
169        while m > 1 {
170            let h = m / 2;
171            let mut j1 = 0;
172            for i in 0..h {
173                let j2 = j1 + t - 1;
174                let s = psi_inv_rev[h + i];
175                for j in j1..=j2 {
176                    let u = a[j];
177                    let v = a[j + t];
178                    a[j] = u + v;
179                    a[j + t] = (u - v) * s;
180                }
181                j1 += 2 * t;
182            }
183            t <<= 1;
184            m >>= 1;
185        }
186        for ai in a.iter_mut() {
187            *ai *= ninv;
188        }
189    }
190
191    fn split_fft(f: &[Self], psi_inv_rev: &[Self]) -> (Vec<Self>, Vec<Self>) {
192        let n_over_2 = f.len() / 2;
193        let mut f0 = vec![Self::zero(); n_over_2];
194        let mut f1 = vec![Self::zero(); n_over_2];
195        let two_inv = (Self::one() + Self::one()).inverse_or_zero();
196        for i in 0..n_over_2 {
197            let two_i = i * 2;
198            let two_zeta_inv = two_inv * psi_inv_rev[n_over_2 + i];
199            f0[i] = two_inv * (f[two_i] + f[two_i + 1]);
200            f1[i] = two_zeta_inv * (f[two_i] - f[two_i + 1]);
201        }
202        (f0, f1)
203    }
204
205    fn merge_fft(f0: &[Self], f1: &[Self], psi_rev: &[Self]) -> Vec<Self> {
206        let n_over_2 = f0.len();
207        let n = 2 * n_over_2;
208        let mut f = vec![Self::zero(); n];
209        for i in 0..n_over_2 {
210            let two_i = i * 2;
211            f[two_i] = f0[i] + psi_rev[n_over_2 + i] * f1[i];
212            f[two_i + 1] = f0[i] - psi_rev[n_over_2 + i] * f1[i];
213        }
214        f
215    }
216}

218impl CyclotomicFourier for Complex64 {
219    fn primitive_root_of_unity(n: usize) -> Self {
220        let angle = 2. * PI / (n as f64);
221        Complex64::new(f64::cos(angle), f64::sin(angle))
222    }
223
224    /// Custom implementation of CyclotomicFourier::bitreversed_powers for
225    /// better precision.
226    fn bitreversed_powers(n: usize) -> Vec<Self> {
227        let mut array = vec![Self::zero(); n];
228        let half_circle = PI;
229        for (i, a) in array.iter_mut().enumerate() {
230            let angle = (i as f64) * half_circle / (n as f64);
231            *a = Self::new(f64::cos(angle), f64::sin(angle));
232        }
233        Self::bitreverse_array(&mut array);
234        array
235    }
236
237    /// Custom implementation of CyclotomicFourier::bitreversed_powers_inverse
238    /// for better precision.
239    fn bitreversed_powers_inverse(n: usize) -> Vec<Self> {
240        let mut array = vec![Self::zero(); n];
241        let half_circle = PI;
242        for (i, a) in array.iter_mut().enumerate() {
243            let angle = (i as f64) * half_circle / (n as f64);
244            *a = Self::new(f64::cos(angle), -f64::sin(angle));
245        }
246        Self::bitreverse_array(&mut array);
247        array
248    }
249}

251impl FastFft for Polynomial<Complex64> {
252    type Field = Complex64;
253    fn fft_inplace(&mut self) {
254        let n = self.coefficients.len();
255        debug_assert!(
256            (1..=512).contains(&n),
257            "unsupported: n = {n} not a power of 2 or larger than 512"
258        );
259        Complex64::fft(&mut self.coefficients, &COMPLEX_BITREVERSED_POWERS);
260    }
261
262    fn ifft_inplace(&mut self) {
263        let n = self.coefficients.len();
264        debug_assert!(
265            (1..=512).contains(&n),
266            "unsupported: n = {n} not a power of 2 or larger than 512"
267        );
268        let psi_inv_rev: Vec<Complex64> =
269            COMPLEX_BITREVERSED_POWERS.iter().map(|c| Complex64::new(c.re, -c.im)).collect();
270        let ninv = Complex64::new(1.0 / (n as f64), 0.0);
271        Complex64::ifft(&mut self.coefficients, &psi_inv_rev, ninv);
272    }
273
274    fn merge_fft(a: &Self, b: &Self) -> Self {
275        let n = a.coefficients.len();
276        debug_assert!(
277            (1..=512).contains(&n),
278            "unsupported: n = {n} not a power of 2 or larger than 512"
279        );
280        Self {
281            coefficients: Self::Field::merge_fft(
282                &a.coefficients,
283                &b.coefficients,
284                &COMPLEX_BITREVERSED_POWERS,
285            ),
286        }
287    }
288
289    fn split_fft(&self) -> (Self, Self) {
290        let n = self.coefficients.len();
291        debug_assert!(
292            (1..=512).contains(&n),
293            "unsupported: n = {n} not a power of 2 or larger than 512"
294        );
295        let psi_inv_rev: Vec<Complex64> =
296            COMPLEX_BITREVERSED_POWERS.iter().map(|c| Complex64::new(c.re, -c.im)).collect();
297        let (a, b) = Self::Field::split_fft(&self.coefficients, &psi_inv_rev);
298        (Self { coefficients: a }, Self { coefficients: b })
299    }
300}
