3/// Samples an integer from {0, ..., 18} according to the distribution χ, which is close to
4/// the half-Gaussian distribution on the natural numbers with mean 0 and standard deviation
5/// equal to sigma_max.
6fn base_sampler(bytes: [u8; 9]) -> i16 {
7    const RCDT: [u128; 18] = [
8        3024686241123004913666,
9        1564742784480091954050,
10        636254429462080897535,
11        199560484645026482916,
12        47667343854657281903,
13        8595902006365044063,
14        1163297957344668388,
15        117656387352093658,
16        8867391802663976,
17        496969357462633,
18        20680885154299,
19        638331848991,
20        14602316184,
21        247426747,
22        3104126,
23        28824,
24        198,
25        1,
26    ];
27
28    let mut bytes = bytes.to_vec();
29    bytes.extend_from_slice(&[0u8; 7]);
30    bytes.reverse();
31    let u = u128::from_be_bytes(bytes.try_into().expect("should have length 16"));
32    RCDT.into_iter().filter(|r| u < *r).count() as i16
33}

35/// Computes an integer approximation of 2^63 * ccs * exp(-x).
36fn approx_exp(x: f64, ccs: f64) -> u64 {
37    // The constants C are used to approximate exp(-x); these
38    // constants are taken from FACCT (up to a scaling factor
39    // of 2^63):
40    //   https://eprint.iacr.org/2018/1234
41    //   https://github.com/raykzhao/gaussian
42    const C: [u64; 13] = [
43        0x00000004741183a3u64,
44        0x00000036548cfc06u64,
45        0x0000024fdcbf140au64,
46        0x0000171d939de045u64,
47        0x0000d00cf58f6f84u64,
48        0x000680681cf796e3u64,
49        0x002d82d8305b0feau64,
50        0x011111110e066fd0u64,
51        0x0555555555070f00u64,
52        0x155555555581ff00u64,
53        0x400000000002b400u64,
54        0x7fffffffffff4800u64,
55        0x8000000000000000u64,
56    ];
57
58    let mut z: u64;
59    let mut y: u64;
60    let twoe63 = 1u64 << 63;
61
62    y = C[0];
63    z = f64::floor(x * (twoe63 as f64)) as u64;
64    for cu in C.iter().skip(1) {
65        let zy = (z as u128) * (y as u128);
66        y = cu - ((zy >> 63) as u64);
67    }
68
69    z = f64::floor((twoe63 as f64) * ccs) as u64;
70
71    (((z as u128) * (y as u128)) >> 63) as u64
72}

74/// A random bool that is true with probability ≈ ccs · exp(-x).
75fn ber_exp<R: Rng>(x: f64, ccs: f64, rng: &mut R) -> bool {
76    const LN2: f64 = core::f64::consts::LN_2;
77    const ILN2: f64 = 1.0 / LN2;
78    let s = f64::floor(x * ILN2);
79    let r = x - s * LN2;
80    let s = (s as u64).min(63);
81    let z = ((approx_exp(r, ccs) << 1) - 1) >> s;
82
83    let mut w = 0_i32;
84    for i in (0..=56).rev().step_by(8) {
85        let mut dest = [0_u8; 1];
86        rng.fill_bytes(&mut dest);
87        let p = u8::from_be_bytes(dest);
88        w = (p as i32) - (z >> i & 0xff) as i32;
89        if w != 0 {
90            break;
91        }
92    }
93    w < 0
94}

96/// Samples an integer from the Gaussian distribution with given mean (mu) and standard deviation
97/// (sigma).
98pub(crate) fn sampler_z<R: Rng>(mu: f64, sigma: f64, sigma_min: f64, rng: &mut R) -> i16 {
99    const SIGMA_MAX: f64 = 1.8205;
100    const INV_2SIGMA_MAX_SQ: f64 = 1f64 / (2f64 * SIGMA_MAX * SIGMA_MAX);
101    let isigma = 1f64 / sigma;
102    let dss = 0.5f64 * isigma * isigma;
103    let s = f64::floor(mu);
104    let r = mu - s;
105    let ccs = sigma_min * isigma;
106    loop {
107        let mut dest = [0_u8; 9];
108        rng.fill_bytes(&mut dest);
109        let z0 = base_sampler(dest);
110
111        let mut dest = [0_u8; 1];
112        rng.fill_bytes(&mut dest);
113        let random_byte: u8 = dest[0];
114
115        // x = ((z-r)^2)/(2*sigma^2) - ((z-b)^2)/(2*sigma0^2)
116        let b = (random_byte & 1) as i16;
117        let z = b + (2 * b - 1) * z0;
118        let zf_min_r = (z as f64) - r;
119        let x = zf_min_r * zf_min_r * dss - (z0 * z0) as f64 * INV_2SIGMA_MAX_SQ;
120
121        if ber_exp(x, ccs, rng) {
122            return z + (s as i16);
123        }
124    }
125}

132    fn test_approx_exp() {
133        let precision = 1u64 << 14;
134        // known answers were generated with the following sage script:
135        //```sage
136        // num_samples = 10
137        // precision = 200
138        // R = Reals(precision)
139        //
140        // print(f"let kats : [(f64, f64, u64);{num_samples}] = [")
141        // for i in range(num_samples):
142        //     x = RDF.random_element(0.0, 0.693147180559945)
143        //     ccs = RDF.random_element(0.0, 1.0)
144        //     res = round(2^63 * R(ccs) * exp(R(-x)))
145        //     print(f"({x}, {ccs}, {res}),")
146        // print("];")
147        // ```
148        let kats: [(f64, f64, u64); 10] = [
149            (0.2314993926072656, 0.8148006314615972, 5962140072160879737),
150            (0.2648875572812225, 0.12769669655309035, 903712282351034505),
151            (0.11251957513682391, 0.9264611470305881, 7635725498677341553),
152            (0.04353439307256617, 0.5306497137523327, 4685877322232397936),
153            (0.41834495299784347, 0.879438856118578, 5338392138535350986),
154            (0.32579398973228557, 0.16513412873289002, 1099603299296456803),
155            (0.5939508073919817, 0.029776019144967303, 151637565622779016),
156            (0.2932367999399056, 0.37123847662857923, 2553827649386670452),
157            (0.5005699297417507, 0.31447208863888976, 1758235618083658825),

169        }
