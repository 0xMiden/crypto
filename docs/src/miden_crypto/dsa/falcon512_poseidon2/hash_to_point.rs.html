11/// Returns a polynomial in Z_p[x]/(phi) representing the hash of the provided message and
12/// nonce using Poseidon2.
13///
14/// Note that, in contrast to the SHAKE256-based reference implementation, this implementation
15/// does not use rejection sampling but instead uses one of the variants listed in the specification
16/// [1]. This variant omits the conditional check in the rejection sampling step at the cost of
17/// having to extract 64 bits, instead of 16 bits, of pseudo-randomness. This makes
18/// the implementation simpler and constant-time at the cost of a higher number of extracted
19/// pseudo-random bits per call to the hash-to-point algorithm.
20///
21/// [1]: https://falcon-sign.info/falcon.pdf
22pub fn hash_to_point_poseidon2(message: Word, nonce: &Nonce) -> Polynomial<FalconFelt> {
23    let mut state = [ZERO; Poseidon2::STATE_WIDTH];
24
25    // absorb the nonce into the state
26    let nonce_elements = nonce.to_elements();
27    for (&n, s) in nonce_elements.iter().zip(state[Poseidon2::RATE_RANGE].iter_mut()) {
28        *s = n;
29    }
30    Poseidon2::apply_permutation(&mut state);
31
32    // absorb message into the state
33    for (&m, s) in message.iter().zip(state[Poseidon2::RATE_RANGE].iter_mut()) {
34        *s = m;
35    }
36
37    // squeeze the coefficients of the polynomial
38    let mut coefficients: Vec<FalconFelt> = Vec::with_capacity(N);
39    for _ in 0..64 {
40        //
41        // Note that `FalconFelt::new((a.as_canonical_u64() % MODULUS as u64) as i16)` will
42        // create a bias as we are mapping $2^64 - 2^31 + 1$ elements to $12289$ elements
43        // and it must not be uniform. A statistical analysis can be applied here to show
44        // that this is still fine: the output distribution is computational IND from
45        // uniform.
46        Poseidon2::apply_permutation(&mut state);
47        state[Poseidon2::RATE_RANGE]
48            .iter()
49            .for_each(|value| coefficients.push(felt_to_falcon_felt(*value)));
50    }
51
52    Polynomial::new(coefficients)
53}

55/// Returns a polynomial in Z_p[x]/(phi) representing the hash of the provided message and
56/// nonce using SHAKE256. This is the hash-to-point algorithm used in the reference implementation.
57#[cfg(test)]
58pub fn hash_to_point_shake256(message: &[u8], nonce: &Nonce) -> Polynomial<FalconFelt> {
59    use sha3::{
60        Shake256,
61        digest::{ExtendableOutput, Update, XofReader},
62    };
63
64    let mut data = vec![];
65    data.extend_from_slice(&nonce.as_bytes());
66    data.extend_from_slice(message);
67    const K: u32 = (1u32 << 16) / MODULUS as u32;
68
69    let mut hasher = Shake256::default();
70    hasher.update(&data);
71    let mut reader = hasher.finalize_xof();
72
73    let mut coefficients: Vec<FalconFelt> = Vec::with_capacity(N);
74    while coefficients.len() != N {
75        let mut randomness = [0u8; 2];
76        reader.read(&mut randomness);
77        let t = ((randomness[0] as u32) << 8) | (randomness[1] as u32);
78        if t < K * MODULUS as u32 {
79            coefficients.push(u32_to_falcon_felt(t));
80        }
81    }
82
83    Polynomial { coefficients }
84}

89/// Converts a Miden field element to a field element in the prime field with characteristic
90/// the Falcon prime.
91///
92/// Note that since `FalconFelt::new` accepts `i16`, we first reduce the canonical value of
93/// the Miden field element modulo the Falcon prime and then cast the resulting value to an `i16`.
94/// Note that this final cast is safe as the Falcon prime is less than `i16::MAX`.
95fn felt_to_falcon_felt(value: Felt) -> FalconFelt {
96    FalconFelt::new((value.as_canonical_u64() % MODULUS as u64) as i16)
97}
98
99/// Converts a `u32` to a field element in the prime field with characteristic the Falcon prime.
100///
101/// Note that since `FalconFelt::new` accepts `i16`, we first reduce the `u32` value modulo
102/// the Falcon prime and then cast the resulting value to an `i16`.
103/// Note that this final cast is safe as the Falcon prime is less than `i16::MAX`.
104#[cfg(test)]
105fn u32_to_falcon_felt(value: u32) -> FalconFelt {
106    FalconFelt::new((value % MODULUS as u32) as i16)
107}
