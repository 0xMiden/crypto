22    /// Parallel implementation of [`Smt::with_entries()`].
23    ///
24    /// This method constructs a new sparse Merkle tree concurrently by processing subtrees in
25    /// parallel, working from the bottom up. The process works as follows:
26    ///
27    /// 1. First, the input key-value pairs are sorted and grouped into subtrees based on their leaf
28    ///    indices. Each subtree covers a range of 256 (2^8) possible leaf positions.
29    ///
30    /// 2. The subtrees are then processed in parallel:
31    ///    - For each subtree, compute the inner nodes from depth D down to depth D-8.
32    ///    - Each subtree computation yields a new subtree root and its associated inner nodes.
33    ///
34    /// 3. These subtree roots are recursively merged to become the "leaves" for the next iteration,
35    ///    which processes the next 8 levels up. This continues until the final root of the tree is
36    ///    computed at depth 0.
37    ///
38    /// # Errors
39    /// Returns an error if the provided entries contain multiple values for the same key.
40    pub(crate) fn with_entries_concurrent(
41        entries: impl IntoIterator<Item = (Word, Word)>,
42    ) -> Result<Self, MerkleError> {
43        let entries: Vec<(Word, Word)> = entries.into_iter().collect();
44
45        if entries.is_empty() {
46            return Ok(Self::default());
47        }
48
49        let (inner_nodes, leaves) = Self::build_subtrees(entries)?;
50
51        // All the leaves are empty
52        if inner_nodes.is_empty() {
53            return Ok(Self::default());
54        }
55
56        let root = inner_nodes.get(&NodeIndex::root()).unwrap().hash();
57        <Self as SparseMerkleTree<SMT_DEPTH>>::from_raw_parts(inner_nodes, leaves, root)
58    }
59
60    /// Similar to `with_entries_concurrent` but used for pre-sorted entries to avoid overhead.
61    pub(crate) fn with_sorted_entries_concurrent(
62        entries: impl IntoIterator<Item = (Word, Word)>,
63    ) -> Result<Self, MerkleError> {
64        let entries: Vec<(Word, Word)> = entries.into_iter().collect();
65
66        if entries.is_empty() {
67            return Ok(Self::default());
68        }
69
70        let (inner_nodes, leaves) = build_subtrees_from_sorted_entries(entries)?;
71
72        // All the leaves are empty
73        if inner_nodes.is_empty() {
74            return Ok(Self::default());
75        }
76
77        let root = inner_nodes.get(&NodeIndex::root()).unwrap().hash();
78        <Self as SparseMerkleTree<SMT_DEPTH>>::from_raw_parts(inner_nodes, leaves, root)
79    }

81    /// Parallel implementation of [`Smt::compute_mutations()`].
82    ///
83    /// This method computes mutations by recursively processing subtrees in parallel, working from
84    /// the bottom up. The process works as follows:
85    ///
86    /// 1. First, the input key-value pairs are sorted and grouped into subtrees based on their leaf
87    ///    indices. Each subtree covers a range of 256 (2^8) possible leaf positions.
88    ///
89    /// 2. The subtrees containing modifications are then processed in parallel:
90    ///    - For each modified subtree, compute node mutations from depth D up to depth D-8
91    ///    - Each subtree computation yields a new root at depth D-8 and its associated mutations
92    ///
93    /// 3. These subtree roots become the "leaves" for the next iteration, which processes the next
94    ///    8 levels up. This continues until reaching the tree's root at depth 0.
95    ///
96    /// # Errors
97    /// Returns an error if mutations would exceed [`MAX_LEAF_ENTRIES`] (1024 entries) in a leaf.
98    pub(crate) fn compute_mutations_concurrent(
99        &self,
100        kv_pairs: impl IntoIterator<Item = (Word, Word)>,
101    ) -> Result<MutationSet<SMT_DEPTH, Word, Word>, MerkleError>
102    where
103        Self: Sized + Sync,
104    {
105        // Collect and sort key-value pairs by their corresponding leaf index
106        let mut sorted_kv_pairs: Vec<_> = kv_pairs.into_iter().collect();
107        sorted_kv_pairs
108            .par_sort_unstable_by_key(|(key, _)| Self::key_to_leaf_index(key).position());
109
110        // Convert sorted pairs into mutated leaves and capture any new pairs
111        let (mut subtree_leaves, new_pairs) =
112            self.sorted_pairs_to_mutated_subtree_leaves(sorted_kv_pairs)?;
113
114        // If no mutations, return an empty mutation set
115        if subtree_leaves.is_empty() {
116            return Ok(MutationSet {
117                old_root: self.root(),
118                new_root: self.root(),
119                node_mutations: NodeMutations::default(),
120                new_pairs,
121            });
122        }
123
124        let mut node_mutations = NodeMutations::default();
125
126        // Process each depth level in reverse, stepping by the subtree depth
127        for depth in (SUBTREE_DEPTH..=SMT_DEPTH).step_by(SUBTREE_DEPTH as usize).rev() {
128            // Parallel processing of each subtree to generate mutations and roots
129            let (mutations_per_subtree, mut subtree_roots): (Vec<_>, Vec<_>) = subtree_leaves
130                .into_par_iter()
131                .map(|subtree| {
132                    debug_assert!(subtree.is_sorted() && !subtree.is_empty());
133                    self.build_subtree_mutations(subtree, SMT_DEPTH, depth)
134                })
135                .unzip();
136
137            // Prepare leaves for the next depth level
138            subtree_leaves = SubtreeLeavesIter::from_leaves(&mut subtree_roots).collect();
139
140            // Aggregate all node mutations
141            node_mutations.extend(mutations_per_subtree.into_iter().flatten());
142
143            debug_assert!(!subtree_leaves.is_empty());
144        }
145
146        let new_root = subtree_leaves[0][0].hash;
147
148        // Create mutation set
149        let mutation_set = MutationSet {
150            old_root: self.root(),
151            new_root,
152            node_mutations,
153            new_pairs,
154        };
155
156        // There should be mutations and new pairs at this point
157        debug_assert!(
158            !mutation_set.node_mutations().is_empty() && !mutation_set.new_pairs().is_empty()
159        );
160
161        Ok(mutation_set)
162    }

167    /// Computes the node mutations and the root of a subtree
168    fn build_subtree_mutations(
169        &self,
170        mut leaves: Vec<SubtreeLeaf>,
171        tree_depth: u8,
172        bottom_depth: u8,
173    ) -> (NodeMutations, SubtreeLeaf)
174    where
175        Self: Sized,
176    {
177        debug_assert!(bottom_depth <= tree_depth);
178        debug_assert!(Integer::is_multiple_of(&bottom_depth, &SUBTREE_DEPTH));
179        debug_assert!(leaves.len() <= usize::pow(2, SUBTREE_DEPTH as u32));
180
181        let subtree_root_depth = bottom_depth - SUBTREE_DEPTH;
182        let mut node_mutations: NodeMutations = Default::default();
183        let mut next_leaves: Vec<SubtreeLeaf> = Vec::with_capacity(leaves.len() / 2);
184
185        for current_depth in (subtree_root_depth..bottom_depth).rev() {
186            debug_assert!(current_depth <= bottom_depth);
187
188            let next_depth = current_depth + 1;
189            let mut iter = leaves.drain(..).peekable();
190
191            while let Some(first_leaf) = iter.next() {
192                // This constructs a valid index because next_depth will never exceed the depth of
193                // the tree.
194                let parent_index = NodeIndex::new_unchecked(next_depth, first_leaf.col).parent();
195                let parent_node = self.get_inner_node(parent_index);
196                let combined_node = fetch_sibling_pair(&mut iter, first_leaf, parent_node);
197                let combined_hash = combined_node.hash();
198
199                let &empty_hash = EmptySubtreeRoots::entry(tree_depth, current_depth);
200
201                // Add the parent node even if it is empty for proper upward updates
202                next_leaves.push(SubtreeLeaf {
203                    col: parent_index.position(),
204                    hash: combined_hash,
205                });
206
207                node_mutations.insert(
208                    parent_index,
209                    if combined_hash != empty_hash {
210                        NodeMutation::Addition(combined_node)
211                    } else {
212                        NodeMutation::Removal
213                    },
214                );
215            }
216            drop(iter);
217            leaves = mem::take(&mut next_leaves);
218        }
219
220        debug_assert_eq!(leaves.len(), 1);
221        let root_leaf = leaves.pop().unwrap();
222        (node_mutations, root_leaf)
223    }

264    /// Constructs a single leaf from an arbitrary amount of key-value pairs.
265    /// Those pairs must all have the same leaf index.
266    ///
267    /// # Errors
268    /// Returns a `MerkleError::DuplicateValuesForIndex` if the provided pairs contain multiple
269    /// values for the same key.
270    ///
271    /// # Returns
272    /// - `Ok(Some(SmtLeaf))` if a valid leaf is constructed.
273    /// - `Ok(None)` if the only provided value is `Self::EMPTY_VALUE`.
274    fn pairs_to_leaf(mut pairs: Vec<(Word, Word)>) -> Result<Option<SmtLeaf>, MerkleError> {
275        assert!(!pairs.is_empty());
276
277        if pairs.len() > 1 {
278            pairs.sort_by(|(key_1, _), (key_2, _)| leaf::cmp_keys(*key_1, *key_2));
279            // Check for duplicates in a sorted list by comparing adjacent pairs
280            if let Some(window) = pairs.windows(2).find(|window| window[0].0 == window[1].0) {
281                // If we find a duplicate, return an error
282                let col = Self::key_to_leaf_index(&window[0].0).index.position();
283                return Err(MerkleError::DuplicateValuesForIndex(col));
284            }
285            Ok(Some(SmtLeaf::new_multiple(pairs).unwrap()))
286        } else {
287            let (key, value) = pairs.pop().unwrap();
288            if value == Self::EMPTY_VALUE {
289                Ok(None)
290            } else {
291                Ok(Some(SmtLeaf::new_single(key, value)))
292            }
293        }
294    }

296    /// Computes leaves from a set of key-value pairs and current leaf values.
297    /// Derived from `sorted_pairs_to_leaves`
298    fn sorted_pairs_to_mutated_subtree_leaves(
299        &self,
300        pairs: Vec<(Word, Word)>,
301    ) -> Result<(MutatedSubtreeLeaves, Map<Word, Word>), MerkleError> {
302        // Map to track new key-value pairs for mutated leaves
303        let mut new_pairs = Map::new();
304
305        let accumulator = process_sorted_pairs_to_leaves(pairs, |leaf_pairs| {
306            let mut leaf = self.get_leaf(&leaf_pairs[0].0);
307
308            let mut leaf_changed = false;
309            for (key, value) in leaf_pairs {
310                // Check if the value has changed
311                let old_value = new_pairs.get(&key).cloned().unwrap_or_else(|| {
312                    // Safe to unwrap: `leaf_pairs` contains keys all belonging to this leaf.
313                    // `SmtLeaf::get_value()` only returns `None` if the key does not belong to the
314                    // leaf, which cannot happen due to the sorting/grouping
315                    // logic in `process_sorted_pairs_to_leaves()`.
316                    leaf.get_value(&key).unwrap()
317                });
318
319                if value != old_value {
320                    // Update the leaf and track the new key-value pair
321                    leaf = self.construct_prospective_leaf(leaf, &key, &value).map_err(
322                        |e| match e {
323                            SmtLeafError::TooManyLeafEntries { actual } => {
324                                MerkleError::TooManyLeafEntries { actual }
325                            },
326                            other => panic!("unexpected SmtLeaf::insert error: {:?}", other),
327                        },
328                    )?;
329                    new_pairs.insert(key, value);
330                    leaf_changed = true;
331                }
332            }
333
334            if leaf_changed {
335                // Only return the leaf if it actually changed
336                Ok(Some(leaf))
337            } else {
338                // Return None if leaf hasn't changed
339                Ok(None)
340            }
341        });
342        // The closure is the only possible source of errors.
343        // Since it never returns an error - only `Ok(Some(_))` or `Ok(None)` - we can safely assume
344        // `accumulator` is always `Ok(_)`.
345        Ok((accumulator?.leaves, new_pairs))
346    }

358/// Helper struct for organizing the data we care about when computing Merkle subtrees.
359///
360/// Note that these represent "conceptual" leaves of some subtree, not necessarily
361/// the leaf type for the sparse Merkle tree.
362#[derive(Debug, Copy, Clone, PartialEq, Eq, PartialOrd, Ord, Default)]
363pub struct SubtreeLeaf {
364    /// The 'value' field of [`NodeIndex`]. When computing a subtree, the depth is already known.
365    pub col: u64,
366    /// The hash of the node this `SubtreeLeaf` represents.
367    pub hash: Word,
368}

370/// Helper struct to organize the return value of [`Smt::sorted_pairs_to_leaves()`].
371#[derive(Debug, Clone)]
372pub(in crate::merkle::smt) struct PairComputations<K, L> {
373    /// Literal leaves to be added to the sparse Merkle tree's internal mapping.

377}

391    leaves: core::iter::Peekable<alloc::vec::Drain<'s, SubtreeLeaf>>,
392}

394impl<'s> SubtreeLeavesIter<'s> {
395    pub(crate) fn from_leaves(leaves: &'s mut Vec<SubtreeLeaf>) -> Self {
396        // TODO: determine if there is any notable performance difference between taking a Vec,
397        // which many need flattening first, vs storing a `Box<dyn Iterator<Item = SubtreeLeaf>>`.
398        // The latter may have self-referential properties that are impossible to express in purely
399        // safe Rust Rust.
400        Self { leaves: leaves.drain(..).peekable() }
401    }
402}
403
404impl Iterator for SubtreeLeavesIter<'_> {
405    type Item = Vec<SubtreeLeaf>;
406
407    /// Each `next()` collects an entire subtree.
408    fn next(&mut self) -> Option<Vec<SubtreeLeaf>> {
409        let mut subtree: Vec<SubtreeLeaf> = Default::default();
410
411        let mut last_subtree_col = 0;
412
413        while let Some(leaf) = self.leaves.peek() {
414            last_subtree_col = u64::max(1, last_subtree_col);
415            let is_exact_multiple = Integer::is_multiple_of(&last_subtree_col, &COLS_PER_SUBTREE);
416            let next_subtree_col = if is_exact_multiple {
417                u64::next_multiple_of(last_subtree_col + 1, COLS_PER_SUBTREE)
418            } else {
419                last_subtree_col.next_multiple_of(COLS_PER_SUBTREE)
420            };
421
422            last_subtree_col = leaf.col;
423            if leaf.col < next_subtree_col {
424                subtree.push(self.leaves.next().unwrap());
425            } else if subtree.is_empty() {
426                continue;
427            } else {
428                break;
429            }
430        }
431
432        if subtree.is_empty() {
433            debug_assert!(self.leaves.peek().is_none());
434            return None;
435        }
436
437        Some(subtree)
438    }
439}

444/// Processes sorted key-value pairs to compute leaves for a subtree.
445///
446/// This function groups key-value pairs by their corresponding column index and processes each
447/// group to construct leaves. The actual construction of the leaf is delegated to the
448/// `process_leaf` callback, allowing flexibility for different use cases (e.g., creating
449/// new leaves or mutating existing ones).
450///
451/// # Parameters
452/// - `pairs`: A vector of sorted key-value pairs. The pairs *must* be sorted by leaf index column
453///   (not simply by key). If the input is not sorted correctly, the function will produce incorrect
454///   results and may panic in debug mode.
455/// - `process_leaf`: A callback function used to process each group of key-value pairs
456///   corresponding to the same column index. The callback takes a vector of key-value pairs for a
457///   single column and returns the constructed leaf for that column.
458///
459/// # Returns
460/// A `PairComputations<u64, Self::Leaf>` containing:
461/// - `nodes`: A mapping of column indices to the constructed leaves.
462/// - `leaves`: A collection of `SubtreeLeaf` structures representing the processed leaves. Each
463///   `SubtreeLeaf` includes the column index and the hash of the corresponding leaf.
464///
465/// # Errors
466/// Returns an error if the `process_leaf` callback fails.
467///
468/// # Panics
469/// This function will panic in debug mode if the input `pairs` are not sorted by column index.
470pub(crate) fn process_sorted_pairs_to_leaves<F>(
471    pairs: Vec<(Word, Word)>,
472    mut process_leaf: F,
473) -> Result<PairComputations<u64, SmtLeaf>, MerkleError>
474where
475    F: FnMut(Vec<(Word, Word)>) -> Result<Option<SmtLeaf>, MerkleError>,
476{
477    debug_assert!(pairs.is_sorted_by_key(|(key, _)| Smt::key_to_leaf_index(key).position()));
478    let mut accumulator: PairComputations<u64, SmtLeaf> = Default::default();
479    // As we iterate, we'll keep track of the kv-pairs we've seen so far that correspond to a
480    // single leaf. When we see a pair that's in a different leaf, we'll swap these pairs
481    // out and store them in our accumulated leaves.
482    let mut current_leaf_buffer: Vec<(Word, Word)> = Default::default();
483    let mut iter = pairs.into_iter().peekable();
484    while let Some((key, value)) = iter.next() {
485        let col = Smt::key_to_leaf_index(&key).index.position();
486        let peeked_col = iter.peek().map(|(key, _v)| {
487            let index = Smt::key_to_leaf_index(key);
488            let next_col = index.index.position();
489            // We panic if `pairs` is not sorted by column.
490            debug_assert!(next_col >= col);
491            next_col
492        });
493        current_leaf_buffer.push((key, value));
494        // If the next pair is the same column as this one, then we're done after adding this
495        // pair to the buffer.
496        if peeked_col == Some(col) {
497            continue;
498        }
499        // Otherwise, the next pair is a different column, or there is no next pair. Either way
500        // it's time to swap out our buffer.
501        let leaf_pairs = mem::take(&mut current_leaf_buffer);
502        // Process leaf and propagate any errors
503        match process_leaf(leaf_pairs) {
504            Ok(Some(leaf)) => {
505                accumulator.nodes.insert(col, leaf);
506            },
507            Ok(None) => {
508                // No leaf was constructed for this column. The column will be skipped.
509            },
510            Err(e) => return Err(e),
511        }
512        debug_assert!(current_leaf_buffer.is_empty());
513    }
514    // Compute the leaves from the nodes concurrently
515    let mut accumulated_leaves: Vec<SubtreeLeaf> = accumulator
516        .nodes
517        .clone()
518        .into_par_iter()
519        .map(|(col, leaf)| SubtreeLeaf { col, hash: Smt::hash_leaf(&leaf) })
520        .collect();
521    // Sort the leaves by column
522    accumulated_leaves.par_sort_by_key(|leaf| leaf.col);
523    // TODO: determine is there is any notable performance difference between computing
524    // subtree boundaries after the fact as an iterator adapter (like this), versus computing
525    // subtree boundaries as we go. Either way this function is only used at the beginning of a
526    // parallel construction, so it should not be a critical path.
527    accumulator.leaves = SubtreeLeavesIter::from_leaves(&mut accumulated_leaves).collect();
528    Ok(accumulator)
529}

531/// Computes the raw parts for a new sparse Merkle tree from a set of key-value pairs.
532///
533/// This function is mostly an implementation detail of
534/// [`Smt::with_entries_concurrent()`].
535///
536/// # Errors
537/// Returns an error if the provided entries contain multiple values for the same key.
538fn build_subtrees_from_sorted_entries(
539    entries: Vec<(Word, Word)>,
540) -> Result<(InnerNodes, Leaves), MerkleError> {
541    let mut accumulated_nodes: InnerNodes = Default::default();
542    let PairComputations {
543        leaves: mut leaf_subtrees,
544        nodes: initial_leaves,
545    } = Smt::sorted_pairs_to_leaves(entries)?;
546    // If there are no leaves, we can return early
547    if initial_leaves.is_empty() {
548        return Ok((accumulated_nodes, initial_leaves));
549    }
550    for current_depth in (SUBTREE_DEPTH..=SMT_DEPTH).step_by(SUBTREE_DEPTH as usize).rev() {
551        let (nodes, mut subtree_roots): (Vec<Map<_, _>>, Vec<SubtreeLeaf>) = leaf_subtrees
552            .into_par_iter()
553            .map(|subtree| {
554                debug_assert!(subtree.is_sorted());
555                debug_assert!(!subtree.is_empty());
556                let (nodes, subtree_root) = build_subtree(subtree, SMT_DEPTH, current_depth);
557                (nodes, subtree_root)
558            })
559            .unzip();
560        leaf_subtrees = SubtreeLeavesIter::from_leaves(&mut subtree_roots).collect();
561        accumulated_nodes.extend(nodes.into_iter().flatten());
562        debug_assert!(!leaf_subtrees.is_empty());
563    }
564    Ok((accumulated_nodes, initial_leaves))
565}

567/// Builds Merkle nodes from a bottom layer of "leaves" -- represented by a horizontal index and
568/// the hash of the leaf at that index. `leaves` *must* be sorted by horizontal index, and
569/// `leaves` must not contain more than one depth-8 subtree's worth of leaves.
570///
571/// This function will then calculate the inner nodes above each leaf for 8 layers, as well as
572/// the "leaves" for the next 8-deep subtree, so this function can effectively be chained into
573/// itself.
574///
575/// # Panics
576/// With debug assertions on, this function panics under invalid inputs: if `leaves` contains
577/// more entries than can fit in a depth-8 subtree, if `leaves` contains leaves belonging to
578/// different depth-8 subtrees, if `bottom_depth` is lower in the tree than the specified
579/// maximum depth (`DEPTH`), or if `leaves` is not sorted.
580pub(crate) fn build_subtree(
581    mut leaves: Vec<SubtreeLeaf>,
582    tree_depth: u8,
583    bottom_depth: u8,
584) -> (Map<NodeIndex, InnerNode>, SubtreeLeaf) {
585    #[cfg(debug_assertions)]
586    {
587        // Ensure that all leaves have unique column indices within this subtree.
588        // In normal usage via public APIs, this should never happen because leaf
589        // construction enforces uniqueness. However, when testing or benchmarking
590        // `build_subtree()` in isolation, duplicate columns can appear if input
591        // constraints are not enforced.
592        use alloc::collections::BTreeSet;
593        let mut seen_cols = BTreeSet::new();
594        for leaf in &leaves {
595            assert!(seen_cols.insert(leaf.col), "Duplicate column found in subtree: {}", leaf.col);
596        }
597    }
598    debug_assert!(bottom_depth <= tree_depth);
599    debug_assert!(Integer::is_multiple_of(&bottom_depth, &SUBTREE_DEPTH));
600    debug_assert!(leaves.len() <= usize::pow(2, SUBTREE_DEPTH as u32));
601    let subtree_root = bottom_depth - SUBTREE_DEPTH;
602    let mut inner_nodes: Map<NodeIndex, InnerNode> = Default::default();
603    let mut next_leaves: Vec<SubtreeLeaf> = Vec::with_capacity(leaves.len() / 2);
604    for next_depth in (subtree_root..bottom_depth).rev() {
605        debug_assert!(next_depth <= bottom_depth);
606        // `next_depth` is the stuff we're making.
607        // `current_depth` is the stuff we have.
608        let current_depth = next_depth + 1;
609        let mut iter = leaves.drain(..).peekable();
610        while let Some(first) = iter.next() {
611            // On non-continuous iterations, including the first iteration, `first_column` may
612            // be a left or right node. On subsequent continuous iterations, we will always call
613            // `iter.next()` twice.
614            // On non-continuous iterations (including the very first iteration), this column
615            // could be either on the left or the right. If the next iteration is not
616            // discontinuous with our right node, then the next iteration's
617            let is_right = first.col.is_odd();
618            let (left, right) = if is_right {
619                // Discontinuous iteration: we have no left node, so it must be empty.
620                let left = SubtreeLeaf {
621                    col: first.col - 1,
622                    hash: *EmptySubtreeRoots::entry(tree_depth, current_depth),
623                };
624                let right = first;
625                (left, right)
626            } else {
627                let left = first;
628                let right_col = first.col + 1;
629                let right = match iter.peek().copied() {
630                    Some(SubtreeLeaf { col, .. }) if col == right_col => {
631                        // Our inputs must be sorted.
632                        debug_assert!(left.col <= col);
633                        // The next leaf in the iterator is our sibling. Use it and consume it!
634                        iter.next().unwrap()
635                    },
636                    // Otherwise, the leaves don't contain our sibling, so our sibling must be
637                    // empty.
638                    _ => SubtreeLeaf {
639                        col: right_col,
640                        hash: *EmptySubtreeRoots::entry(tree_depth, current_depth),
641                    },
642                };
643                (left, right)
644            };
645            let index = NodeIndex::new_unchecked(current_depth, left.col).parent();
646            let node = InnerNode { left: left.hash, right: right.hash };
647            let hash = node.hash();
648            let &equivalent_empty_hash = EmptySubtreeRoots::entry(tree_depth, next_depth);
649            // If this hash is empty, then it doesn't become a new inner node, nor does it count
650            // as a leaf for the next depth.
651            if hash != equivalent_empty_hash {
652                inner_nodes.insert(index, node);
653                next_leaves.push(SubtreeLeaf { col: index.position(), hash });
654            }
655        }
656        // Stop borrowing `leaves`, so we can swap it.
657        // The iterator is empty at this point anyway.
658        drop(iter);
659        // After each depth, consider the stuff we just made the new "leaves", and empty the
660        // other collection.
661        mem::swap(&mut leaves, &mut next_leaves);
662    }
663    debug_assert_eq!(leaves.len(), 1);
664    let root = leaves.pop().unwrap();
665    (inner_nodes, root)
666}

668/// Constructs an `InnerNode` representing the sibling pair of which `first_leaf` is a part:
669/// - If `first_leaf` is a right child, the left child is copied from the `parent_node`.
670/// - If `first_leaf` is a left child, the right child is taken from `iter` if it was also mutated
671///   or copied from the `parent_node`.
672///
673/// Returns the `InnerNode` containing the hashes of the sibling pair.
674pub(crate) fn fetch_sibling_pair(
675    iter: &mut core::iter::Peekable<alloc::vec::Drain<SubtreeLeaf>>,
676    first_leaf: SubtreeLeaf,
677    parent_node: InnerNode,
678) -> InnerNode {
679    let is_right_node = first_leaf.col.is_odd();
680
681    if is_right_node {
682        let left_leaf = SubtreeLeaf {
683            col: first_leaf.col - 1,
684            hash: parent_node.left,
685        };
686        InnerNode {
687            left: left_leaf.hash,
688            right: first_leaf.hash,
689        }
690    } else {
691        let right_col = first_leaf.col + 1;
692        let right_leaf = match iter.peek().copied() {
693            Some(SubtreeLeaf { col, .. }) if col == right_col => iter.next().unwrap(),
694            _ => SubtreeLeaf { col: right_col, hash: parent_node.right },
695        };
696        InnerNode {
697            left: first_leaf.hash,
698            right: right_leaf.hash,
699        }
700    }
701}
