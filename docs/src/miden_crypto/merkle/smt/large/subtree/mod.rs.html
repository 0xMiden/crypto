21/// A mutation converted to subtree-local coordinates.
22struct LocalMutation {
23    /// Local index within the subtree (0-254).
24    local_index: u8,
25    /// The kind of mutation to apply.
26    kind: LocalMutationKind,
27}
28
29/// The kind of mutation to apply to a subtree node.
30enum LocalMutationKind {
31    /// Add or update a node with the given child hashes.
32    Addition {
33        /// The left child hash.
34        left: Word,
35        /// The right child hash.
36        right: Word,
37        /// Whether the left child is non-empty.
38        has_left: bool,
39        /// Whether the right child is non-empty.
40        has_right: bool,
41    },
42    /// Remove the node at this position.
43    Removal,
44}
45
46/// Iterator over non-empty nodes in a subtree.
47///
48/// Yields [`SubtreeNode`] for each node that has at least one non-empty child.
49/// Tracks hash positions for use cases that need them (e.g., rebuild).
50struct SubtreeNodeIter<'a> {
51    subtree: &'a Subtree,
52    /// Current word index in child_bits.
53    word_idx: usize,
54    /// Remaining bits in current word (nodes are cleared as visited).
55    remaining: u64,
56    /// Current position in the hashes vector.
57    hash_idx: usize,
58}
59
60/// A non-empty node yielded by [`SubtreeNodeIter`].
61struct SubtreeNode {
62    /// Local index within the subtree (0-254).
63    local_index: u8,
64    /// Starting position in the hashes vector.
65    hash_start: usize,
66    /// Whether the left child is non-empty.
67    has_left: bool,
68    /// Whether the right child is non-empty.
69    has_right: bool,
70}
71
72impl<'a> SubtreeNodeIter<'a> {
73    fn new(subtree: &'a Subtree) -> Self {
74        Self {
75            subtree,
76            word_idx: 0,
77            remaining: subtree.child_bits[0],
78            hash_idx: 0,
79        }
80    }
81}
82
83impl Iterator for SubtreeNodeIter<'_> {
84    type Item = SubtreeNode;
85
86    fn next(&mut self) -> Option<SubtreeNode> {
87        const NODE_PAIR_MASK: u64 = 0b11;
88        const NODES_PER_WORD: u8 = 32;
89
90        // Skip empty words.
91        while self.remaining == 0 && self.word_idx < 7 {
92            self.word_idx += 1;
93            self.remaining = self.subtree.child_bits[self.word_idx];
94        }
95
96        if self.remaining == 0 {
97            return None;
98        }
99
100        // Find first set bit and its node index.
101        let bit_pos = self.remaining.trailing_zeros() as u8;
102        let node_in_word = bit_pos / Subtree::BITS_PER_NODE as u8;
103        let local_index = (self.word_idx as u8) * NODES_PER_WORD + node_in_word;
104
105        // Clear this node's bits so we don't visit it again.
106        self.remaining &= !(NODE_PAIR_MASK << (node_in_word * Subtree::BITS_PER_NODE as u8));
107
108        // Look up child presence.
109        let bit_offset = (local_index as usize) * Subtree::BITS_PER_NODE;
110        let has_left = self.subtree.get_bit(bit_offset);
111        let has_right = self.subtree.get_bit(bit_offset + 1);
112
113        let node = SubtreeNode {
114            local_index,
115            hash_start: self.hash_idx,
116            has_left,
117            has_right,
118        };
119
120        // Advance hash index for next iteration.
121        self.hash_idx += has_left as usize + has_right as usize;
122
123        Some(node)
124    }
125}

130/// Represents a complete 8-depth subtree that is serialized into a single RocksDB entry.
131///
132/// ### What is stored
133/// - `child_bits` is a 512-bit bitmask (2 bits per node) marking non-empty left/right children.
134/// - `hashes` stores only non-empty child hashes in bit order.
135///
136/// ### Local index layout (how indices are computed)
137/// - Indices are **subtree-local** and follow binary-heap (level-order) layout: `root = 0`;
138///   children of `i` are at `2i+1` and `2i+2`.
139/// - Equivalently, given a `(depth, value)` from the parent tree, the local index is obtained by
140///   taking the node’s depth **relative to the subtree root** and its left-to-right position within
141///   that level (offset by the total number of nodes in all previous levels).
142///
143/// ### In-memory format
144/// - `child_bits`: 8 little-endian u64s (64 bytes total).
145/// - `hashes`: contiguous `Vec<Word>` of non-empty child hashes.
146#[derive(Debug, Clone)]
147pub struct Subtree {
148    /// Index of this subtree's root in the parent SMT.
149    root_index: NodeIndex,
150    /// Bitmask indicating non-empty children.
151    /// For local index i: bit 2*i = left child non-empty, bit 2*i+1 = right child non-empty.
152    /// Stored as 8 little-endian u64s (512 bits total, 2 bits per node x 255 nodes + 2 unused).
153    child_bits: [u64; 8],
154    /// Non-empty child hashes in bit order.
155    hashes: Vec<Word>,
156}
157

181    /// Returns the root index of this subtree in the parent SMT.
182    pub fn root_index(&self) -> NodeIndex {
183        self.root_index
184    }
185
186    /// Returns the number of non-empty inner nodes in this subtree.
187    ///
188    /// A node "exists" if at least one of its two child bits is set. Each node occupies a
189    /// 2-bit pair `(left, right)` in `child_bits`, so we collapse each pair into one bit and
190    /// count. For a word `w`:
191    ///
192    /// 1. `w | (w >> 1)` — propagates right-child bits into the even (left) position
193    /// 2. `& 0x5555...`  — keeps only even positions (one bit per node)
194    /// 3. `count_ones()` — counts occupied nodes
195    ///
196    /// Example with 4 nodes:
197    ///   `w = 0b10_01_11_00`
198    ///
199    ///   Node 0 (bits 0-1): left=0, right=0 → empty
200    ///   Node 1 (bits 2-3): left=1, right=1 → both children
201    ///   Node 2 (bits 4-5): left=1, right=0 → left only
202    ///   Node 3 (bits 6-7): left=0, right=1 → right only
203    ///
204    ///   - `w | (w >> 1)` = `0b11_01_11_10` (OR each pair into even positions)
205    ///   - `& 0x55`       = `0b01_01_01_00` (mask to keep only even bits)
206    ///   - popcount = 3 (nodes 1, 2, 3 exist; node 0 is empty)
207    pub fn len(&self) -> usize {
208        const EVEN_BIT_MASK: u64 = 0x5555_5555_5555_5555;
209
210        self.child_bits
211            .iter()
212            .map(|&w| (((w | (w >> 1)) & EVEN_BIT_MASK).count_ones()) as usize)
213            .sum()
214    }
215
216    /// Returns `true` if this subtree has no nodes.
217    pub fn is_empty(&self) -> bool {
218        self.hashes.is_empty()
219    }

221    /// Returns the inner node at the given global index, if it exists.
222    pub fn get_inner_node(&self, index: NodeIndex) -> Option<InnerNode> {
223        let local_index = Self::global_to_local(index, self.root_index);

225    }

227    /// Converts a global NodeIndex to a local subtree index.
228    ///
229    /// # Panics
230    /// Panics if `global.depth() < base.depth()`.
231    pub fn global_to_local(global: NodeIndex, base: NodeIndex) -> u8 {
232        assert!(
233            global.depth() >= base.depth(),
234            "Global depth is less than base depth = {}, global depth = {}",
235            base.depth(),
236            global.depth()
237        );
238
239        let relative_depth = global.depth() - base.depth();
240        let level_mask = (1u64 << relative_depth) - 1;
241        let local_position = (global.position() & level_mask) as u8;
242        level_mask as u8 + local_position
243    }

245    /// Creates the storage key for a subtree.
246    pub fn subtree_key(root_index: NodeIndex) -> [u8; 9] {
247        let mut key = [0u8; 9];
248        key[0] = root_index.depth();
249        key[1..].copy_from_slice(&root_index.position().to_be_bytes());
250        key
251    }

253    /// Finds the subtree root for a given node index.
254    pub fn find_subtree_root(node_index: NodeIndex) -> NodeIndex {
255        let depth = node_index.depth();
256        if depth < SUBTREE_DEPTH {
257            NodeIndex::root()
258        } else {
259            let relative_depth = depth % SUBTREE_DEPTH;
260            let subtree_root_depth = depth - relative_depth;
261            let base_value = node_index.position() >> relative_depth;
262
263            NodeIndex::new(subtree_root_depth, base_value).unwrap()
264        }
265    }

281    /// Applies a batch of mutations to this subtree.
282    ///
283    /// When mutations only update existing nodes (same structure, different hashes),
284    /// hashes are patched in-place. Structural changes (additions or removals) trigger a rebuild.
285    ///
286    /// - `NodeMutation::Addition(node)` inserts or updates a node
287    /// - `NodeMutation::Removal` removes a node
288    pub fn apply_mutations<'a>(
289        &mut self,
290        mutations: impl IntoIterator<Item = (&'a NodeIndex, &'a NodeMutation)>,
291    ) {
292        let Some((local_mutations, can_patch_in_place)) = self.collect_local_mutations(mutations)
293        else {
294            return;
295        };
296
297        if can_patch_in_place {
298            self.patch_hashes_in_place(&local_mutations);
299        } else {
300            self.rebuild_from_mutations(local_mutations);
301        }
302    }
303
304    /// Inserts or updates an inner node at the given global index.
305    ///
306    /// **Note**: For batch updates, prefer [`apply_mutations`](Self::apply_mutations) which is
307    /// more efficient when applying multiple changes.
308    ///
309    /// Returns the previous node at this index, if any.
310    pub fn insert_inner_node(
311        &mut self,
312        index: NodeIndex,
313        inner_node: InnerNode,
314    ) -> Option<InnerNode> {
315        let local_index = Self::global_to_local(index, self.root_index);
316        let previous = self.get_by_local_index(local_index);
317
318        self.apply_mutations([(&index, &NodeMutation::Addition(inner_node))]);
319
320        previous
321    }
322
323    /// Removes an inner node at the given global index.
324    ///
325    /// **Note**: For batch updates, prefer [`apply_mutations`](Self::apply_mutations) which is
326    /// more efficient when applying multiple changes.
327    ///
328    /// Returns the removed node, if any.
329    pub fn remove_inner_node(&mut self, index: NodeIndex) -> Option<InnerNode> {
330        let local_index = Self::global_to_local(index, self.root_index);
331        let previous = self.get_by_local_index(local_index);
332
333        if previous.is_some() {
334            self.apply_mutations([(&index, &NodeMutation::Removal)]);
335        }
336
337        previous
338    }

343    /// Serializes this subtree into a compact byte representation.
344    ///
345    /// The format is trivial since in-memory layout matches serialization:
346    /// - 64 bytes: `child_bits` as little-endian u64s
347    /// - Variable: non-empty child hashes (32 bytes each)
348    pub fn to_vec(&self) -> Vec<u8> {
349        let mut result =
350            Vec::with_capacity(Self::BITMASK_SIZE + self.hashes.len() * Self::HASH_SIZE);
351
352        for word in &self.child_bits {
353            result.extend_from_slice(&word.to_le_bytes());
354        }
355
356        for hash in &self.hashes {
357            result.extend_from_slice(&hash.as_bytes());
358        }
359
360        result
361    }
362
363    /// Deserializes a subtree from its compact byte representation.
364    ///
365    /// The format is trivial since in-memory layout matches serialization:
366    /// - 64 bytes: `child_bits` as little-endian u64s
367    /// - Variable: non-empty child hashes (32 bytes each)
368    pub fn from_vec(root_index: NodeIndex, data: &[u8]) -> Result<Self, SubtreeError> {
369        if data.len() < Self::BITMASK_SIZE {
370            return Err(SubtreeError::TooShort {
371                found: data.len(),
372                min: Self::BITMASK_SIZE,
373            });
374        }
375
376        let (bits_data, hash_data) = data.split_at(Self::BITMASK_SIZE);
377
378        let mut child_bits = [0u64; 8];
379        for (i, chunk) in bits_data.chunks_exact(8).enumerate() {
380            child_bits[i] = u64::from_le_bytes(chunk.try_into().unwrap());
381        }
382
383        // Bits 510-511 are unused - reject corrupted data where these bits are set.
384        const UNUSED_BITS_MASK: u64 = 0b11 << 62;
385        if child_bits[7] & UNUSED_BITS_MASK != 0 {
386            return Err(SubtreeError::InvalidBitmask);
387        }
388
389        let set_bits: usize = child_bits.iter().map(|w| w.count_ones() as usize).sum();
390        if hash_data.len() != set_bits * Self::HASH_SIZE {
391            return Err(SubtreeError::BadHashLen {
392                expected: set_bits * Self::HASH_SIZE,
393                found: hash_data.len(),
394            });
395        }
396
397        let hashes: Vec<Word> = hash_data
398            .chunks_exact(Self::HASH_SIZE)
399            .map(|chunk| Word::try_from(chunk).map_err(|_| SubtreeError::InvalidHashData))
400            .collect::<Result<_, _>>()?;
401
402        Ok(Self { root_index, child_bits, hashes })
403    }

408    /// Gets an inner node by its local index within this subtree.
409    fn get_by_local_index(&self, local_index: u8) -> Option<InnerNode> {
410        let left_bit = (local_index as usize) * Self::BITS_PER_NODE;
411        let right_bit = left_bit + 1;
412
413        let has_left = self.get_bit(left_bit);
414        let has_right = self.get_bit(right_bit);
415
416        if !has_left && !has_right {
417            return None;
418        }
419
420        let node_depth_in_subtree = Self::local_index_to_depth(local_index);
421        let child_depth = self.root_index.depth() + node_depth_in_subtree + 1;
422        let empty_hash = *EmptySubtreeRoots::entry(SMT_DEPTH, child_depth);
423
424        let left_pos = self.count_bits_before(left_bit);
425        let left = if has_left { self.hashes[left_pos] } else { empty_hash };
426
427        let right_pos = if has_left { left_pos + 1 } else { left_pos };
428        let right = if has_right { self.hashes[right_pos] } else { empty_hash };
429
430        Some(InnerNode { left, right })
431    }

433    /// Collects subtree-local mutations and determines if the structure stays the same.
434    ///
435    /// Returns `None` if there are no mutations to apply.
436    fn collect_local_mutations<'a>(
437        &self,
438        mutations: impl IntoIterator<Item = (&'a NodeIndex, &'a NodeMutation)>,
439    ) -> Option<(Vec<LocalMutation>, bool)> {
440        let mut local_mutations = Vec::new();
441        let mut can_patch_in_place = true;
442
443        for (index, mutation) in mutations {
444            let local_index = Self::global_to_local(*index, self.root_index);
445            let bit_offset = (local_index as usize) * Self::BITS_PER_NODE;
446            let old_has_left = self.get_bit(bit_offset);
447            let old_has_right = self.get_bit(bit_offset + 1);
448
449            let kind = match mutation {
450                NodeMutation::Addition(node) => {
451                    let node_depth_in_subtree = Self::local_index_to_depth(local_index);
452                    let child_depth = self.root_index.depth() + node_depth_in_subtree + 1;
453                    let empty_hash = *EmptySubtreeRoots::entry(SMT_DEPTH, child_depth);
454                    let has_left = node.left != empty_hash;
455                    let has_right = node.right != empty_hash;
456
457                    if old_has_left != has_left || old_has_right != has_right {
458                        can_patch_in_place = false;
459                    }
460
461                    LocalMutationKind::Addition {
462                        left: node.left,
463                        right: node.right,
464                        has_left,
465                        has_right,
466                    }
467                },
468                NodeMutation::Removal => {
469                    if old_has_left || old_has_right {
470                        can_patch_in_place = false;
471                    }
472                    LocalMutationKind::Removal
473                },
474            };
475
476            local_mutations.push(LocalMutation { local_index, kind });
477        }
478
479        (!local_mutations.is_empty()).then_some((local_mutations, can_patch_in_place))
480    }

482    /// Patches hashes in-place when the subtree structure is unchanged.
483    ///
484    /// Called when [`collect_local_mutations`](Self::collect_local_mutations) returns
485    /// `can_patch_in_place = true`, meaning all mutations preserve the existing child bits.
486    fn patch_hashes_in_place(&mut self, local_mutations: &[LocalMutation]) {
487        for m in local_mutations {
488            let LocalMutationKind::Addition { left, right, has_left, has_right } = m.kind else {
489                continue;
490            };
491            let bit_offset = (m.local_index as usize) * Self::BITS_PER_NODE;
492            let hash_pos = self.count_bits_before(bit_offset);
493            if has_left {
494                self.hashes[hash_pos] = left;
495            }
496            if has_right {
497                self.hashes[hash_pos + has_left as usize] = right;
498            }
499        }
500    }

502    /// Rebuilds the subtree when mutations change the structure.
503    ///
504    /// Called when [`collect_local_mutations`](Self::collect_local_mutations) returns
505    /// `can_patch_in_place = false`, meaning at least one mutation adds or removes child bits.
506    /// Performs a merge of sorted existing nodes with sorted mutations.
507    fn rebuild_from_mutations(&mut self, mut local_mutations: Vec<LocalMutation>) {
508        local_mutations.sort_unstable_by_key(|m| m.local_index);
509
510        let mut new_child_bits = [0u64; 8];
511        let mut new_hashes = Vec::with_capacity(self.hashes.len() + local_mutations.len() * 2);
512
513        let mut node_iter = SubtreeNodeIter::new(self);
514        let mut current_node = node_iter.next();
515        let mut mut_idx = 0;
516
517        loop {
518            let node_idx = current_node.as_ref().map(|n| n.local_index);
519            let mutation_idx = local_mutations.get(mut_idx).map(|m| m.local_index);
520
521            match (node_idx, mutation_idx) {
522                (Some(n), Some(m)) if n < m => {
523                    let node = current_node.take().unwrap();
524                    self.copy_node(&node, &mut new_child_bits, &mut new_hashes);
525                    current_node = node_iter.next();
526                },
527                (Some(n), Some(m)) if n > m => {
528                    Self::write_mutation(
529                        &local_mutations[mut_idx],
530                        &mut new_child_bits,
531                        &mut new_hashes,
532                    );
533                    mut_idx += 1;
534                },
535                (Some(_), Some(_)) => {
536                    current_node = node_iter.next();
537                    Self::write_mutation(
538                        &local_mutations[mut_idx],
539                        &mut new_child_bits,
540                        &mut new_hashes,
541                    );
542                    mut_idx += 1;
543                },
544                (Some(_), None) => {
545                    let node = current_node.take().unwrap();
546                    self.copy_node(&node, &mut new_child_bits, &mut new_hashes);
547                    current_node = node_iter.next();
548                },
549                (None, Some(_)) => {
550                    Self::write_mutation(
551                        &local_mutations[mut_idx],
552                        &mut new_child_bits,
553                        &mut new_hashes,
554                    );
555                    mut_idx += 1;
556                },
557                (None, None) => break,
558            }
559        }
560
561        self.child_bits = new_child_bits;
562        self.hashes = new_hashes;
563    }

565    /// Copies a node's hashes to the new subtree being built.
566    fn copy_node(&self, node: &SubtreeNode, new_bits: &mut [u64; 8], new_hashes: &mut Vec<Word>) {
567        let (word_idx, bit_idx) =
568            Self::bit_position((node.local_index as usize) * Self::BITS_PER_NODE);
569        if node.has_left {
570            new_bits[word_idx] |= 1u64 << bit_idx;
571            new_hashes.push(self.hashes[node.hash_start]);
572        }
573        if node.has_right {
574            new_bits[word_idx] |= 1u64 << (bit_idx + 1);
575            new_hashes.push(self.hashes[node.hash_start + node.has_left as usize]);
576        }
577    }
578
579    /// Writes a mutation's data to the new subtree (if it's an addition).
580    fn write_mutation(m: &LocalMutation, new_bits: &mut [u64; 8], new_hashes: &mut Vec<Word>) {
581        let LocalMutationKind::Addition { left, right, has_left, has_right } = &m.kind else {
582            return;
583        };
584        let (word_idx, bit_idx) =
585            Self::bit_position((m.local_index as usize) * Self::BITS_PER_NODE);
586        if *has_left {
587            new_bits[word_idx] |= 1u64 << bit_idx;
588            new_hashes.push(*left);
589        }
590        if *has_right {
591            new_bits[word_idx] |= 1u64 << (bit_idx + 1);
592            new_hashes.push(*right);
593        }
594    }

596    /// Splits a bit offset into `(word_idx, bit_idx)` for indexing into `child_bits`.
597    #[inline]
598    const fn bit_position(bit_offset: usize) -> (usize, usize) {
599        (bit_offset / 64, bit_offset & 0b_0011_1111)
600    }
601
602    /// Gets a bit from the child_bits array.
603    #[inline]
604    fn get_bit(&self, bit_offset: usize) -> bool {
605        let (word_idx, bit_idx) = Self::bit_position(bit_offset);
606        (self.child_bits[word_idx] >> bit_idx) & 1 != 0
607    }
608
609    /// Counts the number of set bits before the given bit offset.
610    #[inline]
611    fn count_bits_before(&self, bit_offset: usize) -> usize {
612        let (word_idx, bit_idx) = Self::bit_position(bit_offset);
613        let mut count = 0;
614
615        for i in 0..word_idx {
616            count += self.child_bits[i].count_ones() as usize;
617        }
618
619        if bit_idx > 0 {
620            let mask = (1u64 << bit_idx) - 1;
621            count += (self.child_bits[word_idx] & mask).count_ones() as usize;
622        }
623
624        count
625    }

627    /// Convert local index to depth within subtree.
628    #[inline]
629    const fn local_index_to_depth(local_index: u8) -> u8 {
630        let n = local_index as u16 + 1;
631        (u16::BITS as u8 - 1) - n.leading_zeros() as u8
632    }
633
634    /// Returns an iterator over all non-empty nodes in this subtree.
635    fn node_iter(&self) -> SubtreeNodeIter<'_> {
636        SubtreeNodeIter::new(self)
637    }

639    /// Returns whether the given mutations would take the patch-in-place path.
640    ///
641    /// `None` means no mutations to apply; `Some(true)` means patch-in-place;
642    /// `Some(false)` means rebuild.
643    #[cfg(test)]
644    fn would_patch_in_place<'a>(
645        &self,
646        mutations: impl IntoIterator<Item = (&'a NodeIndex, &'a NodeMutation)>,
647    ) -> Option<bool> {
648        self.collect_local_mutations(mutations).map(|(_, can_patch)| can_patch)
649    }
