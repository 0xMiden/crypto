15/// Represents a complete 8-depth subtree that is serialized into a single RocksDB entry.
16///
17/// ### What is stored
18/// - `nodes` tracks only **non-empty inner nodes** of this subtree (i.e., nodes for which at least
19///   one child differs from the canonical empty hash). Each entry stores an `InnerNode` (hash
20///   pair).
21///
22/// ### Local index layout (how indices are computed)
23/// - Indices are **subtree-local** and follow binary-heap (level-order) layout: `root = 0`;
24///   children of `i` are at `2i+1` and `2i+2`.
25/// - Equivalently, given a `(depth, value)` from the parent tree, the local index is obtained by
26///   taking the node’s depth **relative to the subtree root** and its left-to-right position within
27///   that level (offset by the total number of nodes in all previous levels).
28///
29/// ### Serialization (`to_vec` / `from_vec`)
30/// - Uses a **512-bit bitmask** (2 bits per node) to mark non-empty left/right children, followed
31///   by a packed stream of `Word` hashes for each set bit.
32/// - Children equal to the canonical empty hash are omitted in the byte representation and
33///   reconstructed on load using `EmptySubtreeRoots` and the child’s depth in the parent tree.
34#[derive(Debug, Clone)]
35pub struct Subtree {
36    /// Index of this subtree's root in the parent SMT.
37    root_index: NodeIndex,
38    /// Inner nodes keyed by subtree-local index (binary-heap order).
39    nodes: Map<u8, InnerNode>,
40}
41
42impl Subtree {
43    const HASH_SIZE: usize = 32;
44    const BITMASK_SIZE: usize = 64;
45    const MAX_NODES: u8 = 255;
46    const BITS_PER_NODE: usize = 2;

79    /// Serializes this subtree into a compact byte representation.
80    ///
81    /// The encoding has two components:
82    ///
83    /// **Bitmask (512 bits)** — Each internal node (up to 255 total) is assigned 2 bits:
84    /// one for the left child and one for the right child. A bit is set if the corresponding
85    /// child differs from the canonical empty hash at its depth. This avoids storing empty
86    /// children.
87    ///
88    /// **Hash data** — For every set bit in the mask, the corresponding 32-byte `Word` hash
89    /// is appended to the data section. Hashes are written in breadth-first (local index)
90    /// order, scanning children left-then-right.
91    ///
92    /// On deserialization, omitted children are reconstructed using `EmptySubtreeRoots`.
93    pub fn to_vec(&self) -> Vec<u8> {
94        let mut data = Vec::with_capacity(self.len() * Self::HASH_SIZE);
95        let mut bitmask = [0u8; Self::BITMASK_SIZE];
96
97        for local_index in 0..Self::MAX_NODES {
98            if let Some(node) = self.nodes.get(&local_index) {
99                let bit_offset = (local_index as usize) * Self::BITS_PER_NODE;
100                let node_depth_in_subtree = Self::local_index_to_depth(local_index);
101                let child_depth = self.root_index.depth() + node_depth_in_subtree + 1;
102                let empty_hash = *EmptySubtreeRoots::entry(SMT_DEPTH, child_depth);
103
104                if node.left != empty_hash {
105                    Self::set_bit(&mut bitmask, bit_offset);
106                    data.extend_from_slice(&node.left.as_bytes());
107                }
108
109                if node.right != empty_hash {
110                    Self::set_bit(&mut bitmask, bit_offset + 1);
111                    data.extend_from_slice(&node.right.as_bytes());
112                }
113            }
114        }
115
116        let mut result = Vec::with_capacity(Self::BITMASK_SIZE + data.len());
117        result.extend_from_slice(&bitmask);
118        result.extend_from_slice(&data);
119        result
120    }
121
122    #[inline]
123    fn set_bit(bitmask: &mut [u8], bit_offset: usize) {
124        bitmask[bit_offset / 8] |= 1 << (bit_offset % 8);
125    }
126
127    #[inline]
128    fn get_bit(bitmask: &[u8], bit_offset: usize) -> bool {
129        (bitmask[bit_offset / 8] >> (bit_offset % 8)) & 1 != 0
130    }
131
132    /// Deserializes a subtree from its compact byte representation.
133    ///
134    /// The first 512 bits form the bitmask, which indicates which child hashes
135    /// are present for each internal node (2 bits per node). For every set bit,
136    /// a `Word` hash is read sequentially from the data section.
137    ///
138    /// When a child bit is unset, the corresponding hash is reconstructed from
139    /// `EmptySubtreeRoots` based on the child’s depth in the full tree.
140    ///
141    /// Errors are returned if the byte slice is too short, contains an unexpected
142    /// number of hashes, or leaves unconsumed data at the end.
143    pub fn from_vec(root_index: NodeIndex, data: &[u8]) -> Result<Self, SubtreeError> {
144        if data.len() < Self::BITMASK_SIZE {
145            return Err(SubtreeError::TooShort {
146                found: data.len(),
147                min: Self::BITMASK_SIZE,
148            });
149        }
150        let (bitmask, hash_data) = data.split_at(Self::BITMASK_SIZE);
151        let present_hashes: usize = bitmask.iter().map(|&byte| byte.count_ones() as usize).sum();
152        if hash_data.len() != present_hashes * Self::HASH_SIZE {
153            return Err(SubtreeError::BadHashLen {
154                expected: present_hashes * Self::HASH_SIZE,
155                found: hash_data.len(),
156            });
157        }
158
159        let mut nodes = Map::new();
160        let mut hash_chunks = hash_data.chunks_exact(Self::HASH_SIZE);
161
162        // Process each potential node position
163        for local_index in 0..Self::MAX_NODES {
164            let bit_offset = (local_index as usize) * Self::BITS_PER_NODE;
165            let has_left = Self::get_bit(bitmask, bit_offset);
166            let has_right = Self::get_bit(bitmask, bit_offset + 1);
167
168            if has_left || has_right {
169                // Calculate depth for empty hash lookup
170                let node_depth_in_subtree = Self::local_index_to_depth(local_index);
171                let child_depth = root_index.depth() + node_depth_in_subtree + 1;
172                let empty_hash = *EmptySubtreeRoots::entry(SMT_DEPTH, child_depth);
173
174                // Get left child hash
175                let left_hash = if has_left {
176                    let hash_bytes = hash_chunks
177                        .next()
178                        .ok_or(SubtreeError::MissingLeft { index: local_index })?;
179                    Word::try_from(hash_bytes)
180                        .map_err(|_| SubtreeError::BadLeft { index: local_index })?
181                } else {
182                    empty_hash
183                };
184
185                // Get right child hash
186                let right_hash = if has_right {
187                    let hash_bytes = hash_chunks
188                        .next()
189                        .ok_or(SubtreeError::MissingRight { index: local_index })?;
190                    Word::try_from(hash_bytes)
191                        .map_err(|_| SubtreeError::BadRight { index: local_index })?
192                } else {
193                    empty_hash
194                };
195
196                let inner_node = InnerNode { left: left_hash, right: right_hash };
197                nodes.insert(local_index, inner_node);
198            }
199        }
200
201        // Ensure all hash data was consumed
202        if hash_chunks.next().is_some() {
203            return Err(SubtreeError::ExtraData);
204        }
205
206        Ok(Self { root_index, nodes })
207    }

209    fn global_to_local(global: NodeIndex, base: NodeIndex) -> u8 {
210        assert!(
211            global.depth() >= base.depth(),
212            "Global depth is less than base depth = {}, global depth = {}",
213            base.depth(),
214            global.depth()
215        );
216
217        // Calculate the relative depth within the subtree
218        let relative_depth = global.depth() - base.depth();
219        // Calculate the base offset in a binary tree of given relative depth
220        let base_offset = (1 << relative_depth) - 1;
221        // Mask out the lower `relative_depth` bits to find the local position in the subtree
222        let mask = (1 << relative_depth) - 1;
223        let local_position = (global.position() & mask) as u8;
224        base_offset + local_position
225    }

227    pub fn subtree_key(root_index: NodeIndex) -> [u8; 9] {
228        let mut key = [0u8; 9];
229        key[0] = root_index.depth();
230        key[1..].copy_from_slice(&root_index.position().to_be_bytes());
231        key
232    }
233
234    pub fn find_subtree_root(node_index: NodeIndex) -> NodeIndex {
235        let depth = node_index.depth();
236        if depth < SUBTREE_DEPTH {
237            NodeIndex::root()
238        } else {
239            let subtree_root_depth = depth - (depth % SUBTREE_DEPTH);
240            let relative_depth = depth - subtree_root_depth;
241            let base_value = node_index.position() >> relative_depth;
242
243            NodeIndex::new(subtree_root_depth, base_value).unwrap()
244        }
245    }
246
247    pub fn is_empty(&self) -> bool {
248        self.len() == 0
249    }

251    /// Convert local index to depth within subtree
252    #[inline]
253    const fn local_index_to_depth(local_index: u8) -> u8 {
254        let n = local_index as u16 + 1;
255        (u16::BITS as u8 - 1) - n.leading_zeros() as u8
256    }
