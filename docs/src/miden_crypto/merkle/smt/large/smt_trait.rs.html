18impl<S: SmtStorage> SparseMerkleTree<SMT_DEPTH> for LargeSmt<S> {
19    type Key = Word;
20    type Value = Word;
21    type Leaf = SmtLeaf;
22    type Opening = SmtProof;

24    const EMPTY_VALUE: Self::Value = EMPTY_WORD;
25    const EMPTY_ROOT: Word = *EmptySubtreeRoots::entry(SMT_DEPTH, 0);

41        self.in_memory_nodes[ROOT_MEMORY_INDEX] = root;
42    }

44    fn get_inner_node(&self, index: NodeIndex) -> InnerNode {
45        if index.depth() < IN_MEMORY_DEPTH {
46            let memory_index = to_memory_index(&index);
47            // Reconstruct InnerNode from flat layout: left at 2*i, right at 2*i+1
48            return InnerNode {
49                left: self.in_memory_nodes[memory_index * 2],
50                right: self.in_memory_nodes[memory_index * 2 + 1],
51            };
52        }
53
54        self.storage
55            .get_inner_node(index)
56            .expect("Failed to get inner node")
57            .unwrap_or_else(|| EmptySubtreeRoots::get_inner_node(SMT_DEPTH, index.depth()))
58    }
59
60    fn insert_inner_node(&mut self, index: NodeIndex, inner_node: InnerNode) -> Option<InnerNode> {
61        if index.depth() < IN_MEMORY_DEPTH {
62            let i = to_memory_index(&index);
63            // Get the old node before replacing
64            let old_left = self.in_memory_nodes[i * 2];
65            let old_right = self.in_memory_nodes[i * 2 + 1];
66
67            // Store new node in flat layout
68            self.in_memory_nodes[i * 2] = inner_node.left;
69            self.in_memory_nodes[i * 2 + 1] = inner_node.right;
70
71            // Check if the old node was empty
72            if is_empty_parent(old_left, old_right, index.depth() + 1) {
73                return None;
74            }
75
76            return Some(InnerNode { left: old_left, right: old_right });
77        }
78        self.storage
79            .set_inner_node(index, inner_node)
80            .expect("Failed to store inner node")
81    }
82
83    fn remove_inner_node(&mut self, index: NodeIndex) -> Option<InnerNode> {
84        if index.depth() < IN_MEMORY_DEPTH {
85            let memory_index = to_memory_index(&index);
86            // Get the old node before replacing with empty hashes
87            let old_left = self.in_memory_nodes[memory_index * 2];
88            let old_right = self.in_memory_nodes[memory_index * 2 + 1];
89
90            // Replace with empty hashes
91            let child_depth = index.depth() + 1;
92            let empty_hash = *EmptySubtreeRoots::entry(SMT_DEPTH, child_depth);
93            self.in_memory_nodes[memory_index * 2] = empty_hash;
94            self.in_memory_nodes[memory_index * 2 + 1] = empty_hash;
95
96            // Return the old node if it wasn't already empty
97            if is_empty_parent(old_left, old_right, child_depth) {
98                return None;
99            }
100
101            return Some(InnerNode { left: old_left, right: old_right });
102        }
103        self.storage.remove_inner_node(index).expect("Failed to remove inner node")
104    }
105
106    fn insert(&mut self, key: Self::Key, value: Self::Value) -> Result<Self::Value, MerkleError> {
107        let old_value = self.get_value(&key);
108        // if the old value and new value are the same, there is nothing to update
109        if value == old_value {
110            return Ok(value);
111        }
112
113        let mutations = self.compute_mutations([(key, value)])?;
114        self.apply_mutations(mutations).expect("Failed to apply mutations in insert");
115
116        Ok(old_value)
117    }
118
119    fn insert_value(
120        &mut self,
121        key: Self::Key,
122        value: Self::Value,
123    ) -> Result<Option<Self::Value>, MerkleError> {
124        // inserting an `EMPTY_VALUE` is equivalent to removing any value associated with `key`
125        let index = Self::key_to_leaf_index(&key).position();
126        if value != Self::EMPTY_VALUE {
127            match self.storage.insert_value(index, key, value) {
128                Ok(prev) => Ok(prev),
129                Err(StorageError::Leaf(SmtLeafError::TooManyLeafEntries { actual })) => {
130                    Err(MerkleError::TooManyLeafEntries { actual })
131                },
132                Err(_) => {
133                    panic!("Storage error during insert_value");
134                },
135            }
136        } else {
137            Ok(self.storage.remove_value(index, key).map_err(LargeSmtError::from)?)
138        }
139    }
140
141    fn get_value(&self, key: &Self::Key) -> Self::Value {
142        let leaf_pos = LeafIndex::<SMT_DEPTH>::from(*key);
143        match self.storage.get_leaf(leaf_pos.position()) {
144            Ok(Some(leaf)) => leaf.get_value(key).unwrap_or_default(),
145            Ok(None) => EMPTY_WORD,
146            Err(_) => {
147                panic!("Storage error during get_leaf in get_value");
148            },
149        }
150    }
151
152    fn get_leaf(&self, key: &Word) -> Self::Leaf {
153        let leaf_pos = LeafIndex::<SMT_DEPTH>::from(*key).position();
154        match self.storage.get_leaf(leaf_pos) {
155            Ok(Some(leaf)) => leaf,
156            Ok(None) => SmtLeaf::new_empty((*key).into()),
157            Err(_) => {
158                panic!("Storage error during get_leaf in get_leaf");
159            },
160        }
161    }
162
163    fn hash_leaf(leaf: &Self::Leaf) -> Word {
164        leaf.hash()
165    }
166
167    fn construct_prospective_leaf(
168        &self,
169        mut existing_leaf: SmtLeaf,
170        key: &Word,
171        value: &Word,
172    ) -> Result<SmtLeaf, SmtLeafError> {
173        debug_assert_eq!(existing_leaf.index(), Self::key_to_leaf_index(key));
174
175        match existing_leaf {
176            SmtLeaf::Empty(_) => Ok(SmtLeaf::new_single(*key, *value)),
177            _ => {
178                if *value != EMPTY_WORD {
179                    existing_leaf.insert(*key, *value)?;
180                } else {
181                    existing_leaf.remove(*key);
182                }
183
184                Ok(existing_leaf)
185            },
186        }
187    }

189    fn open(&self, key: &Self::Key) -> Self::Opening {
190        let leaf = self.get_leaf(key);
191
192        let mut idx: NodeIndex = LeafIndex::from(*key).into();
193
194        let subtree_roots: Vec<NodeIndex> = (0..NUM_SUBTREE_LEVELS)
195            .scan(idx.parent(), |cursor, _| {
196                let subtree_root = Subtree::find_subtree_root(*cursor);
197                *cursor = subtree_root.parent();
198                Some(subtree_root)
199            })
200            .collect();
201        // cache subtrees in memory
202        let mut cache = Map::<NodeIndex, Subtree>::new();
203        for &root in &subtree_roots {
204            let subtree =
205                match self.storage.get_subtree(root).expect("storage error fetching subtree") {
206                    Some(st) => st,
207                    None => Subtree::new(root),
208                };
209            cache.insert(root, subtree);
210        }
211        let mut path = Vec::with_capacity(idx.depth() as usize);
212        while idx.depth() > 0 {
213            let is_right = idx.is_position_odd();
214            idx = idx.parent();
215
216            let sibling_hash = if idx.depth() < IN_MEMORY_DEPTH {
217                // top levels in memory
218                let InnerNode { left, right } = self.get_inner_node(idx);
219                if is_right { left } else { right }
220            } else {
221                // deep levels come from our 5 preloaded subtrees
222                let root = Subtree::find_subtree_root(idx);
223                let subtree = &cache[&root];
224                let InnerNode { left, right } = subtree
225                    .get_inner_node(idx)
226                    .unwrap_or_else(|| EmptySubtreeRoots::get_inner_node(SMT_DEPTH, idx.depth()));
227                if is_right { left } else { right }
228            };
229
230            path.push(sibling_hash);
231        }
232
233        let merkle_path =
234            SparseMerklePath::from_sized_iter(path).expect("failed to convert to SparseMerklePath");
235        Self::path_and_leaf_to_opening(merkle_path, leaf)
236    }
237
238    fn key_to_leaf_index(key: &Word) -> LeafIndex<SMT_DEPTH> {
239        let most_significant_felt = key[3];
240        LeafIndex::new_max_depth(most_significant_felt.as_canonical_u64())
241    }
242
243    fn path_and_leaf_to_opening(path: SparseMerklePath, leaf: SmtLeaf) -> SmtProof {
244        SmtProof::new_unchecked(path, leaf)
245    }
