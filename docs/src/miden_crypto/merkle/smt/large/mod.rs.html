1//! Large-scale Sparse Merkle Tree backed by pluggable storage.
2//!
3//! `LargeSmt` stores the top of the tree (depths 0–23) in memory and persists the lower
4//! depths (24–64) in storage as fixed-size subtrees. This hybrid layout scales beyond RAM
5//! while keeping common operations fast. With the `rocksdb` feature enabled, the lower
6//! subtrees and leaves are stored in RocksDB. On reload, the in-memory top is reconstructed
7//! from cached depth-24 subtree roots.
8//!
9//! Examples below require the `rocksdb` feature.
10//!
11//! Load an existing RocksDB-backed tree with root validation:
12//! ```no_run
13//! # #[cfg(feature = "rocksdb")]
14//! # {
15//! use miden_crypto::{
16//!     Word,
17//!     merkle::smt::{LargeSmt, RocksDbConfig, RocksDbStorage},
18//! };
19//!
20//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
21//! # let expected_root: Word = miden_crypto::EMPTY_WORD;
22//! let storage = RocksDbStorage::open(RocksDbConfig::new("/path/to/db"))?;
23//! let smt = LargeSmt::load_with_root(storage, expected_root)?;
24//! assert_eq!(smt.root(), expected_root);
25//! # Ok(())
26//! # }
27//! # }
28//! ```
29//!
30//! Load an existing tree without root validation (use with caution):
31//! ```no_run
32//! # #[cfg(feature = "rocksdb")]
33//! # {
34//! use miden_crypto::merkle::smt::{LargeSmt, RocksDbConfig, RocksDbStorage};
35//!
36//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
37//! let storage = RocksDbStorage::open(RocksDbConfig::new("/path/to/db"))?;
38//! let smt = LargeSmt::load(storage)?;
39//! let _root = smt.root();
40//! # Ok(())
41//! # }
42//! # }
43//! ```
44//!
45//! Initialize an empty RocksDB-backed tree and bulk-load entries:
46//! ```no_run
47//! # #[cfg(feature = "rocksdb")]
48//! # {
49//! use miden_crypto::{
50//!     Felt, Word,
51//!     merkle::smt::{LargeSmt, RocksDbConfig, RocksDbStorage},
52//! };
53//!
54//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
55//! let path = "/path/to/new-db";
56//! if std::path::Path::new(path).exists() {
57//!     std::fs::remove_dir_all(path)?;
58//! }
59//! std::fs::create_dir_all(path)?;
60//!
61//! let storage = RocksDbStorage::open(RocksDbConfig::new(path))?;
62//! let mut smt = LargeSmt::new(storage)?; // empty tree
63//!
64//! // Prepare initial entries
65//! let entries = vec![
66//!     (
67//!         Word::new([Felt::new(1), Felt::new(0), Felt::new(0), Felt::new(0)]),
68//!         Word::new([Felt::new(10), Felt::new(20), Felt::new(30), Felt::new(40)]),
69//!     ),
70//!     (
71//!         Word::new([Felt::new(2), Felt::new(0), Felt::new(0), Felt::new(0)]),
72//!         Word::new([Felt::new(11), Felt::new(22), Felt::new(33), Felt::new(44)]),
73//!     ),
74//! ];
75//!
76//! // Bulk insert entries (faster than compute_mutations + apply_mutations)
77//! smt.insert_batch(entries)?;
78//! # Ok(())
79//! # }
80//! # }
81//! ```
82//!
83//! Apply batch updates (insertions and deletions):
84//! ```no_run
85//! # #[cfg(feature = "rocksdb")]
86//! # {
87//! use miden_crypto::{
88//!     EMPTY_WORD, Felt, Word,
89//!     merkle::smt::{LargeSmt, RocksDbConfig, RocksDbStorage},
90//! };
91//!
92//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
93//! let storage = RocksDbStorage::open(RocksDbConfig::new("/path/to/db"))?;
94//! let mut smt = LargeSmt::load(storage)?;
95//!
96//! let k1 = Word::new([Felt::new(101), Felt::new(0), Felt::new(0), Felt::new(0)]);
97//! let v1 = Word::new([Felt::new(1), Felt::new(2), Felt::new(3), Felt::new(4)]);
98//! let k2 = Word::new([Felt::new(202), Felt::new(0), Felt::new(0), Felt::new(0)]);
99//! let k3 = Word::new([Felt::new(303), Felt::new(0), Felt::new(0), Felt::new(0)]);
100//! let v3 = Word::new([Felt::new(7), Felt::new(7), Felt::new(7), Felt::new(7)]);
101//!
102//! // EMPTY_WORD marks deletions
103//! let updates = vec![(k1, v1), (k2, EMPTY_WORD), (k3, v3)];
104//! smt.insert_batch(updates)?;
105//! # Ok(())
106//! # }
107//! # }
108//! ```
109//!
110//! Quick initialization with `with_entries` (best for modest datasets/tests):
111//! ```no_run
112//! # #[cfg(feature = "rocksdb")]
113//! # {
114//! use miden_crypto::{
115//!     Felt, Word,
116//!     merkle::smt::{LargeSmt, RocksDbConfig, RocksDbStorage},
117//! };
118//!
119//! # fn main() -> Result<(), Box<dyn std::error::Error>> {
120//! // Note: `with_entries` expects an EMPTY storage and performs an all-at-once build.
121//! // Prefer `insert_batch` for large bulk loads.
122//! let path = "/path/to/new-db";
123//! if std::path::Path::new(path).exists() {
124//!     std::fs::remove_dir_all(path)?;
125//! }
126//! std::fs::create_dir_all(path)?;
127//!
128//! let storage = RocksDbStorage::open(RocksDbConfig::new(path))?;
129//! let entries = vec![
130//!     (
131//!         Word::new([Felt::new(1), Felt::new(0), Felt::new(0), Felt::new(0)]),
132//!         Word::new([Felt::new(10), Felt::new(20), Felt::new(30), Felt::new(40)]),
133//!     ),
134//!     (
135//!         Word::new([Felt::new(2), Felt::new(0), Felt::new(0), Felt::new(0)]),
136//!         Word::new([Felt::new(11), Felt::new(22), Felt::new(33), Felt::new(44)]),
137//!     ),
138//! ];
139//! let _smt = LargeSmt::with_entries(storage, entries)?;
140//! # Ok(())
141//! # }
142//! # }
143//! ```
144//!
145//! ## Performance and Memory Considerations
146//!
147//! The `apply_mutations()` and `apply_mutations_with_reversion()` methods use batched
148//! operations: they preload all affected subtrees and leaves before applying changes
149//! atomically. This approach reduces I/O at the cost of higher temporary memory usage.
150//!
151//! ### Memory Usage
152//!
153//! Peak memory is proportional to:
154//! - The number of mutated leaves
155//! - The number of distinct storage subtrees touched by those mutations
156//!
157//! This memory is temporary and released immediately after the batch commits.
158//!
159//! ### Locality Matters
160//!
161//! Memory usage scales with how dispersed updates are, not just their count:
162//! - **Localized updates**: Keys with shared high-order bits fall into the same storage subtrees
163//! - **Scattered updates**: Keys spread across many storage subtrees require loading more distinct
164//!   subtrees
165//!
166//! ### Guidelines
167//!
168//! For typical batches (up to ~10,000 updates) with reasonable locality, the working set
169//! is modest. Very large or highly scattered batches will use more
170//! memory proportionally.
171//!
172//! To optimize memory and I/O: group updates by key locality so that keys sharing
173//! high-order bits are processed together.

224/// SMT construction phase**.

251/// A large-scale Sparse Merkle tree mapping 256-bit keys to 256-bit values, backed by pluggable
252/// storage. Both keys and values are represented by 4 field elements.
253///
254/// Unlike the regular `Smt`, this implementation is designed for very large trees by using external
255/// storage (such as RocksDB) for the bulk of the tree data, while keeping only the upper levels (up
256/// to depth 24) in memory. This hybrid approach allows the tree to scale beyond memory limitations
257/// while maintaining good performance for common operations.
258///
259/// All leaves sit at depth 64. The most significant element of the key is used to identify the leaf
260/// to which the key maps.
261///
262/// A leaf is either empty, or holds one or more key-value pairs. An empty leaf hashes to the empty
263/// word. Otherwise, a leaf hashes to the hash of its key-value pairs, ordered by key first, value
264/// second.
265///
266/// The tree structure:
267/// - Depths 0-23: Stored in memory as a flat array for fast access
268/// - Depths 24-64: Stored in external storage organized as subtrees for efficient batch operations
269///
270/// `LargeSmt` does **not** implement [`Clone`]. Copying an instance would only duplicate the
271/// in-memory nodes while continuing to share the storage backend, which is misleading. If you need
272/// to share an instance between threads or components, wrap it in an
273/// [`Arc`](alloc::sync::Arc) explicitly so the ownership semantics are clear.
274#[derive(Debug)]
275pub struct LargeSmt<S: SmtStorage> {
276    storage: S,
277    /// Flat vector representation of in-memory nodes.
278    /// Index 0 is unused; index 1 is root.
279    /// For node at index i: left child at 2*i, right child at 2*i+1.
280    in_memory_nodes: Vec<Word>,
281    /// Cached count of non-empty leaves. Initialized from storage on load,
282    /// updated after each mutation.
283    leaf_count: usize,
284    /// Cached count of key-value entries across all leaves. Initialized from
285    /// storage on load, updated after each mutation.
286    entry_count: usize,
287}
288
289impl<S: SmtStorage> LargeSmt<S> {
290    // CONSTANTS
291    // --------------------------------------------------------------------------------------------
292    /// The default value used to compute the hash of empty leaves.
293    pub const EMPTY_VALUE: Word = <Self as SparseMerkleTree<SMT_DEPTH>>::EMPTY_VALUE;
294
295    /// Subtree depths for the subtrees stored in storage.
296    pub const SUBTREE_DEPTHS: [u8; 5] = [56, 48, 40, 32, 24];
297
298    // PUBLIC ACCESSORS
299    // --------------------------------------------------------------------------------------------
300
301    /// Returns the depth of the tree
302    pub const fn depth(&self) -> u8 {
303        SMT_DEPTH
304    }
305
306    /// Returns the root of the tree
307    pub fn root(&self) -> Word {
308        self.in_memory_nodes[ROOT_MEMORY_INDEX]
309    }
310
311    /// Returns the number of non-empty leaves in this tree.
312    ///
313    /// Note that this may return a different value from [Self::num_entries()] as a single leaf may
314    /// contain more than one key-value pair.
315    pub fn num_leaves(&self) -> usize {
316        self.leaf_count
317    }
318
319    /// Returns the number of key-value pairs with non-default values in this tree.
320    ///
321    /// Note that this may return a different value from [Self::num_leaves()] as a single leaf may
322    /// contain more than one key-value pair.
323    pub fn num_entries(&self) -> usize {
324        self.entry_count
325    }
326
327    /// Returns the leaf to which `key` maps
328    pub fn get_leaf(&self, key: &Word) -> SmtLeaf {
329        <Self as SparseMerkleTree<SMT_DEPTH>>::get_leaf(self, key)
330    }
331
332    /// Returns the value associated with `key`
333    pub fn get_value(&self, key: &Word) -> Word {
334        <Self as SparseMerkleTree<SMT_DEPTH>>::get_value(self, key)
335    }
336
337    /// Returns an opening of the leaf associated with `key`. Conceptually, an opening is a Merkle
338    /// path to the leaf, as well as the leaf itself.
339    pub fn open(&self, key: &Word) -> SmtProof {
340        <Self as SparseMerkleTree<SMT_DEPTH>>::open(self, key)
341    }
342
343    /// Returns a boolean value indicating whether the SMT is empty.
344    pub fn is_empty(&self) -> bool {
345        let root = self.root();
346        debug_assert_eq!(self.leaf_count == 0, root == Self::EMPTY_ROOT);
347        root == Self::EMPTY_ROOT
348    }
349
350    // ITERATORS
351    // --------------------------------------------------------------------------------------------
352
353    /// Returns an iterator over the leaves of this [`LargeSmt`].
354    /// Note: This iterator returns owned SmtLeaf values.
355    ///
356    /// # Errors
357    /// Returns an error if the storage backend fails to create the iterator.
358    pub fn leaves(
359        &self,
360    ) -> Result<impl Iterator<Item = (LeafIndex<SMT_DEPTH>, SmtLeaf)>, LargeSmtError> {
361        let iter = self.storage.iter_leaves()?;
362        Ok(iter.map(|(idx, leaf)| (LeafIndex::new_max_depth(idx), leaf)))
363    }
364
365    /// Returns an iterator over the key-value pairs of this [`LargeSmt`].
366    /// Note: This iterator returns owned (Word, Word) tuples.
367    ///
368    /// # Errors
369    /// Returns an error if the storage backend fails to create the iterator.
370    pub fn entries(&self) -> Result<impl Iterator<Item = (Word, Word)>, LargeSmtError> {
371        let leaves_iter = self.leaves()?;
372        Ok(leaves_iter.flat_map(|(_, leaf)| {
373            // Collect the (Word, Word) tuples into an owned Vec
374            // This ensures they outlive the 'leaf' from which they are derived.
375            let owned_entries: Vec<(Word, Word)> = leaf.entries().to_vec();
376            // Return an iterator over this owned Vec
377            owned_entries.into_iter()
378        }))
379    }
380
381    /// Returns an iterator over the inner nodes of this [`LargeSmt`].
382    ///
383    /// # Errors
384    /// Returns an error if the storage backend fails during iteration setup.
385    pub fn inner_nodes(&self) -> Result<impl Iterator<Item = InnerNodeInfo> + '_, LargeSmtError> {
386        // Pre-validate that storage is accessible
387        let _ = self.storage.iter_subtrees()?;
388        Ok(LargeSmtInnerNodeIterator::new(self))
389    }
390
391    // STATE MUTATORS
392    // --------------------------------------------------------------------------------------------
393
394    /// Inserts a value at the specified key, returning the previous value associated with that key.
395    /// Recall that by definition, any key that hasn't been updated is associated with
396    /// [`Self::EMPTY_VALUE`].
397    ///
398    /// This also recomputes all hashes between the leaf (associated with the key) and the root,
399    /// updating the root itself.
400    ///
401    /// # Errors
402    /// Returns an error if inserting the key-value pair would exceed
403    /// [`MAX_LEAF_ENTRIES`](super::MAX_LEAF_ENTRIES) (1024 entries) in the leaf.
404    pub fn insert(&mut self, key: Word, value: Word) -> Result<Word, MerkleError> {
405        <Self as SparseMerkleTree<SMT_DEPTH>>::insert(self, key, value)
406    }
407
408    // HELPERS
409    // --------------------------------------------------------------------------------------------
410
411    /// Helper to get an in-memory node if not empty.
412    ///
413    /// # Panics
414    /// With debug assertions on, panics if `index.depth() >= IN_MEMORY_DEPTH`.
415    fn get_non_empty_inner_node(&self, index: NodeIndex) -> Option<InnerNode> {
416        debug_assert!(index.depth() < IN_MEMORY_DEPTH, "Only for in-memory nodes");
417
418        let memory_index = to_memory_index(&index);
419
420        let left = self.in_memory_nodes[memory_index * 2];
421        let right = self.in_memory_nodes[memory_index * 2 + 1];
422
423        // Check if both children are empty
424        let child_depth = index.depth() + 1;
425        if is_empty_parent(left, right, child_depth) {
426            None
427        } else {
428            Some(InnerNode { left, right })
429        }
430    }

433    // --------------------------------------------------------------------------------------------

435    #[cfg(test)]
436    pub(crate) fn in_memory_nodes(&self) -> &Vec<Word> {
437        &self.in_memory_nodes
438    }

444/// Checks if a node with the given children is empty.
445/// A node is considered empty if both children equal the empty hash for that depth.
446pub(super) fn is_empty_parent(left: Word, right: Word, child_depth: u8) -> bool {
447    let empty_hash = *EmptySubtreeRoots::entry(SMT_DEPTH, child_depth);
448    left == empty_hash && right == empty_hash
449}
450
451/// Converts a NodeIndex to a flat vector index using 1-indexed layout.
452/// Index 0 is unused, index 1 is root.
453/// For a node at index i: left child at 2*i, right child at 2*i+1.
454pub(super) fn to_memory_index(index: &NodeIndex) -> usize {
455    debug_assert!(index.depth() < IN_MEMORY_DEPTH);
456    debug_assert!(index.position() < (1 << index.depth()));
457    (1usize << index.depth()) + index.position() as usize
458}

460impl<S: SmtStorage> PartialEq for LargeSmt<S> {
461    /// Compares two LargeSmt instances based on their root hash and metadata.
462    ///
463    /// Note: This comparison only checks the root hash and counts, not the underlying
464    /// storage contents. Two SMTs with the same root should be cryptographically
465    /// equivalent, but this doesn't verify the storage backends are identical.
466    fn eq(&self, other: &Self) -> bool {
467        self.root() == other.root()
468            && self.leaf_count == other.leaf_count
469            && self.entry_count == other.entry_count
470    }
471}
