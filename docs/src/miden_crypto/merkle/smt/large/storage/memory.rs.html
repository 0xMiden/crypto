15/// In-memory storage for a Sparse Merkle Tree (SMT), implementing the `SmtStorage` trait.
16///
17/// This structure stores the SMT's leaf nodes and subtrees directly in memory.
18///
19/// It is primarily intended for scenarios where data persistence to disk is not a
20/// primary concern. Common use cases include:
21/// - Testing environments.
22/// - Managing SMT instances with a limited operational lifecycle.
23/// - Situations where a higher-level application architecture handles its own data persistence
24///   strategy.
25#[derive(Debug, Clone)]
26pub struct MemoryStorage {
27    pub leaves: Map<u64, SmtLeaf>,
28    pub subtrees: Map<NodeIndex, Subtree>,
29}
30
31impl MemoryStorage {
32    /// Creates a new, empty in-memory storage for a Sparse Merkle Tree.
33    ///
34    /// Initializes empty maps for leaves and subtrees.
35    pub fn new() -> Self {
36        Self { leaves: Map::new(), subtrees: Map::new() }
37    }
38}
39
40impl Default for MemoryStorage {
41    fn default() -> Self {
42        Self::new()
43    }
44}

46impl SmtStorage for MemoryStorage {
47    /// Gets the total number of non-empty leaves currently stored.
48    fn leaf_count(&self) -> Result<usize, StorageError> {
49        Ok(self.leaves.len())
50    }
51
52    /// Gets the total number of key-value entries currently stored.
53    fn entry_count(&self) -> Result<usize, StorageError> {
54        Ok(self.leaves.values().map(|leaf| leaf.num_entries()).sum())
55    }
56
57    /// Inserts a key-value pair into the leaf at the given index.
58    ///
59    /// - If the leaf at `index` does not exist, a new `SmtLeaf::Single` is created.
60    /// - If the leaf exists, the key-value pair is inserted into it.
61    /// - Returns the previous value associated with the key, if any.
62    ///
63    /// # Panics
64    /// Panics in debug builds if `value` is `EMPTY_WORD`.
65    fn insert_value(
66        &mut self,
67        index: u64,
68        key: Word,
69        value: Word,
70    ) -> Result<Option<Word>, StorageError> {
71        debug_assert_ne!(value, EMPTY_WORD);
72
73        match self.leaves.get_mut(&index) {
74            Some(leaf) => Ok(leaf.insert(key, value)?),
75            None => {
76                self.leaves.insert(index, SmtLeaf::Single((key, value)));
77                Ok(None)
78            },
79        }
80    }
81
82    /// Removes a key-value pair from the leaf at the given `index`.
83    ///
84    /// - If the leaf at `index` exists and the `key` is found within that leaf, the key-value pair
85    ///   is removed, and the old `Word` value is returned in `Ok(Some(Word))`.
86    /// - If the leaf at `index` exists but the `key` is not found within that leaf, `Ok(None)` is
87    ///   returned (as `leaf.get_value(&key)` would be `None`).
88    /// - If the leaf at `index` does not exist, `Ok(None)` is returned, as no value could be
89    ///   removed.
90    fn remove_value(&mut self, index: u64, key: Word) -> Result<Option<Word>, StorageError> {
91        let old_value = match self.leaves.entry(index) {
92            MapEntry::Occupied(mut entry) => {
93                let (old_value, is_empty) = entry.get_mut().remove(key);
94                if is_empty {
95                    entry.remove();
96                }
97                old_value
98            },
99            // Leaf at index does not exist, so no value could be removed.
100            MapEntry::Vacant(_) => None,
101        };
102        Ok(old_value)
103    }
104
105    /// Retrieves a single leaf node.
106    fn get_leaf(&self, index: u64) -> Result<Option<SmtLeaf>, StorageError> {
107        Ok(self.leaves.get(&index).cloned())
108    }
109
110    /// Sets multiple leaf nodes in storage.
111    ///
112    /// If a leaf at a given index already exists, it is overwritten.
113    fn set_leaves(&mut self, leaves_map: Map<u64, SmtLeaf>) -> Result<(), StorageError> {
114        self.leaves.extend(leaves_map);
115        Ok(())
116    }
117
118    /// Removes a single leaf node.
119    fn remove_leaf(&mut self, index: u64) -> Result<Option<SmtLeaf>, StorageError> {
120        Ok(self.leaves.remove(&index))
121    }
122
123    /// Retrieves multiple leaf nodes. Returns Ok(None) for indices not found.
124    fn get_leaves(&self, indices: &[u64]) -> Result<Vec<Option<SmtLeaf>>, StorageError> {
125        let leaves = indices.iter().map(|idx| self.leaves.get(idx).cloned()).collect();
126        Ok(leaves)
127    }
128
129    /// Returns true if the storage has any leaves.
130    fn has_leaves(&self) -> Result<bool, StorageError> {
131        Ok(!self.leaves.is_empty())
132    }
133
134    /// Retrieves a single Subtree (representing deep nodes) by its root NodeIndex.
135    /// Assumes index.depth() >= IN_MEMORY_DEPTH. Returns Ok(None) if not found.
136    fn get_subtree(&self, index: NodeIndex) -> Result<Option<Subtree>, StorageError> {
137        Ok(self.subtrees.get(&index).cloned())
138    }
139
140    /// Retrieves multiple Subtrees.
141    /// Assumes index.depth() >= IN_MEMORY_DEPTH for all indices. Returns Ok(None) for indices not
142    /// found.
143    fn get_subtrees(&self, indices: &[NodeIndex]) -> Result<Vec<Option<Subtree>>, StorageError> {
144        let subtrees: Vec<_> = indices.iter().map(|idx| self.subtrees.get(idx).cloned()).collect();
145        Ok(subtrees)
146    }
147
148    /// Sets a single Subtree (representing deep nodes) by its root NodeIndex.
149    ///
150    /// If a subtree with the same root NodeIndex already exists, it is overwritten.
151    /// Assumes `subtree.root_index().depth() >= IN_MEMORY_DEPTH`.
152    fn set_subtree(&mut self, subtree: &Subtree) -> Result<(), StorageError> {
153        self.subtrees.insert(subtree.root_index(), subtree.clone());
154        Ok(())
155    }
156
157    /// Sets multiple Subtrees (representing deep nodes) by their root NodeIndex.
158    ///
159    /// If a subtree with a given root NodeIndex already exists, it is overwritten.
160    /// Assumes `subtree.root_index().depth() >= IN_MEMORY_DEPTH` for all subtrees in the vector.
161    fn set_subtrees(&mut self, subtrees_vec: Vec<Subtree>) -> Result<(), StorageError> {
162        self.subtrees
163            .extend(subtrees_vec.into_iter().map(|subtree| (subtree.root_index(), subtree)));
164        Ok(())
165    }
166
167    /// Removes a single Subtree (representing deep nodes) by its root NodeIndex.
168    fn remove_subtree(&mut self, index: NodeIndex) -> Result<(), StorageError> {
169        self.subtrees.remove(&index);
170        Ok(())
171    }
172
173    /// Retrieves a single inner node from a Subtree.
174    ///
175    /// This function is intended for accessing nodes within a Subtree, meaning
176    /// `index.depth()` must be greater than or equal to `IN_MEMORY_DEPTH`.
177    ///
178    /// # Errors
179    /// - `StorageError::Unsupported`: If `index.depth() < IN_MEMORY_DEPTH`.
180    ///
181    /// Returns `Ok(None)` if the subtree or the specific inner node within it is not found.
182    fn get_inner_node(&self, index: NodeIndex) -> Result<Option<InnerNode>, StorageError> {
183        if index.depth() < IN_MEMORY_DEPTH {
184            return Err(StorageError::Unsupported(
185                "Cannot get inner node from upper part of the tree".into(),
186            ));
187        }
188        let subtree_root_index = Subtree::find_subtree_root(index);
189        Ok(self
190            .subtrees
191            .get(&subtree_root_index)
192            .and_then(|subtree| subtree.get_inner_node(index)))
193    }
194
195    /// Sets a single inner node within a Subtree.
196    ///
197    /// - `index.depth()` must be greater than or equal to `IN_MEMORY_DEPTH`.
198    /// - If the target Subtree does not exist, it is created.
199    /// - The `node` is then inserted into the Subtree.
200    ///
201    /// Returns the `InnerNode` that was previously at this `index`, if any.
202    ///
203    /// # Errors
204    /// - `StorageError::Unsupported`: If `index.depth() < IN_MEMORY_DEPTH`.
205    fn set_inner_node(
206        &mut self,
207        index: NodeIndex,
208        node: InnerNode,
209    ) -> Result<Option<InnerNode>, StorageError> {
210        if index.depth() < IN_MEMORY_DEPTH {
211            return Err(StorageError::Unsupported(
212                "Cannot set inner node in upper part of the tree".into(),
213            ));
214        }
215        let subtree_root_index = Subtree::find_subtree_root(index);
216        let mut subtree = self
217            .subtrees
218            .remove(&subtree_root_index)
219            .unwrap_or_else(|| Subtree::new(subtree_root_index));
220        let old_node = subtree.insert_inner_node(index, node);
221        self.subtrees.insert(subtree_root_index, subtree);
222        Ok(old_node)
223    }
224
225    /// Removes a single inner node from a Subtree.
226    ///
227    /// - `index.depth()` must be greater than or equal to `IN_MEMORY_DEPTH`.
228    /// - If the Subtree becomes empty after removing the node, the Subtree itself is removed from
229    ///   storage.
230    ///
231    /// Returns the `InnerNode` that was removed, if any.
232    ///
233    /// # Errors
234    /// - `StorageError::Unsupported`: If `index.depth() < IN_MEMORY_DEPTH`.
235    fn remove_inner_node(&mut self, index: NodeIndex) -> Result<Option<InnerNode>, StorageError> {
236        if index.depth() < IN_MEMORY_DEPTH {
237            return Err(StorageError::Unsupported(
238                "Cannot remove inner node from upper part of the tree".into(),
239            ));
240        }
241        let subtree_root_index = Subtree::find_subtree_root(index);
242
243        let inner_node: Option<InnerNode> =
244            self.subtrees.remove(&subtree_root_index).and_then(|mut subtree| {
245                let old_node = subtree.remove_inner_node(index);
246                if !subtree.is_empty() {
247                    self.subtrees.insert(subtree_root_index, subtree);
248                }
249                old_node
250            });
251        Ok(inner_node)
252    }
253
254    /// Applies a set of updates atomically to the storage.
255    ///
256    /// This method handles updates to:
257    /// - Leaves: Inserts new or updated leaves, removes specified leaves.
258    /// - Subtrees: Inserts new or updated subtrees, removes specified subtrees.
259    fn apply(&mut self, updates: StorageUpdates) -> Result<(), StorageError> {
260        let StorageUpdateParts {
261            leaf_updates,
262            subtree_updates,
263            leaf_count_delta: _,
264            entry_count_delta: _,
265        } = updates.into_parts();
266
267        for (index, leaf_opt) in leaf_updates {
268            if let Some(leaf) = leaf_opt {
269                self.leaves.insert(index, leaf);
270            } else {
271                self.leaves.remove(&index);
272            }
273        }
274        for update in subtree_updates {
275            match update {
276                SubtreeUpdate::Store { index, subtree } => {
277                    self.subtrees.insert(index, subtree);
278                },
279                SubtreeUpdate::Delete { index } => {
280                    self.subtrees.remove(&index);
281                },
282            }
283        }
284        Ok(())
285    }

287    /// Returns an iterator over all (index, SmtLeaf) pairs in the storage.
288    ///
289    /// The iterator provides access to the current state of the leaves.
290    fn iter_leaves(&self) -> Result<Box<dyn Iterator<Item = (u64, SmtLeaf)> + '_>, StorageError> {
291        let leaves_vec = self.leaves.iter().map(|(&k, v)| (k, v.clone())).collect::<Vec<_>>();
292        Ok(Box::new(leaves_vec.into_iter()))
293    }
294
295    /// Returns an iterator over all Subtrees in the storage.
296    ///
297    /// The iterator provides access to the current subtrees from storage.
298    fn iter_subtrees(&self) -> Result<Box<dyn Iterator<Item = Subtree> + '_>, StorageError> {
299        let subtrees_vec = self.subtrees.values().cloned().collect::<Vec<_>>();
300        Ok(Box::new(subtrees_vec.into_iter()))
301    }

307    fn get_depth24(&self) -> Result<Vec<(u64, Word)>, StorageError> {
308        Ok(Vec::new())
