1//! This module contains the definition of [`History`], a simple container for some number of
2//! historical versions of a given merkle tree.
3//!
4//! This history consists of a series of _deltas_ from the current state of the tree, moving
5//! backward in history away from that current state. These deltas are then used to form an overlay
6//! that represents the changes to be made on top of the current tree to put it _back_ in that
7//! historical state.
8//!
9//! It provides functionality for adding new states to the history, as well as for querying the
10//! history at a given point in time.
11//!
12//! # Complexity
13//!
14//! Versions in this structure are _complete_. This means that the data stored for any given
15//! historical version is sufficient to apply atop the current tree to return it to the state
16//! corresponding to that historical version.
17//!
18//! # Performance
19//!
20//! This structure operates entirely in memory, and is hence reasonably quick to query. As of the
21//! current time, no detailed benchmarking has taken place for the history.
22

49/// A collection of changes to arbitrary non-leaf nodes in a merkle tree.
50///
51/// All changes to nodes between versions `v` and `v + 1` must be explicitly "undone" in the
52/// `NodeChanges` representing version `v`. This includes nodes that were defaulted in version `v`
53/// that were given an explicit value in version `v + 1`, where the `NodeChanges` must explicitly
54/// set those nodes back to the default.
55///
56/// Failure to do so will result in incorrect values when those nodes are queried at a point in the
57/// history corresponding to version `v`.
58pub type NodeChanges = Map<NodeIndex, Word>;
59
60/// The type of the keys that need to be changed by the delta to return the target tree to the state
61/// represented by the delta.
62pub type ChangedKeys = Map<Word, Word>;

64// HISTORY
65// ================================================================================================

67/// A History contains a sequence of versions atop a given tree.
68///
69/// The versions are _cumulative_, meaning that querying the history must account for changes from
70/// the current tree that take place in versions that are not the queried version or the current
71/// tree.
72#[derive(Clone, Debug)]
73pub struct History {
74    /// The maximum number of historical versions to be stored.
75    max_count: usize,
76
77    /// The deltas that make up the history for this tree.
78    ///
79    /// It will never contain more than `max_count` deltas, and is ordered with the oldest data at
80    /// the lowest index.
81    ///
82    /// # Implementation Note
83    ///
84    /// We use a [`VecDeque`] instead of a [`Vec`] so that we can both have efficient access to any
85    /// delta, while also making removals from the end efficient. As [`Self::truncate`] only ever
86    /// removes some oldest `n` entries, this is the exact behavior we want.
87    deltas: VecDeque<Delta>,
88}
89
90impl History {
91    /// Constructs a new history container, containing at most `max_count` historical versions for
92    /// a tree.
93    #[must_use]
94    pub fn empty(max_count: usize) -> Self {
95        let deltas = VecDeque::new();
96        Self { max_count, deltas }
97    }
98
99    /// Gets the maximum number of versions that this history can store.
100    #[must_use]
101    pub fn max_versions(&self) -> usize {
102        self.max_count
103    }
104
105    /// Gets the current number of versions in the history.
106    #[must_use]
107    pub fn num_versions(&self) -> usize {
108        self.deltas.len()
109    }
110
111    /// Returns all the roots that the history knows about.
112    ///
113    /// The iteration order of the roots is guaranteed to move backward in time, with earlier items
114    /// being roots from versions closer to the present.
115    ///
116    /// # Complexity
117    ///
118    /// Calling this method provides an iterator whose consumption requires a traversal of all the
119    /// versions. The method's complexity is thus `O(n)` in the number of versions.
120    pub fn roots(&self) -> impl Iterator<Item = RootValue> {
121        self.deltas.iter().rev().map(|d| d.root)
122    }
123
124    /// Returns the root value that corresponds to the provided `version`.
125    pub fn root_for_version(&self, version: VersionId) -> Result<RootValue> {
126        let ix = self.find_latest_corresponding_version(version)?;
127
128        // The direct index is safe here because `find_latest_...` will have returned an error if
129        // there is no such version, and is hence guaranteed to have returned a valid index.
130        Ok(self.deltas[ix].root)
131    }
132
133    /// Adds a version to the history with the provided `root` and represented by the changes from
134    /// the current tree given in `nodes` and `leaves`.
135    ///
136    /// If adding this version would result in exceeding `self.max_count` historical versions, then
137    /// the oldest of the versions is automatically removed.
138    ///
139    /// # Gotchas
140    ///
141    /// When constructing the `nodes` and `leaves`, keep in mind that those collections must contain
142    /// entries for the **default value of a node or leaf** at any position where the tree was
143    /// sparse in the state represented by `root`. If this is not done, incorrect values may be
144    /// returned.
145    ///
146    /// This is necessary because the changes are the _reverse_ from what one might expect. Namely,
147    /// the changes in a given version `v` must "_revert_" the changes made in the transition from
148    /// version `v` to version `v + 1`.
149    ///
150    /// # Errors
151    ///
152    /// - [`HistoryError::NonMonotonicVersions`] if the provided version is not greater than the
153    ///   previously added version.
154    pub fn add_version(
155        &mut self,
156        root: RootValue,
157        version_id: VersionId,
158        nodes: NodeChanges,
159        changed_keys: ChangedKeys,
160    ) -> Result<()> {
161        // We need to fail early if the provided new version is not monotonic with respect to the
162        // latest version in the history.
163        if let Some(v) = self.deltas.iter().last()
164            && v.version_id >= version_id
165        {
166            return Err(HistoryError::NonMonotonicVersions(version_id, v.version_id));
167        }
168
169        // We then check if we would exceed our version count limit, and remove the oldest if so.
170        if self.num_versions() >= self.max_versions() {
171            self.deltas.pop_front();
172        }
173
174        // We then need to update all the older deltas with the necessary additional changes
175        // represented in this newly-added version.
176        for delta in &mut self.deltas {
177            // The root and the version remain the same, but we may need to change the nodes and
178            // keys.
179            for (ix, value) in &nodes {
180                delta.nodes.entry(*ix).or_insert(*value);
181            }
182            for (key, value) in &changed_keys {
183                // If the delta has removed something, we never want to re-add it over the top.
184                delta.changed_keys.entry(*key).or_insert(*value);
185            }
186        }
187
188        self.deltas.push_back(Delta::new(root, version_id, nodes, changed_keys));
189
190        Ok(())
191    }
192
193    /// Adds a version to the history and represented by the changes from the current tree given
194    /// `mutations`.
195    ///
196    /// If adding this version would result in exceeding `self.max_count` historical versions, then
197    /// the oldest of the versions is automatically removed.
198    ///
199    /// # Gotchas
200    ///
201    /// When constructing the `mutations`, keep in mind that the set must contain entries for the
202    /// **default value of a node or leaf** at any position where the tree was sparse in the state
203    /// represented by `root`. If this is not done, incorrect values may be returned.
204    ///
205    /// This is necessary because the changes are the _reverse_ from what one might expect. Namely,
206    /// the changes in a given version `v` must "_revert_" the changes made in the transition from
207    /// version `v` to version `v + 1`.
208    ///
209    /// # Errors
210    ///
211    /// - [`HistoryError::NonMonotonicVersions`] if the provided version is not greater than the
212    ///   previously added version.
213    pub fn add_version_from_mutation_set(
214        &mut self,
215        version_id: VersionId,
216        mutations: MutationSet,
217    ) -> Result<()> {
218        let mut changed_keys = ChangedKeys::default();
219        mutations.new_pairs.into_iter().for_each(|(k, v)| {
220            changed_keys.insert(k, v);
221        });
222
223        // The node changes are more complex, as we have to explicitly handle reversions to empty
224        // specially.
225        let node_changes: NodeChanges = mutations
226            .node_mutations
227            .into_iter()
228            .map(|(ix, m)| match m {
229                NodeMutation::Removal => (ix, *EmptySubtreeRoots::entry(SMT_DEPTH, ix.depth())),
230                NodeMutation::Addition(n) => (ix, n.hash()),
231            })
232            .collect();
233
234        // Now we can simply delegate to the standard function.
235        self.add_version(mutations.new_root, version_id, node_changes, changed_keys)
236    }
237
238    /// Returns the index in the sequence of deltas of the version that corresponds to the provided
239    /// `version_id`.
240    ///
241    /// To "correspond" means that it either has the provided `version_id`, or is the newest version
242    /// with a `version_id` less than the provided id. In either case, it is the correct version to
243    /// be used to query the tree state in the provided `version_id`.
244    ///
245    /// # Complexity
246    ///
247    /// Finding the latest corresponding version in the history requires a linear traversal of the
248    /// history entries, and hence has complexity `O(n)` in the number of versions.
249    ///
250    /// # Errors
251    ///
252    /// - [`HistoryError::HistoryEmpty`] if the history is empty and hence there is no version to
253    ///   find.
254    /// - [`HistoryError::VersionTooOld`] if the history does not contain the data to provide a
255    ///   coherent overlay for the provided `version_id` due to `version_id` being older than the
256    ///   oldest version stored.
257    fn find_latest_corresponding_version(&self, version_id: VersionId) -> Result<usize> {
258        // If the version is older than the oldest, we error.
259        if let Some(oldest_version) = self.deltas.front() {
260            if oldest_version.version_id > version_id {
261                return Err(HistoryError::VersionTooOld);
262            }
263        } else {
264            return Err(HistoryError::VersionTooOld);
265        }
266
267        // As we want the NEWEST delta that satisfies the version, we look for the position at which
268        // the delta cannot be used and move back by one.
269        let ix = self
270            .deltas
271            .iter()
272            .position(|d| d.version_id > version_id)
273            .unwrap_or_else(|| self.num_versions())
274            .checked_sub(1)
275            .expect(
276                "Subtraction should not overflow as we have ruled out the no-version \
277                case, and in the other cases the left operand will be >= 1",
278            );
279
280        Ok(ix)
281    }
282
283    /// Returns a view of the history that allows querying as a single unified overlay on the
284    /// current state of the merkle tree as if the overlay was reverting the tree to the state
285    /// corresponding to the specified `version_id`.
286    ///
287    /// Note that the history may not contain a version that directly corresponds to `version_id`.
288    /// In such a case, the view will instead use the newest version coherent with the provided
289    /// `version_id`, as this is the correct version for the provided id. Note that this will be
290    /// incorrect if the versions stored in the history do not represent contiguous changes from the
291    /// current tree.
292    ///
293    /// # Complexity
294    ///
295    /// The computational complexity of this method is linear in the number of versions stored in
296    /// the history.
297    ///
298    /// # Errors
299    ///
300    /// - [`HistoryError::VersionTooOld`] if the history does not contain the data to provide a
301    ///   coherent overlay for the provided `version_id` due to `version_id` being older than the
302    ///   oldest version stored.
303    pub fn get_view_at(&self, version_id: VersionId) -> Result<HistoryView<'_>> {
304        HistoryView::new_of(version_id, self)
305    }
306
307    /// Removes all versions in the history that are older than the version denoted by the provided
308    /// `version_id`.
309    ///
310    /// If `version_id` is not a version known by the history, it will keep the newest version that
311    /// is capable of serving as that version in queries.
312    ///
313    /// # Complexity
314    ///
315    /// The computational complexity of this method is linear in the number of versions stored in
316    /// the history prior to any removals.
317    pub fn truncate(&mut self, version_id: VersionId) -> usize {
318        // We start by getting the index to truncate to, though it is not an error to remove
319        // something too old.
320        let truncate_ix = self.find_latest_corresponding_version(version_id).unwrap_or(0);
321
322        for _ in 0..truncate_ix {
323            self.deltas.pop_front();
324        }
325
326        truncate_ix
327    }
328
329    /// Removes all versions from the history.
330    pub fn clear(&mut self) {
331        self.deltas.clear();
332    }
333}

349/// A read-only view of the history overlay on the tree at a specified place in the history.
350#[derive(Copy, Clone, Debug)]
351pub struct HistoryView<'history> {
352    /// The delta corresponding to this overlay.
353    delta: &'history Delta,
354}
355
356impl<'history> HistoryView<'history> {
357    /// Constructs a new history view that acts as a single overlay of the state represented by the
358    /// history at the provided `version`.
359    ///
360    /// # Complexity
361    ///
362    /// The computational complexity of this method is linear in the number of versions stored in
363    /// the history.
364    ///
365    /// # Errors
366    ///
367    /// - [`HistoryError::VersionTooOld`] if the history does not contain the data to provide a
368    ///   coherent overlay for the provided `version`.
369    fn new_of(version: VersionId, history: &'history History) -> Result<Self> {
370        let version_ix = history.find_latest_corresponding_version(version)?;
371        let delta = &history.deltas[version_ix];
372        Ok(Self { delta })
373    }
374
375    /// Gets the value of the node in the history at the provided `index`, or returns `None` if the
376    /// version does not overlay the current tree at that node.
377    #[must_use]
378    pub fn node_value(&self, index: &NodeIndex) -> Option<&Word> {
379        self.delta.nodes.get(index)
380    }
381
382    /// Queries the value of a specific `key` in a leaf in the overlay, returning the value for that
383    /// `key` if it has been changed, and [`None`] otherwise.
384    #[must_use]
385    pub fn value(&self, key: &Word) -> Option<Word> {
386        self.delta.changed_keys.get(key).cloned()
387    }
388
389    /// Returns `true` if the key is removed by this delta, and `false` otherwise.
390    #[must_use]
391    pub fn is_key_removed(&self, key: &Word) -> bool {
392        self.delta.changed_keys.get(key).map(Word::is_empty).unwrap_or(false)
393    }
394
395    /// Returns an iterator which yields the entries that are added by this view.
396    pub fn changed_keys(&self) -> impl Iterator<Item = (Word, Word)> + 'history {
397        self.delta.changed_keys.iter().map(|(k, v)| (*k, *v))
398    }
399}

401// DELTA

404/// A delta for a state `n` represents the changes (to both nodes and leaves) that need to be
405/// applied on top of the state `n + 1` to yield the correct tree for state `n`.
406///
407/// # Cumulative Deltas and Temporal Ordering
408///
409/// In order to best represent the history of a merkle tree, these deltas are constructed to take
410/// advantage of two main properties:
411///
412/// - They are _cumulative_, which reduces their practical memory usage. This does, however, mean
413///   that querying the state of older blocks is more expensive than querying newer ones.
414/// - Deltas are applied in **temporally reversed order** from what one might expect. Most
415///   conventional applications of deltas bring something from the past into the future through
416///   application. In our case, the application of one or more deltas moves the tree into a **past
417///   state**.
418///
419/// # Construction
420///
421/// While the [`Delta`] type is visible in the interface of the history, it is only intended to be
422/// constructed by the history. Users should not be allowed to construct it directly.
423#[derive(Clone, Debug, PartialEq)]
424struct Delta {
425    /// The root of the tree in the `version` corresponding to the application of the reversions in
426    /// this delta to the previous tree state.
427    root: RootValue,
428
429    /// The version of the tree represented by the delta.
430    version_id: VersionId,
431
432    /// Any changes to the non-leaf nodes in the tree for this delta.
433    nodes: NodeChanges,
434
435    /// The keys that need to be changed by the delta to return the target tree to the state
436    /// represented by the delta.
437    changed_keys: ChangedKeys,
438}
439
440impl Delta {
441    /// Creates a new delta with the provided `root`, representing the provided changes to the
442    /// `nodes` in the merkle tree, and using `added_keys` and `removed_keys` to represent the
443    /// changes to entries in the tree.
444    #[must_use]
445    fn new(
446        root: RootValue,
447        version_id: VersionId,
448        nodes: NodeChanges,
449        changed_keys: ChangedKeys,
450    ) -> Self {
451        Self { root, version_id, nodes, changed_keys }
452    }
