1//! This module contains the definition of [`History`], a simple container for some number of
2//! historical versions of a given merkle tree.
3//!
4//! This history consists of a series of _deltas_ from the current state of the tree, moving
5//! backward in history away from that current state. These deltas are then used to form a "merged
6//! overlay" that represents the changes to be made on top of the current tree to put it _back_ in
7//! that historical state.
8//!
9//! It provides functionality for adding new states to the history, as well as for querying the
10//! history at a given point in time.
11//!
12//! # Complexity
13//!
14//! Versions in this structure are _cumulative_. To get the entire picture of an arbitrary node or
15//! leaf at version `v` it may be necessary to check for changes in all versions between `v` and the
16//! current tree state. This gives worst-case complexity `O(v)` when querying a node or leaf for the
17//! version `v`.
18//!
19//! This is acceptable overhead as we assert that newer versions are far more likely to be queried
20//! than older versions. Nevertheless, it may be improved in future using a sharing approach, but
21//! that potential improvement is being ignored for now for the sake of simplicity.
22//!
23//! # Performance
24//!
25//! This structure operates entirely in memory, and is hence reasonably quick to query. As of the
26//! current time, no detailed benchmarking has taken place for the history, but based on some basic
27//! profiling the major time taken is in chasing pointers throughout memory due to the use of
28//! [`Map`]s, but this is unavoidable in the current structure and may need to be revisited in
29//! the future.
30

57/// A compact leaf is a mapping from full word-length keys to word-length values, intended to be
58/// stored in the leaves of an otherwise shallower merkle tree.
59///
60/// We use a BTreeMap as we need a guaranteed iteration order over the keys.
61pub type CompactLeaf = BTreeMap<Word, Word>;
62
63/// A collection of changes to arbitrary non-leaf nodes in a merkle tree.
64///
65/// All changes to nodes between versions `v` and `v + 1` must be explicitly "undone" in the
66/// `NodeChanges` representing version `v`. This includes nodes that were defaulted in version `v`
67/// that were given an explicit value in version `v + 1`, where the `NodeChanges` must explicitly
68/// set those nodes back to the default.
69///
70/// Failure to do so will result in incorrect values when those nodes are queried at a point in the
71/// history corresponding to version `v`.
72pub type NodeChanges = Map<NodeIndex, Word>;
73
74/// A collection of changes to arbitrary leaf nodes in a merkle tree.
75///
76/// While represented as a single leaf, it only contains the changes to the leaf as part of the
77/// delta, and still needs to be combined with the actual leaf data for querying.
78///
79/// Note that if in the version of the tree represented by these `LeafChanges` had the default value
80/// at the leaf, this default value must be made concrete in the map. Failure to do so will retain a
81/// newer, non-default value for that leaf, and thus result in incorrect query results at this point
82/// in the history.
83pub type LeafChanges = Map<LeafIndex<SMT_DEPTH>, CompactLeaf>;

88/// A History contains a sequence of versions atop a given tree.
89///
90/// The versions are _cumulative_, meaning that querying the history must account for changes from
91/// the current tree that take place in versions that are not the queried version or the current
92/// tree.
93#[derive(Clone, Debug)]
94pub struct History {
95    /// The maximum number of historical versions to be stored.
96    max_count: usize,
97
98    /// The deltas that make up the history for this tree.
99    ///
100    /// It will never contain more than `max_count` deltas, and is ordered with the oldest data at
101    /// the lowest index.
102    ///
103    /// # Implementation Note
104    ///
105    /// As we are targeting small numbers of history items (e.g. 30), having a sequence with an
106    /// allocated capacity equal to the small maximum number of items is perfectly sane. This will
107    /// avoid costly reallocations in the fast path.
108    ///
109    /// We use a [`VecDeque`] instead of a [`Vec`] or [`alloc::collections::LinkedList`] as we
110    /// estimate that the vast majority of removals will be the oldest entries as new ones are
111    /// pushed. This means that we can optimize for those removals along with indexing performance,
112    /// rather than optimizing for more rare removals from the middle of the sequence.
113    deltas: VecDeque<Delta>,
114}
115
116impl History {
117    /// Constructs a new history container, containing at most `max_count` historical versions for
118    /// a tree.
119    #[must_use]
120    pub fn empty(max_count: usize) -> Self {
121        // We allocate one more than we actually need to store to allow us to insert and THEN
122        // remove, rather than the other way around. This leads to negligible increases in memory
123        // usage while allowing for cleaner code.
124        let deltas = VecDeque::with_capacity(max_count + 1);
125        Self { max_count, deltas }
126    }
127
128    /// Gets the maximum number of versions that this history can store.
129    #[must_use]
130    pub fn max_versions(&self) -> usize {
131        self.max_count
132    }
133
134    /// Gets the current number of versions in the history.
135    #[must_use]
136    pub fn num_versions(&self) -> usize {
137        self.deltas.len()
138    }
139
140    /// Returns all the roots that the history knows about.
141    ///
142    /// The iteration order of the roots is guaranteed to move backward in time, with earlier items
143    /// being roots from versions closer to the present.
144    ///
145    /// # Complexity
146    ///
147    /// Calling this method provides an iterator whose consumption requires a traversal of all the
148    /// versions. The method's complexity is thus `O(n)` in the number of versions.
149    pub fn roots(&self) -> impl Iterator<Item = RootValue> {
150        self.deltas.iter().rev().map(|d| d.root)
151    }
152
153    /// Returns the root value that corresponds to the provided `version`.
154    pub fn root_for_version(&self, version: VersionId) -> Result<RootValue> {
155        let ix = self.find_latest_corresponding_version(version)?;
156
157        // The direct index is safe here because `find_latest_...` will have returned an error if
158        // there is no such version, and is hence guaranteed to have returned a valid index.
159        Ok(self.deltas[ix].root)
160    }
161
162    /// Adds a version to the history with the provided `root` and represented by the changes from
163    /// the current tree given in `nodes` and `leaves`.
164    ///
165    /// If adding this version would result in exceeding `self.max_count` historical versions, then
166    /// the oldest of the versions is automatically removed.
167    ///
168    /// # Gotchas
169    ///
170    /// When constructing the `nodes` and `leaves`, keep in mind that those collections must contain
171    /// entries for the **default value of a node or leaf** at any position where the tree was
172    /// sparse in the state represented by `root`. If this is not done, incorrect values may be
173    /// returned.
174    ///
175    /// This is necessary because the changes are the _reverse_ from what one might expect. Namely,
176    /// the changes in a given version `v` must "_revert_" the changes made in the transition from
177    /// version `v` to version `v + 1`.
178    ///
179    /// # Errors
180    ///
181    /// - [`HistoryError::NonMonotonicVersions`] if the provided version is not greater than the
182    ///   previously added version.
183    pub fn add_version(
184        &mut self,
185        root: RootValue,
186        version_id: VersionId,
187        nodes: NodeChanges,
188        leaves: LeafChanges,
189    ) -> Result<()> {
190        if let Some(v) = self.deltas.iter().last() {
191            if v.version_id < version_id {
192                self.deltas.push_back(Delta::new(root, version_id, nodes, leaves));
193                if self.num_versions() > self.max_versions() {
194                    self.deltas.pop_front();
195                }
196
197                Ok(())
198            } else {
199                Err(HistoryError::NonMonotonicVersions(version_id, v.version_id))
200            }
201        } else {
202            self.deltas.push_back(Delta::new(root, version_id, nodes, leaves));
203
204            Ok(())
205        }
206    }
207
208    /// Adds a version to the history and represented by the changes from the current tree given
209    /// `mutations`.
210    ///
211    /// If adding this version would result in exceeding `self.max_count` historical versions, then
212    /// the oldest of the versions is automatically removed.
213    ///
214    /// # Gotchas
215    ///
216    /// When constructing the `mutations`, keep in mind that the set must contain entries for the
217    /// **default value of a node or leaf** at any position where the tree was sparse in the state
218    /// represented by `root`. If this is not done, incorrect values may be returned.
219    ///
220    /// This is necessary because the changes are the _reverse_ from what one might expect. Namely,
221    /// the changes in a given version `v` must "_revert_" the changes made in the transition from
222    /// version `v` to version `v + 1`.
223    ///
224    /// # Errors
225    ///
226    /// - [`HistoryError::NonMonotonicVersions`] if the provided version is not greater than the
227    ///   previously added version.
228    pub fn add_version_from_mutation_set(
229        &mut self,
230        version_id: VersionId,
231        mutations: MutationSet,
232    ) -> Result<()> {
233        // The leaf changes must be grouped by parent leaf when being inserted, so we do that here.
234        let mut leaf_changes = LeafChanges::default();
235        for (key, val) in mutations.new_pairs {
236            leaf_changes.entry(LeafIndex::from(key)).or_default().insert(key, val);
237        }
238
239        // The node changes are more complex, as we have to explicitly handle reversions to empty
240        // specially.
241        let node_changes: NodeChanges = mutations
242            .node_mutations
243            .into_iter()
244            .map(|(ix, m)| match m {
245                NodeMutation::Removal => (ix, *EmptySubtreeRoots::entry(SMT_DEPTH, ix.depth())),
246                NodeMutation::Addition(n) => (ix, n.hash()),
247            })
248            .collect();
249
250        // Now we can simply delegate to the standard function.
251        self.add_version(mutations.new_root, version_id, node_changes, leaf_changes)
252    }
253
254    /// Returns the index in the sequence of deltas of the version that corresponds to the provided
255    /// `version_id`.
256    ///
257    /// To "correspond" means that it either has the provided `version_id`, or is the newest version
258    /// with a `version_id` less than the provided id. In either case, it is the correct version to
259    /// be used to query the tree state in the provided `version_id`.
260    ///
261    /// # Complexity
262    ///
263    /// Finding the latest corresponding version in the history requires a linear traversal of the
264    /// history entries, and hence has complexity `O(n)` in the number of versions.
265    ///
266    /// # Errors
267    ///
268    /// - [`HistoryError::HistoryEmpty`] if the history is empty and hence there is no version to
269    ///   find.
270    /// - [`HistoryError::VersionTooOld`] if the history does not contain the data to provide a
271    ///   coherent overlay for the provided `version_id` due to `version_id` being older than the
272    ///   oldest version stored.
273    fn find_latest_corresponding_version(&self, version_id: VersionId) -> Result<usize> {
274        // If the version is older than the oldest, we error.
275        if let Some(oldest_version) = self.deltas.front() {
276            if oldest_version.version_id > version_id {
277                return Err(HistoryError::VersionTooOld);
278            }
279        } else {
280            return Err(HistoryError::VersionTooOld);
281        }
282
283        let ix = self
284            .deltas
285            .iter()
286            .position(|d| d.version_id > version_id)
287            .unwrap_or_else(|| self.num_versions())
288            .checked_sub(1)
289            .expect(
290                "Subtraction should not overflow as we have ruled out the no-version \
291                case, and in the other cases the left operand will be >= 1",
292            );
293
294        Ok(ix)
295    }
296
297    /// Returns a view of the history that allows querying as a single unified overlay on the
298    /// current state of the merkle tree as if the overlay was reverting the tree to the state
299    /// corresponding to the specified `version_id`.
300    ///
301    /// Note that the history may not contain a version that directly corresponds to `version_id`.
302    /// In such a case, the view will instead use the newest version coherent with the provided
303    /// `version_id`, as this is the correct version for the provided id. Note that this will be
304    /// incorrect if the versions stored in the history do not represent contiguous changes from the
305    /// current tree.
306    ///
307    /// # Complexity
308    ///
309    /// The computational complexity of this method is linear in the number of versions stored in
310    /// the history.
311    ///
312    /// # Errors
313    ///
314    /// - [`HistoryError::VersionTooOld`] if the history does not contain the data to provide a
315    ///   coherent overlay for the provided `version_id` due to `version_id` being older than the
316    ///   oldest version stored.
317    pub fn get_view_at(&self, version_id: VersionId) -> Result<HistoryView<'_>> {
318        HistoryView::new_of(version_id, self)
319    }
320
321    /// Removes all versions in the history that are older than the version denoted by the provided
322    /// `version_id`.
323    ///
324    /// If `version_id` is not a version known by the history, it will keep the newest version that
325    /// is capable of serving as that version in queries.
326    ///
327    /// # Complexity
328    ///
329    /// The computational complexity of this method is linear in the number of versions stored in
330    /// the history prior to any removals.
331    pub fn truncate(&mut self, version_id: VersionId) -> usize {
332        // We start by getting the index to truncate to, though it is not an error to remove
333        // something too old.
334        let truncate_ix = self.find_latest_corresponding_version(version_id).unwrap_or(0);
335
336        for _ in 0..truncate_ix {
337            self.deltas.pop_front();
338        }
339
340        truncate_ix
341    }
342
343    /// Removes all versions from the history.
344    pub fn clear(&mut self) {
345        self.deltas.clear();
346    }
347}

363/// A read-only view of the history overlay on the tree at a specified place in the history.
364#[derive(Debug)]
365pub struct HistoryView<'history> {
366    /// The version of the history pointed to by the history view.
367    version: VersionId,
368
369    /// The index of the target version in the history.
370    version_ix: usize,
371
372    /// The history that actually stores the data that will be queried.
373    history: &'history History,
374}
375
376impl<'history> HistoryView<'history> {
377    /// Constructs a new history view that acts as a single overlay of the state represented by the
378    /// history at the provided `version`.
379    ///
380    /// # Complexity
381    ///
382    /// The computational complexity of this method is linear in the number of versions stored in
383    /// the history.
384    ///
385    /// # Errors
386    ///
387    /// - [`HistoryError::VersionTooOld`] if the history does not contain the data to provide a
388    ///   coherent overlay for the provided `version`.
389    fn new_of(version: VersionId, history: &'history History) -> Result<Self> {
390        let version_ix = history.find_latest_corresponding_version(version)?;
391        Ok(Self { version, version_ix, history })
392    }
393
394    /// Gets the value of the node in the history at the provided `index`, or returns `None` if the
395    /// version does not overlay the current tree at that node.
396    ///
397    /// # Complexity
398    ///
399    /// The computational complexity of this method is linear in the number of versions due to the
400    /// need to traverse to find the correct overlay value.
401    #[must_use]
402    pub fn node_value(&self, index: &NodeIndex) -> Option<&Word> {
403        self.history
404            .deltas
405            .iter()
406            .skip(self.version_ix)
407            .find_map(|v| v.nodes.get(index))
408    }
409
410    /// Gets a single leaf that represents the delta from the current version of the tree to the
411    /// point in the history at the specified `index`.
412    ///
413    /// If the specified version does not overlay the current tree at that leaf, it will return an
414    /// empty compact leaf.
415    ///
416    /// # Complexity
417    ///
418    /// The computational complexity of this method is linear in the number of versions due to the
419    /// need to traverse to find the correct overlay value.
420    #[must_use]
421    pub fn leaf_delta(&self, index: &LeafIndex<SMT_DEPTH>) -> CompactLeaf {
422        let mut leaf = CompactLeaf::default();
423
424        // We want to keep the _oldest_ change for any particular key in a leaf.
425        for delta in self.history.deltas.iter().skip(self.version_ix) {
426            if let Some(leaf_delta) = delta.leaves.get(index) {
427                for (key, value) in leaf_delta {
428                    leaf.entry(*key).or_insert(*value);
429                }
430            }
431        }
432
433        leaf
434    }
435
436    /// Queries the value of a specific `key` in a leaf in the overlay, returning the value for that
437    /// `key` if it has been changed, and [`None`] otherwise.
438    ///
439    /// # Complexity
440    ///
441    /// The computational complexity of this method is linear in the number of versions due to the
442    /// need to traverse to find the correct overlay value.
443    #[must_use]
444    pub fn value(&self, key: &Word) -> Option<Word> {
445        self.leaf_delta(&LeafIndex::from(*key)).get(key).copied()
446    }
447
448    /// Returns an iterator which yields the entries that are changed by this view.
449    ///
450    /// This iterator yields entries in an order such that they are sorted by their leaf index, and
451    /// entries that share a leaf index are sorted by key. It includes key-value pairs where the
452    /// value is the empty word, as these are necessary for merging with entries in the full tree.
453    pub fn entries(&self) -> impl Iterator<Item = TreeEntry> + 'history {
454        // It is safe to call this directly here as the construction of `HistoryView` has ensured
455        // that we have such a version.
456        HistoricalEntriesIterator::new(self.history, self.version)
457    }
458}

460// DELTA
461// ================================================================================================
462
463/// A delta for a state `n` represents the changes (to both nodes and leaves) that need to be
464/// applied on top of the state `n + 1` to yield the correct tree for state `n`.
465///
466/// # Cumulative Deltas and Temporal Ordering
467///
468/// In order to best represent the history of a merkle tree, these deltas are constructed to take
469/// advantage of two main properties:
470///
471/// - They are _cumulative_, which reduces their practical memory usage. This does, however, mean
472///   that querying the state of older blocks is more expensive than querying newer ones.
473/// - Deltas are applied in **temporally reversed order** from what one might expect. Most
474///   conventional applications of deltas bring something from the past into the future through
475///   application. In our case, the application of one or more deltas moves the tree into a **past
476///   state**.
477///
478/// # Construction
479///
480/// While the [`Delta`] type is visible in the interface of the history, it is only intended to be
481/// constructed by the history. Users should not be allowed to construct it directly.
482#[derive(Clone, Debug, PartialEq)]
483struct Delta {
484    /// The root of the tree in the `version` corresponding to the application of the reversions in
485    /// this delta to the previous tree state.
486    root: RootValue,
487
488    /// The version of the tree represented by the delta.
489    version_id: VersionId,
490
491    /// Any changes to the non-leaf nodes in the tree for this delta.
492    nodes: NodeChanges,
493
494    /// Any changes to the leaf nodes in the tree for this delta.
495    ///
496    /// Note that the leaf state is **not represented compactly**, and describes the entire state
497    /// of the leaf in the corresponding version.
498    leaves: LeafChanges,
499}
500
501impl Delta {
502    /// Creates a new delta with the provided `root`, and representing the provided
503    /// changes to `nodes` and `leaves` in the merkle tree.
504    #[must_use]
505    fn new(
506        root: RootValue,
507        version_id: VersionId,
508        nodes: NodeChanges,
509        leaves: LeafChanges,
510    ) -> Self {
511        Self { root, version_id, nodes, leaves }
512    }
513}

515// ENTRIES ITERATOR
516// ================================================================================================
517
518/// An iterator over the historical value for each changed entry at a given point in the history.
519///
520/// This iterator yields entries in an order such that they are sorted by their leaf index, and
521/// entries that share a leaf index are sorted by key. It includes key-value pairs where the value
522/// is the empty word, as these are necessary for merging with entries in the full tree.
523#[derive(Debug)]
524pub struct HistoricalEntriesIterator<'history> {
525    /// The history over which the iterator is defined.
526    history: &'history History,
527
528    /// The version in the history to be working from.
529    version: VersionId,
530
531    /// The set of all changed leaves in the deltas that make up this iterator that have not yet
532    /// been visited by the iterator.
533    ///
534    /// We use a BTreeSet specifically as we need sorted iteration behavior.
535    changed_leaves: BTreeSet<LeafIndex<SMT_DEPTH>>,
536
537    /// The current state of the iterator's iteration behavior.
538    position: HistoricalEntriesIteratorState,
539}
540
541impl<'history> HistoricalEntriesIterator<'history> {
542    /// Creates a new historical entries iterator that represents a coherent set of delta entries at
543    /// the position in the history given by `version_ix`.
544    fn new(history: &'history History, version: VersionId) -> Self {
545        let changed_leaves = history
546            .deltas
547            .iter()
548            .skip(
549                history
550                    .find_latest_corresponding_version(version)
551                    .expect("Caller has guaranteed existence of a corresponding version"),
552            )
553            .flat_map(|d| d.leaves.keys())
554            .copied()
555            .collect();
556
557        // We want to start not pointing to any leaf as we can only advance when `next` is called.
558        let current_leaf_index = HistoricalEntriesIteratorState::NotInLeaf;
559
560        Self {
561            history,
562            version,
563            changed_leaves,
564            position: current_leaf_index,
565        }
566    }
567}
568
569impl<'history> Iterator for HistoricalEntriesIterator<'history> {
570    type Item = TreeEntry;
571
572    fn next(&mut self) -> Option<Self::Item> {
573        match &mut self.position {
574            HistoricalEntriesIteratorState::NotInLeaf => {
575                // If we are not inside a leaf we need to see if we can become so.
576                if let Some(ix) = self.changed_leaves.pop_first() {
577                    // If we can move into a new leaf, we transition the state into that leaf and
578                    // return the entry.
579                    let leaf_delta = self
580                        .history
581                        .get_view_at(self.version)
582                        .expect(
583                            "Version was guaranteed to exist before construction of the iterator",
584                        )
585                        .leaf_delta(&ix);
586
587                    // As we are querying based on `changed_leaves`, each of the `leaf_delta`
588                    // results should contain at least one item.
589                    let (key, value) = leaf_delta
590                        .first_key_value()
591                        .expect("At least one item guaranteed by construction");
592                    let item = TreeEntry { key: *key, value: *value };
593
594                    // At this point we now have the item, but we need to set up the state to point
595                    // to this item as we return it.
596                    self.position = HistoricalEntriesIteratorState::InLeaf { value: leaf_delta };
597
598                    Some(item)
599                } else {
600                    // If we cannot move to a new leaf index, the iterator is done.
601                    None
602                }
603            },
604            HistoricalEntriesIteratorState::InLeaf { value } => {
605                // If we are already inside a leaf, there are two cases that can occur when
606                // advancing.
607                value.pop_first().expect("InLeaf implies there is at least one entry in value");
608                if let Some((k, v)) = value.first_key_value() {
609                    // The first (and simplest) case is that we have another entry in the current
610                    // leaf value. In this case, the item is just the front of the leaf value, and
611                    // we re-write the key to point to it while leaving the leaf index the same.
612                    let item = TreeEntry { key: *k, value: *v };
613
614                    Some(item)
615                } else {
616                    // Here, we have no further entries in the current leaf, so we have to check if
617                    // there is another leaf to move to. In other words, we are implicitly in the
618                    // `NotInLeaf` state, so we can just call `next` recursively.
619                    //
620                    // This is not a stack overflow risk as it should only ever recurse once.
621                    self.position = HistoricalEntriesIteratorState::NotInLeaf;
622                    self.next()
623                }
624            },
625        }
626    }
627}

629/// The state that tracks where the iterator is in the iteration process.
630#[derive(Debug)]
631enum HistoricalEntriesIteratorState {
632    /// It currently does not point to any underlying leaf index.
633    NotInLeaf,
634
635    /// It is currently pointing to the specified key within the specified index.
636    InLeaf {
637        /// The combined full delta that represents the compact leaf.
638        value: CompactLeaf,
639    },
640}
