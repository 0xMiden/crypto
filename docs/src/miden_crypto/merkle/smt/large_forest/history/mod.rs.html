1//! This module contains the definition of [`History`], a simple container for some number of
2//! historical versions of a given merkle tree.
3//!
4//! This history consists of a series of _deltas_ from the current state of the tree, moving
5//! backward in history away from that current state. These deltas are then used to form a "merged
6//! overlay" that represents the changes to be made on top of the current tree to put it _back_ in
7//! that historical state.
8//!
9//! It provides functionality for adding new states to the history, as well as for querying the
10//! history at a given point in time.
11//!
12//! # Complexity
13//!
14//! Versions in this structure are _cumulative_. To get the entire picture of an arbitrary node or
15//! leaf at version `v` it may be necessary to check for changes in all versions between `v` and the
16//! current tree state. This gives worst-case complexity `O(v)` when querying a node or leaf for the
17//! version `v`.
18//!
19//! This is acceptable overhead as we assert that newer versions are far more likely to be queried
20//! than older versions. Nevertheless, it may be improved in future using a sharing approach, but
21//! that potential improvement is being ignored for now for the sake of simplicity.
22//!
23//! # Performance
24//!
25//! This structure operates entirely in memory, and is hence reasonably quick to query. As of the
26//! current time, no detailed benchmarking has taken place for the history, but based on some basic
27//! profiling the major time taken is in chasing pointers throughout memory due to the use of
28//! [`Map`]s, but this is unavoidable in the current structure and may need to be revisited in
29//! the future.

55/// A compact leaf is a mapping from full word-length keys to word-length values, intended to be
56/// stored in the leaves of an otherwise shallower merkle tree.
57pub type CompactLeaf = Map<Word, Word>;
58
59/// A collection of changes to arbitrary non-leaf nodes in a merkle tree.
60///
61/// All changes to nodes between versions `v` and `v + 1` must be explicitly "undone" in the
62/// `NodeChanges` representing version `v`. This includes nodes that were defaulted in version `v`
63/// that were given an explicit value in version `v + 1`, where the `NodeChanges` must explicitly
64/// set those nodes back to the default.
65///
66/// Failure to do so will result in incorrect values when those nodes are queried at a point in the
67/// history corresponding to version `v`.
68pub type NodeChanges = Map<NodeIndex, Word>;
69
70/// A collection of changes to arbitrary leaf nodes in a merkle tree.
71///
72/// This represents the state of the leaf wholesale, rather than as a delta from the newer version.
73/// This massively simplifies querying leaves in the history.
74///
75/// Note that if in the version of the tree represented by these `LeafChanges` had the default value
76/// at the leaf, this default value must be made concrete in the map. Failure to do so will retain a
77/// newer, non-default value for that leaf, and thus result in incorrect query results at this point
78/// in the history.
79pub type LeafChanges = Map<LeafIndex<SMT_DEPTH>, CompactLeaf>;

84/// A History contains a sequence of versions atop a given tree.
85///
86/// The versions are _cumulative_, meaning that querying the history must account for changes from
87/// the current tree that take place in versions that are not the queried version or the current
88/// tree.
89#[allow(dead_code)] // Temporary
90#[derive(Clone, Debug)]
91pub struct History {
92    /// The maximum number of historical versions to be stored.
93    max_count: usize,
94
95    /// The deltas that make up the history for this tree.
96    ///
97    /// It will never contain more than `max_count` deltas, and is ordered with the oldest data at
98    /// the lowest index.
99    ///
100    /// # Implementation Note
101    ///
102    /// As we are targeting small numbers of history items (e.g. 30), having a sequence with an
103    /// allocated capacity equal to the small maximum number of items is perfectly sane. This will
104    /// avoid costly reallocations in the fast path.
105    ///
106    /// We use a [`VecDeque`] instead of a [`Vec`] or [`alloc::collections::LinkedList`] as we
107    /// estimate that the vast majority of removals will be the oldest entries as new ones are
108    /// pushed. This means that we can optimize for those removals along with indexing performance,
109    /// rather than optimizing for more rare removals from the middle of the sequence.
110    deltas: VecDeque<Delta>,
111}
112
113#[allow(dead_code)] // Temporary
114impl History {
115    /// Constructs a new history container, containing at most `max_count` historical versions for
116    /// a tree.
117    #[must_use]
118    pub fn empty(max_count: usize) -> Self {
119        // We allocate one more than we actually need to store to allow us to insert and THEN
120        // remove, rather than the other way around. This leads to negligible increases in memory
121        // usage while allowing for cleaner code.
122        let deltas = VecDeque::with_capacity(max_count + 1);
123        Self { max_count, deltas }
124    }
125
126    /// Gets the maximum number of versions that this history can store.
127    #[must_use]
128    pub fn max_versions(&self) -> usize {
129        self.max_count
130    }
131
132    /// Gets the current number of versions in the history.
133    #[must_use]
134    pub fn num_versions(&self) -> usize {
135        self.deltas.len()
136    }
137
138    /// Returns all the roots that the history knows about.
139    ///
140    /// The iteration order of the roots is guaranteed to move backward in time, with earlier items
141    /// being roots from versions closer to the present.
142    ///
143    /// # Complexity
144    ///
145    /// Calling this method provides an iterator whose consumption requires a traversal of all the
146    /// versions. The method's complexity is thus `O(n)` in the number of versions.
147    pub fn roots(&self) -> impl Iterator<Item = RootValue> {
148        self.deltas.iter().rev().map(|d| d.root)
149    }
150
151    /// Gets the version corresponding to the provided `root`, or returns [`None`] if the provided
152    /// `root` is not found within this history.
153    pub fn version(&self, root: RootValue) -> Option<VersionId> {
154        self.deltas
155            .iter()
156            .find_map(|d| if d.root == root { Some(d.version_id) } else { None })
157    }
158
159    /// Returns `true` if `root` is in the history and `false` otherwise.
160    ///
161    /// # Complexity
162    ///
163    /// Calling this method requires a traversal of all the versions and is hence linear in the
164    /// number of history versions.
165    #[must_use]
166    pub fn is_known_root(&self, root: RootValue) -> bool {
167        self.deltas.iter().any(|r| r.root == root)
168    }
169
170    /// Returns the root value that corresponds to the provided `version`.
171    pub fn root_for_version(&self, version: VersionId) -> Result<RootValue> {
172        let ix = self.find_latest_corresponding_version(version)?;
173
174        // The direct index is safe here because `find_latest_...` will have returned an error if
175        // there is no such version, and is hence guaranteed to have returned a valid index.
176        Ok(self.deltas[ix].root)
177    }
178
179    /// Adds a version to the history with the provided `root` and represented by the changes from
180    /// the current tree given in `nodes` and `leaves`.
181    ///
182    /// If adding this version would result in exceeding `self.max_count` historical versions, then
183    /// the oldest of the versions is automatically removed.
184    ///
185    /// # Gotchas
186    ///
187    /// When constructing the `nodes` and `leaves`, keep in mind that those collections must contain
188    /// entries for the **default value of a node or leaf** at any position where the tree was
189    /// sparse in the state represented by `root`. If this is not done, incorrect values may be
190    /// returned.
191    ///
192    /// This is necessary because the changes are the _reverse_ from what one might expect. Namely,
193    /// the changes in a given version `v` must "_revert_" the changes made in the transition from
194    /// version `v` to version `v + 1`.
195    ///
196    /// # Errors
197    ///
198    /// - [`HistoryError::NonMonotonicVersions`] if the provided version is not greater than the
199    ///   previously added version.
200    pub fn add_version(
201        &mut self,
202        root: RootValue,
203        version_id: VersionId,
204        nodes: NodeChanges,
205        leaves: LeafChanges,
206    ) -> Result<()> {
207        if let Some(v) = self.deltas.iter().last() {
208            if v.version_id < version_id {
209                self.deltas.push_back(Delta::new(root, version_id, nodes, leaves));
210                if self.num_versions() > self.max_versions() {
211                    self.deltas.pop_front();
212                }
213
214                Ok(())
215            } else {
216                Err(HistoryError::NonMonotonicVersions(version_id, v.version_id))
217            }
218        } else {
219            self.deltas.push_back(Delta::new(root, version_id, nodes, leaves));
220
221            Ok(())
222        }
223    }
224
225    /// Returns the index in the sequence of deltas of the version that corresponds to the provided
226    /// `version_id`.
227    ///
228    /// To "correspond" means that it either has the provided `version_id`, or is the newest version
229    /// with a `version_id` less than the provided id. In either case, it is the correct version to
230    /// be used to query the tree state in the provided `version_id`.
231    ///
232    /// # Complexity
233    ///
234    /// Finding the latest corresponding version in the history requires a linear traversal of the
235    /// history entries, and hence has complexity `O(n)` in the number of versions.
236    ///
237    /// # Errors
238    ///
239    /// - [`HistoryError::HistoryEmpty`] if the history is empty and hence there is no version to
240    ///   find.
241    /// - [`HistoryError::VersionTooOld`] if the history does not contain the data to provide a
242    ///   coherent overlay for the provided `version_id` due to `version_id` being older than the
243    ///   oldest version stored.
244    fn find_latest_corresponding_version(&self, version_id: VersionId) -> Result<usize> {
245        // If the version is older than the oldest, we error.
246        if let Some(oldest_version) = self.deltas.front() {
247            if oldest_version.version_id > version_id {
248                return Err(HistoryError::VersionTooOld);
249            }
250        } else {
251            return Err(HistoryError::VersionTooOld);
252        }
253
254        let ix = self
255            .deltas
256            .iter()
257            .position(|d| d.version_id > version_id)
258            .unwrap_or_else(|| self.num_versions())
259            .checked_sub(1)
260            .expect(
261                "Subtraction should not overflow as we have ruled out the no-version \
262                case, and in the other cases the left operand will be >= 1",
263            );
264
265        Ok(ix)
266    }
267
268    /// Returns a view of the history that allows querying as a single unified overlay on the
269    /// current state of the merkle tree as if the overlay was reverting the tree to the state
270    /// corresponding to the specified `version_id`.
271    ///
272    /// Note that the history may not contain a version that directly corresponds to `version_id`.
273    /// In such a case, the view will instead use the newest version coherent with the provided
274    /// `version_id`, as this is the correct version for the provided id. Note that this will be
275    /// incorrect if the versions stored in the history do not represent contiguous changes from the
276    /// current tree.
277    ///
278    /// # Complexity
279    ///
280    /// The computational complexity of this method is linear in the number of versions stored in
281    /// the history.
282    ///
283    /// # Errors
284    ///
285    /// - [`HistoryError::VersionTooOld`] if the history does not contain the data to provide a
286    ///   coherent overlay for the provided `version_id` due to `version_id` being older than the
287    ///   oldest version stored.
288    pub fn get_view_at(&self, version_id: VersionId) -> Result<HistoryView<'_>> {
289        let version_index = self.find_latest_corresponding_version(version_id)?;
290        Ok(HistoryView::new_of(version_index, self))
291    }
292
293    /// Removes all versions in the history that are older than the version denoted by the provided
294    /// `version_id`.
295    ///
296    /// If `version_id` is not a version known by the history, it will keep the newest version that
297    /// is capable of serving as that version in queries.
298    ///
299    /// # Complexity
300    ///
301    /// The computational complexity of this method is linear in the number of versions stored in
302    /// the history prior to any removals.
303    pub fn truncate(&mut self, version_id: VersionId) -> usize {
304        // We start by getting the index to truncate to, though it is not an error to remove
305        // something too old.
306        let truncate_ix = self.find_latest_corresponding_version(version_id).unwrap_or(0);
307
308        for _ in 0..truncate_ix {
309            self.deltas.pop_front();
310        }
311
312        truncate_ix
313    }
314
315    /// Removes all versions from the history.
316    pub fn clear(&mut self) {
317        self.deltas.clear();
318    }
319}

324/// A read-only view of the history overlay on the tree at a specified place in the history.
325#[allow(dead_code)] // Temporary
326#[derive(Debug)]
327pub struct HistoryView<'history> {
328    /// The index of the target version in the history.
329    version_ix: usize,
330
331    /// The history that actually stores the data that will be queried.
332    history: &'history History,
333}
334
335#[allow(dead_code)] // Temporary
336impl<'history> HistoryView<'history> {
337    /// Constructs a new history view that acts as a single overlay of the state represented by the
338    /// oldest delta for which `f` returns true.
339    ///
340    /// # Complexity
341    ///
342    /// The computational complexity of this method is linear in the number of versions stored in
343    /// the history.
344    fn new_of(version_ix: usize, history: &'history History) -> Self {
345        Self { version_ix, history }
346    }
347
348    /// Gets the value of the node in the history at the provided `index`, or returns `None` if the
349    /// version does not overlay the current tree at that node.
350    ///
351    /// # Complexity
352    ///
353    /// The computational complexity of this method is linear in the number of versions due to the
354    /// need to traverse to find the correct overlay value.
355    #[must_use]
356    pub fn node_value(&self, index: &NodeIndex) -> Option<&Word> {
357        self.history
358            .deltas
359            .iter()
360            .skip(self.version_ix)
361            .find_map(|v| v.nodes.get(index))
362    }
363
364    /// Gets the value of the entire leaf in the history at the specified `index`, or returns `None`
365    /// if the version does not overlay the current tree at that leaf.
366    ///
367    /// # Complexity
368    ///
369    /// The computational complexity of this method is linear in the number of versions due to the
370    /// need to traverse to find the correct overlay value.
371    #[must_use]
372    pub fn leaf_value(&self, index: &LeafIndex<SMT_DEPTH>) -> Option<&CompactLeaf> {
373        self.history
374            .deltas
375            .iter()
376            .skip(self.version_ix)
377            .find_map(|v| v.leaves.get(index))
378    }
379
380    /// Queries the value of a specific key in a leaf in the overlay, returning:
381    ///
382    /// - `None` if the version does not overlay that leaf in the current tree,
383    /// - `Some(None)` if the version does overlay that leaf but the compact leaf does not contain
384    ///   that value,
385    /// - and `Some(Some(v))` if the version does overlay the leaf and the key exists in that leaf.
386    ///
387    /// # Complexity
388    ///
389    /// The computational complexity of this method is linear in the number of versions due to the
390    /// need to traverse to find the correct overlay value.
391    #[must_use]
392    pub fn value(&self, key: &Word) -> Option<Option<&Word>> {
393        self.leaf_value(&LeafIndex::from(*key)).map(|leaf| leaf.get(key))
394    }
395}

400/// A delta for a state `n` represents the changes (to both nodes and leaves) that need to be
401/// applied on top of the state `n + 1` to yield the correct tree for state `n`.
402///
403/// # Cumulative Deltas and Temporal Ordering
404///
405/// In order to best represent the history of a merkle tree, these deltas are constructed to take
406/// advantage of two main properties:
407///
408/// - They are _cumulative_, which reduces their practical memory usage. This does, however, mean
409///   that querying the state of older blocks is more expensive than querying newer ones.
410/// - Deltas are applied in **temporally reversed order** from what one might expect. Most
411///   conventional applications of deltas bring something from the past into the future through
412///   application. In our case, the application of one or more deltas moves the tree into a **past
413///   state**.
414///
415/// # Construction
416///
417/// While the [`Delta`] type is visible in the interface of the history, it is only intended to be
418/// constructed by the history. Users should not be allowed to construct it directly.
419#[derive(Clone, Debug, PartialEq)]
420struct Delta {
421    /// The root of the tree in the `version` corresponding to the application of the reversions in
422    /// this delta to the previous tree state.
423    pub root: RootValue,
424
425    /// The version of the tree represented by the delta.
426    pub version_id: VersionId,
427
428    /// Any changes to the non-leaf nodes in the tree for this delta.
429    pub nodes: NodeChanges,
430
431    /// Any changes to the leaf nodes in the tree for this delta.
432    ///
433    /// Note that the leaf state is **not represented compactly**, and describes the entire state
434    /// of the leaf in the corresponding version.
435    pub leaves: LeafChanges,
436}
437
438#[allow(dead_code)] // Temporary
439impl Delta {
440    /// Creates a new delta with the provided `root`, and representing the provided
441    /// changes to `nodes` and `leaves` in the merkle tree.
442    #[must_use]
443    fn new(
444        root: RootValue,
445        version_id: VersionId,
446        nodes: NodeChanges,
447        leaves: LeafChanges,
448    ) -> Self {
449        Self { root, version_id, nodes, leaves }
450    }
