1//! This module contains the data types used by the forest to store and manage the lineages that it
2//! knows about.
3
4use core::iter::once;
5
6use crate::merkle::smt::{
7    VersionId,
8    large_forest::{history::History, root::RootValue},
9};
10
11// LINEAGE DATA
12// ================================================================================================
13
14/// The data that the forest stores in memory for each lineage of trees.
15#[derive(Clone, Debug)]
16pub(super) struct LineageData {
17    /// The history of changes made to the lineage, representing a contiguous set of historical
18    /// trees in the lineage up to the configured maximum number of versions.
19    pub history: History,
20
21    /// The version of the latest tree in the lineage.
22    pub latest_version: VersionId,
23
24    /// The value of the root for the latest tree in the lineage.
25    pub latest_root: RootValue,
26}
27
28impl LineageData {
29    /// Gets an iterator that yields all roots in the lineage.
30    ///
31    /// The iteration order of the roots is guaranteed to move backward in time, with earlier items
32    /// in the iterator being roots from versions closer to the present. The current root of the
33    /// lineage will always be the first item that the iterator yields.
34    pub(super) fn roots(&self) -> impl Iterator<Item = RootValue> {
35        once(self.latest_root).chain(self.history.roots())
36    }
37
38    /// Truncates the information on this tree to the provided `version`, returning `true` if the
39    /// history is empty after truncation, and `false` otherwise.
40    ///
41    /// If the latest version in the lineage is older than the specified `version`, this latest
42    /// version is always retained. In other words, the method cannot prune a lineage from the
43    /// forest entirely.
44    pub(super) fn truncate(&mut self, version: VersionId) -> bool {
45        if version >= self.latest_version {
46            // Truncation in the history is defined such that it never removes a version that could
47            // possibly serve as the latest delta for a newer version. This is because it cannot
48            // safely know if a version `v` is between the latest delta `d` and the current version
49            // `c`, as it has no knowledge of the current version.
50            //
51            // Thus, if we have a version `v` such that `d <= v < c`, we need to retain the
52            // reversion delta `d` in the history to correctly service queries for `v`. If, however,
53            // we have `d < c <= v` we need to explicitly remove the last delta as well.
54            //
55            // To that end, we handle the latter case first, by explicitly calling
56            // `History::clear()`.
57            self.history.clear();
58            true
59        } else {
60            // The other case is `v < c`, which is handled simply by the truncation mechanism in the
61            // history as we want. In other words, it retains the necessary delta, and so we can
62            // just call it here.
63            self.history.truncate(version);
64            false
65        }
66    }
67}
