1//! Contains internal functionality for interacting with RocksDB that is not exposed in the RocksDB
2//! crate or the RocksDB C wrapper.
3
4use crate::merkle::smt::large_forest::backend::persistent::WriteBatch;
5
6/// Merges the provided [`WriteBatch`]es into a single batch using efficient raw-memory operations.
7///
8/// This is intended to be equivalent to `WriteBatchInternal::Append` in the C++ codebase, but that
9/// function is not exposed in the C API or the Rust wrapper for RocksDB.
10///
11/// **This function must be revisited every time the RocksDB crate is updated to ensure that it
12/// operates correctly.**
13///
14/// # `WriteBatch` Layout
15///
16/// The basic format of a `WriteBatch` is as follows:
17///
18/// ```text
19/// [sequence: u64le][count: u32le][record0][record1]...
20/// ```
21///
22/// Each record contains:
23///
24/// - A **type tag** (e.g. `kTypeColumnFamilyValue` or `kTypeColumnFamilyDeletion`) which determines
25///   how to read the rest of the underlying data.
26/// - A **column family ID** `varint32_cf_id` which identifies the column family into which the
27///   record will be written.
28/// - A **key** `&[u8]` that is the key in the column.
29/// - A **value** `&[u8]` that is the value for the given key in the column.
30///
31/// # Alternatives
32///
33/// An alternative to this would be to use a transaction-based RocksDB instance instead, and perform
34/// the merging using `Transaction::rebuild_from_writebatch`. This is semantically equivalent, but
35/// requires involving the whole transaction machinery. It ends up being significantly more
36/// heavyweight than this solution, almost doubling the amount of time the batch forest update takes
37/// to run.
38///
39/// # Panics
40///
41/// - If the data for either the left or right batch is corrupt, as this indicates a bug in the
42///   underlying RocksDB implementation and should not be continued with.
43pub fn merge_batches(left: WriteBatch, right: &WriteBatch) -> WriteBatch {
44    const SEQUENCE_SIZE: usize = size_of::<u64>();
45    const COUNT_SIZE: usize = size_of::<u32>();
46    const HEADER_SIZE: usize = SEQUENCE_SIZE + COUNT_SIZE;
47
48    let left_data = left.data();
49    let right_data = right.data();
50
51    // We initially check that we have enough data to make this at all sane, returning a corruption
52    // error if not.
53    if right_data.len() < HEADER_SIZE {
54        panic!("Right write batch contained insufficient data of {} bytes", right_data.len());
55    }
56    if left_data.len() < HEADER_SIZE {
57        panic!("Left write batch contained insufficient data of {} bytes", left_data.len());
58    }
59
60    // We then parse out the counts, returning an error if they are not valid u32 values.
61    let left_count = u32::from_le_bytes(
62        left_data[8..12]
63            .try_into()
64            .unwrap_or_else(|e| panic!("Left's count was not a valid u32: {e}")),
65    );
66    let right_count = u32::from_le_bytes(
67        right_data[8..12]
68            .try_into()
69            .unwrap_or_else(|e| panic!("Right's count was not a valid u32: {e}")),
70    );
71
72    // If that is good, we take `left` as our starting point, update the counts, and then append the
73    // records from right.
74    let mut new_data = left_data.to_vec();
75    new_data[SEQUENCE_SIZE..HEADER_SIZE].copy_from_slice(
76        left_count
77            .checked_add(right_count)
78            .expect("Overflow cannot occur")
79            .to_le_bytes()
80            .as_ref(),
81    );
82    new_data.extend_from_slice(&right_data[HEADER_SIZE..]);
83
84    WriteBatch::from_data(&new_data)
85}
