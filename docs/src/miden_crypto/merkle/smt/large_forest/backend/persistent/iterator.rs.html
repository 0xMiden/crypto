18/// The type of the underlying iterator over the database.
19///
20/// The type of items that it yields is not statically known, as it just provides bytes for
21/// key-value pairs. Decoding these into the correct types is up to the client of the type.
22pub type DBIterator<'db> = db::DBIteratorWithThreadMode<'db, super::DB>;
23
24// ENTRIES ITERATOR
25// ================================================================================================
26
27/// An iterator over the entries for a given tree in the backend.
28pub struct PersistentBackendEntriesIterator<'db> {
29    /// The lineage whose leaves are being iterated over.
30    pub lineage: LineageId,
31
32    /// The iterator over all leaves in the database.
33    iterator: DBIterator<'db>,
34
35    /// State-machine tracking.
36    state: PersistentBackendEntriesIteratorState,
37}
38impl<'db> PersistentBackendEntriesIterator<'db> {
39    /// Constructs a new such iterator in the starting state.
40    ///
41    /// The provided `iterator` must yield items where the key decodes to a `LeafKey` and the value
42    /// decodes to an `SmtLeaf`. If this is not the case, iteration will panic.
43    ///
44    /// For performance, this iterator should be passed a prefix iterator over the database with the
45    /// correct prefix (corresponding to the provided `lineage`) set, but it will still function
46    /// properly if this is not the case.
47    pub fn new(lineage: LineageId, iterator: DBIterator<'db>) -> Self {
48        let state = PersistentBackendEntriesIteratorState::NotInLeaf;
49        Self { lineage, iterator, state }
50    }
51}
52
53/// The internal state machine for the iterator.
54enum PersistentBackendEntriesIteratorState {
55    /// The iterator is not currently iterating over a particular leaf.
56    NotInLeaf,
57
58    /// The iterator is iterating over a particular leaf, with `leaf_entries` remaining to be
59    /// yielded in that leaf.
60    InLeaf {
61        /// The iterator over the current leaf's entries.
62        leaf_entries: Box<dyn Iterator<Item = (Word, Word)>>,
63    },
64}
65
66impl<'db> Iterator for PersistentBackendEntriesIterator<'db> {
67    type Item = TreeEntry;
68
69    /// Advances the iterator and returns the next item if present.
70    ///
71    /// # Panics
72    ///
73    /// - If unable to read from the underlying database.
74    fn next(&mut self) -> Option<Self::Item> {
75        loop {
76            match &mut self.state {
77                PersistentBackendEntriesIteratorState::NotInLeaf => {
78                    // Here we are not in a leaf of the targeted tree, so we have to see if we _can_
79                    // be.
80                    if let Some(entry) = self.iterator.next() {
81                        let (key_bytes, value_bytes) = entry.expect("Able to read from the DB");
82                        let key = LeafKey::read_from_bytes(&key_bytes)
83                            .expect("Leaf key data read from disk is not corrupt");
84
85                        // If the key isn't for the correct lineage (which can happen even with the
86                        // bloom filter), we need to advance by returning to the loop.
87                        if key.lineage != self.lineage {
88                            continue;
89                        }
90
91                        // If the key is valid, we need to read out the leaf itself and then start
92                        // iterating over that.
93                        let leaf = SmtLeaf::read_from_bytes(&value_bytes)
94                            .expect("Leaf data read from disk is not corrupt");
95                        let mut leaf_entries = leaf.into_entries();
96                        leaf_entries.sort_by_key(|(k, _)| *k);
97
98                        // We change state to being in the leaf, and then recurse to return a value.
99                        self.state = PersistentBackendEntriesIteratorState::InLeaf {
100                            leaf_entries: Box::new(leaf_entries.into_iter()),
101                        };
102                    } else {
103                        return None;
104                    }
105                },
106                PersistentBackendEntriesIteratorState::InLeaf { leaf_entries } => {
107                    if let Some((key, value)) = leaf_entries.next() {
108                        // Here we have an entry in the leaf, so we simply need to return it.
109                        return Some(TreeEntry { key, value });
110                    } else {
111                        // If we've run out of entries in the leaf itself, we need to see if there
112                        // is another valid leaf. We do this by changing state and looping to use
113                        // the existing logic.
114                        self.state = PersistentBackendEntriesIteratorState::NotInLeaf;
115                    }
116                },
117            }
118        }
119    }
120}
