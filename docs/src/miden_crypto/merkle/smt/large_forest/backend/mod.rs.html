29/// The backing storage for the SMT forest, providing the necessary high-level methods for
30/// performing operations on the full trees that make up the forest, while allowing the forest
31/// itself to be storage agnostic.
32///
33/// # Backend Data Storage
34///
35/// Having a generic [`Backend`] provides no guarantees to the user about how it stores data and
36/// what patterns are used for data access under the hood. It is, however, guaranteed to store
37/// _only_ the data necessary to describe the latest state of each tree in the forest.
38///
39/// # Error Handling
40///
41/// We separate errors in backend implementations into two semantic categories:
42///
43/// 1. **User-Derived Errors:** These are errors that arise downstream of data provided by the user.
44///    These errors must be signaled by returning an [`Err`] variant with an appropriate error.
45/// 2. **Internal Errors:** These are errors that are not derived from data provided by the user.
46///    Signaling such an error is up to the implementation, but can be done through both panicking
47///    and returning the [`BackendError::Internal`] variant as appropriate. These **may leave the
48///    backend in an inconsistent state** as they are designed to effect program termination or
49///    perform it directly.
50///
51/// The only reason that [`BackendError::Internal`] exists is to allow certain failures to result in
52/// termination at the level of the _forest_ instead of the _backend_ as this can sometimes lead to
53/// cleaner logic. If this is not appropriate, a panic is a better option.
54///
55/// # Expected Behavior
56///
57/// Certain methods on this trait (e.g. [`Backend::update_tree`]) provide behaviors expected for
58/// that method. These combine with the following trait-level behavior requirements to become part
59/// of the contract of the method, but a portion that cannot be encoded in the type system. Any
60/// failure to conform to these expected behaviors is **considered a bug in the implementation** of
61/// the backend, and must be rectified.
62///
63/// The following behavior is expected of all methods in implementations of this trait:
64///
65/// - For any failure derived from user input (see _User-Derived Errors_ above), the data and the
66///   backend must be **left in a consistent state** when the error is returned to the caller.
67/// - Failures derived from user input (see _User-Derived Errors_ above) must be signaled to the
68///   caller by returning a variant of [`BackendError`] that is **not [`BackendError::Internal`]**.
69///   Methods may place additional constraints on which errors are used to signal certain failures.
70pub trait Backend
71where
72    Self: Debug,
73{
74    // QUERIES
75    // ============================================================================================
76
77    /// Returns an opening for the specified `key` in the SMT with the specified `lineage`.
78    ///
79    /// It is the responsibility of the forest to ensure lineage existence before querying the
80    /// backend. The backend must return an error if the lineage does not exist.
81    fn open(&self, lineage: LineageId, key: Word) -> Result<SmtProof>;
82
83    /// Returns the value associated with the provided `key` in the SMT with the specified
84    /// `lineage`, or [`None`] if no such value exists.
85    ///
86    /// It is the responsibility of the forest to ensure lineage existence before querying the
87    /// backend. The backend must return an error if the lineage does not exist.
88    fn get(&self, lineage: LineageId, key: Word) -> Result<Option<Word>>;
89
90    /// Returns the version of the tree with the specified `lineage`.
91    ///
92    /// It is the responsibility of the forest to ensure lineage existence before querying the
93    /// backend. The backend must return an error if the lineage does not exist.
94    fn version(&self, lineage: LineageId) -> Result<VersionId>;
95
96    /// Returns an iterator over all the lineages that the backend knows about.
97    ///
98    /// The iteration order is unspecified.
99    fn lineages(&self) -> Result<impl Iterator<Item = LineageId>>;
100
101    /// Returns an iterator over all the trees (and their corresponding roots) that the backend
102    /// knows about.
103    ///
104    /// The iteration order is unspecified.
105    fn trees(&self) -> Result<impl Iterator<Item = TreeWithRoot>>;
106
107    /// Returns the total number of (key-value) entries in the specified `lineage`.
108    ///
109    /// It is the responsibility of the forest to ensure lineage existence before querying the
110    /// backend. The backend must return an error if the lineage does not exist.
111    fn entry_count(&self, lineage: LineageId) -> Result<usize>;
112
113    /// Returns an iterator that yields the populated (key-value) entries for the specified
114    /// `lineage`.
115    ///
116    /// It is the responsibility of the forest to ensure lineage existence before querying the
117    /// backend. The backend must return an error if the lineage does not exist.
118    ///
119    /// This iterator must yield entries in an order such that they are sorted by their leaf index,
120    /// and entries that share a leaf index are sorted by key. It must not include key-value pairs
121    /// where the value is the empty word.
122    fn entries(&self, lineage: LineageId) -> Result<impl Iterator<Item = TreeEntry>>;
123
124    // SINGLE-TREE MODIFIERS
125    // ============================================================================================
126
127    /// Adds a new `lineage` to the forest with the provided `version` and sets the associated SMT
128    /// to have the value created by applying `updates` to the empty tree, returning the new root of
129    /// that tree.
130    ///
131    /// # Expected Behavior
132    ///
133    /// Implementations must guarantee the following behavior in addition to the global invariants:
134    ///
135    /// - If the provided `lineage` conflicts with an already-existing lineage in the backend, it
136    ///   must return [`BackendError::DuplicateLineage`].
137    fn add_lineage(
138        &mut self,
139        lineage: LineageId,
140        version: VersionId,
141        updates: SmtUpdateBatch,
142    ) -> Result<TreeWithRoot>;
143
144    /// Performs the provided `updates` on the tree with the specified `lineage`, returning the
145    /// mutation set that will revert the changes made to the tree.
146    ///
147    /// # Expected Behavior
148    ///
149    /// Implementations must guarantee the following behavior in addition to the global invariants:
150    ///
151    /// - At most one new root must be added to the forest for the entire batch.
152    /// - If applying the provided `updates` results in no changes to the tree, no new tree must be
153    ///   allocated.
154    fn update_tree(
155        &mut self,
156        lineage: LineageId,
157        new_version: VersionId,
158        updates: SmtUpdateBatch,
159    ) -> Result<MutationSet>;
160
161    // MULTI-TREE MODIFIERS
162    // ============================================================================================
163
164    /// Performs the provided `updates` on the forest, setting all new tree states to have the
165    /// provided `new_version` and returning a vector of the mutation sets that reverse the changes
166    /// to each changed tree.
167    ///
168    /// # Expected Behavior
169    ///
170    /// Implementations must guarantee the following behavior in addition to the global invariants:
171    ///
172    /// - At most one new root must be added to the forest for each target root in the provided
173    ///   `updates`.
174    /// - If applying the provided `updates` results in no changes to a given lineage of trees in
175    ///   the forest, then no new tree must be allocated in that lineage.
176    fn update_forest(
177        &mut self,
178        new_version: VersionId,
179        updates: SmtForestUpdateBatch,
180    ) -> Result<Vec<(LineageId, MutationSet)>>;
181}

186/// The error type for use within Backends.
187#[derive(Debug, Error)]
188pub enum BackendError {
189    /// Raised when there is a conflict between an existing lineage ID and one already in the
190    /// forest.
191    #[error("Duplicate lineage ID {0} provided")]
192    DuplicateLineage(LineageId),
193
194    /// Raised for arbitrary errors that are not derived from user-input. These should be considered
195    /// fatal by callers, but exist to forward the termination decision up to an appropriate level.
196    #[error(transparent)]
197    Internal(Box<dyn core::error::Error + Sync + Send>),
198
199    /// Raised when there is an error with the merkle tree semantics within the backend.
200    #[error(transparent)]
201    Merkle(#[from] MerkleError),
202
203    /// Raised for arbitrary other errors within the backend that are derived from user-input and
204    /// hence non-fatal.
205    #[error(transparent)]
206    Other(Box<dyn core::error::Error + Sync + Send>),
207
208    /// Raised when the backend is queried for a lineage it doesn't know about.
209    #[error("Lineage {0} is not known by the backend")]
210    UnknownLineage(LineageId),
211}
212
213impl BackendError {
214    /// Constructs an internal error variant from the provided concrete error `e`.
215    fn internal_from<E: core::error::Error + Sync + Send + 'static>(e: E) -> Self {
216        Self::Internal(Box::new(e))
217    }
218}
219
220/// The result type for use with backends.
221pub type Result<T> = core::result::Result<T, BackendError>;
