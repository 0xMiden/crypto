1//! This module contains the definition of the [`ForestOperation`] type that encapsulates the
2//! possible modifications made to a tree, as well as the concept of a [`SmtUpdateBatch`] of
3//! operations to be performed on a single tree in the forest. This is then extended to
4//! [`SmtForestUpdateBatch`], which defines a batch of operations across multiple trees.
5
6use alloc::vec::Vec;
7
8use crate::{EMPTY_WORD, Map, Set, Word, merkle::smt::large_forest::root::LineageId};

10// FOREST OPERATION
11// ================================================================================================
12
13/// The operations that can be performed on an arbitrary leaf in a tree in a forest.
14#[derive(Clone, Debug, Eq, PartialEq)]
15pub enum ForestOperation {
16    /// An insertion of `value` under `key` into the tree.
17    ///
18    /// If `key` already exists in the tree, the associated value will be replaced with `value`
19    /// instead.
20    Insert { key: Word, value: Word },
21
22    /// The removal of the `key` and its associated value from the tree.
23    Remove { key: Word },
24}
25impl ForestOperation {
26    /// Insert the provided `value` into a tree under the provided `key`.
27    pub fn insert(key: Word, value: Word) -> Self {
28        Self::Insert { key, value }
29    }
30
31    /// Remove the provided `key` and its associated value from a tree.
32    pub fn remove(key: Word) -> Self {
33        Self::Remove { key }
34    }
35
36    /// Retrieves the key from the operation.
37    pub fn key(&self) -> Word {
38        match self {
39            ForestOperation::Insert { key, .. } => *key,
40            ForestOperation::Remove { key } => *key,
41        }
42    }
43}
44
45impl From<ForestOperation> for (Word, Word) {
46    fn from(value: ForestOperation) -> Self {
47        match value {
48            ForestOperation::Insert { key, value } => (key, value),
49            ForestOperation::Remove { key } => (key, EMPTY_WORD),
50        }
51    }
52}

57/// A batch of operations that can be performed on an arbitrary tree in a forest.
58#[derive(Clone, Debug, Eq, PartialEq)]
59pub struct SmtUpdateBatch {
60    /// The operations to be performed on a tree.
61    operations: Vec<ForestOperation>,
62}
63impl SmtUpdateBatch {
64    /// Creates an empty batch of operations that, when applied, will produce a tree with the
65    /// provided `version` when applied.
66    pub fn empty() -> Self {
67        Self { operations: vec![] }
68    }
69
70    /// Creates a batch containing the provided `operations` that will produce a tree with the
71    /// provided `version` when applied.
72    pub fn new(operations: impl Iterator<Item = ForestOperation>) -> Self {
73        Self {
74            operations: operations.collect::<Vec<_>>(),
75        }
76    }
77
78    /// Adds the provided `operations` to the batch.
79    pub fn add_operations(&mut self, operations: impl Iterator<Item = ForestOperation>) {
80        self.operations.extend(operations);
81    }
82
83    /// Adds the [`ForestOperation::Insert`] operation for the provided `key` and `value` pair to
84    /// the batch.
85    pub fn add_insert(&mut self, key: Word, value: Word) {
86        self.operations.push(ForestOperation::insert(key, value));
87    }
88
89    /// Adds the [`ForestOperation::Remove`] operation for the provided `key` to the batch.
90    pub fn add_remove(&mut self, key: Word) {
91        self.operations.push(ForestOperation::remove(key));
92    }
93
94    /// Consumes the batch as a vector of operations, containing the last operation for any given
95    /// `key` in the case that multiple operations per key are encountered.
96    ///
97    /// This vector is guaranteed to be sorted by the key on which an operation is performed.
98    pub fn consume(self) -> Vec<ForestOperation> {
99        // As we want to keep the LAST operation for each key, rather than the first, we filter in
100        // reverse.
101        let mut seen_keys: Set<Word> = Set::new();
102        let mut ops = self
103            .operations
104            .into_iter()
105            .rev()
106            .filter(|o| seen_keys.insert(o.key()))
107            .collect::<Vec<_>>();
108        ops.sort_by_key(|o| o.key());
109        ops
110    }
111}
112
113impl IntoIterator for SmtUpdateBatch {
114    type Item = ForestOperation;
115    type IntoIter = alloc::vec::IntoIter<Self::Item>;
116
117    /// Consumes the batch as an iterator yielding operations while respecting the guarantees given
118    /// by [`Self::consume`].
119    ///
120    /// The iteration order is unspecified.
121    fn into_iter(self) -> Self::IntoIter {
122        self.consume().into_iter()
123    }
124}
125
126impl From<SmtUpdateBatch> for Vec<(Word, Word)> {
127    fn from(value: SmtUpdateBatch) -> Self {
128        value
129            .consume()
130            .into_iter()
131            .map(|op| match op {
132                ForestOperation::Insert { key, value } => (key, value),
133                ForestOperation::Remove { key } => (key, EMPTY_WORD),
134            })
135            .collect()
136    }
137}
138
139impl<I> From<I> for SmtUpdateBatch
140where
141    I: Iterator<Item = ForestOperation>,
142{
143    fn from(value: I) -> Self {
144        Self::new(value)
145    }
146}
147
148impl From<SmtUpdateBatch> for Vec<ForestOperation> {
149    /// The vector is guaranteed to be sorted by the key on which an operation is performed, and to
150    /// only contain the _last_ operation to be performed on any given key.
151    fn from(value: SmtUpdateBatch) -> Self {
152        value.consume()
153    }
154}
155
156impl Default for SmtUpdateBatch {
157    fn default() -> Self {
158        Self::empty()
159    }
160}

165/// A batch of operations that can be performed on an arbitrary forest, consisting of operations
166/// associated with specified trees in that forest.
167#[derive(Clone, Debug, Eq, PartialEq)]
168pub struct SmtForestUpdateBatch {
169    /// The operations associated with each targeted lineage in the forest.
170    operations: Map<LineageId, SmtUpdateBatch>,
171}
172
173impl SmtForestUpdateBatch {
174    /// Creates a new, empty, batch of operations.
175    pub fn empty() -> Self {
176        Self { operations: Map::new() }
177    }
178
179    /// Adds the provided `operations` to be performed on the tree with the specified `lineage`.
180    pub fn add_operations(
181        &mut self,
182        lineage: LineageId,
183        operations: impl Iterator<Item = ForestOperation>,
184    ) {
185        let batch = self.operations.entry(lineage).or_insert_with(SmtUpdateBatch::empty);
186        batch.add_operations(operations);
187    }
188
189    /// Gets the batch of operations for the tree with the specified `lineage` for inspection and/or
190    /// modification.
191    ///
192    /// It is assumed that calling this means that the caller wants to insert operations into the
193    /// associated batch, so a batch will be created even if one was not previously present.
194    pub fn operations(&mut self, lineage: LineageId) -> &mut SmtUpdateBatch {
195        self.operations.entry(lineage).or_insert_with(SmtUpdateBatch::empty)
196    }
197
198    /// Gets an iterator over the lineages
199    pub fn lineages(&self) -> impl Iterator<Item = &LineageId> {
200        self.operations.keys()
201    }
202
203    /// Consumes the batch as a map of batches, with each individual batch guaranteed to be in
204    /// sorted order and contain only the last operation in the batch for any given key.
205    pub fn consume(self) -> Map<LineageId, Vec<ForestOperation>> {
206        self.operations.into_iter().map(|(k, v)| (k, v.consume())).collect()
207    }
208}
209
210impl IntoIterator for SmtForestUpdateBatch {
211    type Item = (LineageId, Vec<ForestOperation>);
212    type IntoIter = crate::MapIntoIter<LineageId, Vec<ForestOperation>>;
213
214    /// Consumes the batch as an iterator yielding pairs of `(lineage, operations)` while respecting
215    /// the guarantees given by [`Self::consume`].
216    ///
217    /// The iteration order is unspecified.
218    fn into_iter(self) -> Self::IntoIter {
219        self.consume().into_iter()
220    }
221}

228    use itertools::Itertools;

233    #[test]
234    fn tree_batch() {
235        let mut rng = ContinuousRng::new([0x12; 32]);
236
237        // We start by creating an empty tree batch.
238        let mut batch = SmtUpdateBatch::empty();
239
240        // Let's make three operations on different keys...
241        let o1_key: Word = rng.value();
242        let o1_value: Word = rng.value();
243        let o2_key: Word = rng.value();
244        let o3_key: Word = rng.value();
245        let o3_value: Word = rng.value();
246
247        let o1 = ForestOperation::insert(o1_key, o1_value);
248        let o2 = ForestOperation::remove(o2_key);
249        let o3 = ForestOperation::insert(o3_key, o3_value);
250
251        // ... and stick them in the batch in various ways
252        batch.add_operations(vec![o1.clone()].into_iter());
253        batch.add_remove(o2_key);
254        batch.add_insert(o3_key, o3_value);
255
256        // We save a copy of the batch for later as we have more testing to do.
257        let batch_tmp = batch.clone();
258
259        // If we then consume the batch, we should have the operations ordered by their key.
260        let ops = batch.consume();
261        assert!(ops.is_sorted_by_key(|o| o.key()));
262
263        // Let's now make two additional operations with keys that overlay with keys from the first
264        // three...
265        let o4_key = o2_key;
266        let o4_value: Word = rng.value();
267        let o5_key = o1_key;
268
269        let o4 = ForestOperation::insert(o4_key, o4_value);
270        let o5 = ForestOperation::remove(o5_key);
271
272        // ... and also stick them into the batch.
273        let mut batch = batch_tmp;
274        batch.add_operations(vec![o4.clone(), o5.clone()].into_iter());
275
276        // Now if we consume the batch we should have three operations, and they should be the last
277        // operation for each key.
278        let ops = batch.consume();
279
280        assert_eq!(ops.len(), 3);
281        assert!(ops.is_sorted_by_key(|o| o.key()));
282
283        assert!(ops.contains(&o3));
284        assert!(ops.contains(&o4));
285        assert!(!ops.contains(&o2));
286        assert!(ops.contains(&o5));
287        assert!(!ops.contains(&o1));
288    }
289
290    #[test]
291    fn forest_batch() {
292        let mut rng = ContinuousRng::new([0x13; 32]);
293
294        // We can start by creating an empty forest batch.
295        let mut batch = SmtForestUpdateBatch::empty();
296
297        // Let's start by adding a few operations to a tree.
298        let t1_lineage: LineageId = rng.value();
299        let t1_o1 = ForestOperation::insert(rng.value(), rng.value());
300        let t1_o2 = ForestOperation::remove(rng.value());
301        batch.add_operations(t1_lineage, vec![t1_o1, t1_o2].into_iter());
302
303        // We can also add them differently.
304        let t2_lineage: LineageId = rng.value();
305        let t2_o1 = ForestOperation::remove(rng.value());
306        let t2_o2 = ForestOperation::insert(rng.value(), rng.value());
307        batch.operations(t2_lineage).add_operations(vec![t2_o1, t2_o2].into_iter());
308
309        // When we consume the batch, each per-tree batch should be unique by key and sorted.
310        let ops = batch.consume();
311        assert_eq!(ops.len(), 2);
312
313        let t1_ops = ops.get(&t1_lineage).unwrap();
314        assert!(t1_ops.is_sorted_by_key(|o| o.key()));
315        assert_eq!(t1_ops.iter().unique_by(|o| o.key()).count(), 2);
316
317        let t2_ops = ops.get(&t2_lineage).unwrap();
318        assert!(t2_ops.is_sorted_by_key(|o| o.key()));
319        assert_eq!(t2_ops.iter().unique_by(|o| o.key()).count(), 2);
320    }
