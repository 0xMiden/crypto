1//! This module contains the configuration structure for the forest.
2
3// CONSTANTS
4// ================================================================================================
5
6/// The default number of historical versions of each tree to keep.
7pub const DEFAULT_MAX_HISTORY_VERSIONS: usize = 10;
8
9/// The minimum number of historical versions per lineage that the forest can store.
10pub const MIN_HISTORY_VERSIONS: usize = 1;
11
12// CONFIG
13// ================================================================================================
14
15/// The configuration for the forest's behavior.
16#[derive(Clone, Debug, Eq, PartialEq)]
17pub struct Config {
18    /// The maximum number of historical versions that the forest will keep for any given lineage.
19    max_historical_versions: usize,
20}
21
22/// This block contains the accessors for the configuration options.
23impl Config {
24    /// The maximum number of historical versions that the forest will keep for any given lineage.
25    ///
26    /// If this field is set to `n`, the forest will implicitly store `n + 1` versions of a given
27    /// lineage once the latest version in that lineage is accounted for.
28    ///
29    /// Defaults to [`DEFAULT_MAX_HISTORY_VERSIONS`].
30    pub fn max_history_versions(&self) -> usize {
31        self.max_historical_versions
32    }
33}
34
35// BUILDERS
36// ================================================================================================
37
38/// This impl block contains the builder functions for the configuration options.
39impl Config {
40    /// Sets the maximum number of historical versions that the forest will store for any given
41    /// lineage, clamping to [`MIN_HISTORY_VERSIONS`] on the low end.
42    ///
43    /// If this field is set to `n`, the forest will implicitly store `n + 1` versions of a given
44    /// lineage once the latest version in that lineage is accounted for.
45    ///
46    /// This defaults to [`DEFAULT_MAX_HISTORY_VERSIONS`].
47    pub fn with_max_history_versions(mut self, max_historical_versions: usize) -> Self {
48        self.max_historical_versions = if max_historical_versions < MIN_HISTORY_VERSIONS {
49            MIN_HISTORY_VERSIONS
50        } else {
51            max_historical_versions
52        };
53        self
54    }
55}
56
57// TRAIT IMPLS
58// ================================================================================================
59
60/// Please see individual methods on [`Config`] for the default value of each configuration option.
61impl Default for Config {
62    fn default() -> Self {
63        let max_historical_versions = DEFAULT_MAX_HISTORY_VERSIONS;
64        Self { max_historical_versions }
65    }
66}
