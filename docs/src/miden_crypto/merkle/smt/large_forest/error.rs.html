16// LARGE SMT FOREST ERROR
17// ================================================================================================
18
19/// The type of errors returned by operations on the large SMT forest.
20#[derive(Debug, Error)]
21pub enum LargeSmtForestError {
22    /// Raised when the provided version for any update is older than the latest-known version for
23    /// the lineage being updated.
24    #[error("Version {provided} is not newer than latest-known {latest}")]
25    BadVersion { provided: VersionId, latest: VersionId },
26
27    /// Raised when there is a conflict between an existing lineage ID and one already in the
28    /// forest.
29    #[error("Duplicate lineage ID {0} provided")]
30    DuplicateLineage(LineageId),
31
32    /// Raised for arbitrary errors that are not derived from user-input. These **must be considered
33    /// fatal by the caller**, but exist to provide the caller with control over process termination
34    /// (e.g. for improved diagnostics or error reporting) wherever possible.
35    #[error(transparent)]
36    Fatal(Box<dyn core::error::Error + Sync + Send>),
37
38    /// Errors in the history subsystem of the forest.
39    #[error(transparent)]
40    History(#[from] HistoryError),
41
42    /// Errors with the merkle tree operations of the forest.
43    #[error(transparent)]
44    Merkle(#[from] MerkleError),
45
46    /// Errors in working with leaves in the merkle trees.
47    #[error(transparent)]
48    SmtLeaf(#[from] SmtLeafError),
49
50    /// Errors in the construction and manipulation of SMT proofs.
51    #[error(transparent)]
52    SmtProof(#[from] SmtProofError),
53
54    /// Raised when an operation specifies a lineage that is not known.
55    #[error("The lineage {0:?} is not in the forest")]
56    UnknownLineage(LineageId),
57
58    /// Raised when an operation specifies a tree that is not known.
59    #[error("The tree {0} is not in the forest")]
60    UnknownTree(TreeId),
61
62    /// Raised when an operation requests a version that is not known.
63    #[error("The version {0} is not known by the forest")]
64    UnknownVersion(VersionId),
65
66    /// Raised for arbitrary other errors.
67    #[error(transparent)]
68    Other(#[from] Box<dyn core::error::Error + Sync + Send>),
69}
70
71impl LargeSmtForestError {
72    /// Constructs a fatal error variant from the provided concrete error `e`.
73    pub fn fatal_from<E: core::error::Error + Sync + Send + 'static>(e: E) -> Self {
74        Self::Fatal(Box::new(e))
75    }
76}
77
78/// We want to forward backend errors specifically when we can, so we manually implement the
79/// conversion.
80impl From<BackendError> for LargeSmtForestError {
81    fn from(value: BackendError) -> Self {
82        match value {
83            BackendError::DuplicateLineage(l) => LargeSmtForestError::DuplicateLineage(l),
84            BackendError::Internal(e) => LargeSmtForestError::Fatal(e),
85            BackendError::Merkle(e) => LargeSmtForestError::from(e),
86            BackendError::Other(e) => LargeSmtForestError::from(e),
87            BackendError::UnknownLineage(t) => LargeSmtForestError::UnknownLineage(t),
88        }
89    }
90}

92/// The result type for use within the large SMT forest portion of the library.
93pub type Result<T> = core::result::Result<T, LargeSmtForestError>;
