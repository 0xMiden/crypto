1//! This module contains the implementation of the iterator over the entries of an arbitrary tree in
2//! the forest.
3//!
4//! # Performance
5//!
6//! The performance of this iterator has a significant dependency on the tree that it is running
7//! over. Due to the differing performance characteristics of backends, we cannot provide exact
8//! performance bounds, but the following general rules apply.
9//!
10//! - Iterating over the entries of the **latest tree in a lineage** is going to be **the fastest
11//!   possible query**. This depends only on the direct iteration performance of the backend in
12//!   question.
13//! - Iterating over the entries of **a historical tree is going to be slower**. This is because it
14//!   has to do work to merge the entries provided by the history with the entries of the full tree
15//!   in order to create a coherent picture of the historical tree.
16//!
17//! We highly recommend benchmarking the iteration behavior on the concrete workload(s) you are
18//! concerned about, rather than trying to statically reason about performance of this iterator.
19

28// ENTRIES ITERATOR
29// ================================================================================================
30
31/// An iterator over the entries of an arbitrary tree in the forest, yielding entries in an
32/// arbitrary order.
33///
34/// It is split into two variants for performance, as iterating over a full tree is significantly
35/// simpler than iterating over a historical tree. While it would be nice to be able to return one
36/// of two different iterators depending on the circumstances of construction, Rust's `impl Trait`
37/// bounds do not allow for this.
38///
39/// The iterator **must never transition between variants** during the process of iteration.
40pub(super) enum EntriesIterator<'forest> {
41    /// An iterator over a tree in the forest that is formed from a merger of the full tree and a
42    /// historical overlay.
43    WithHistory {
44        /// The iterator over the entries in the full tree.
45        ///
46        /// This iterator should never yield any entries where `value == EMPTY_WORD`.
47        full_tree_iter: Peekable<Box<dyn Iterator<Item = TreeEntry> + 'forest>>,
48
49        /// The iterator over the entries in the history.
50        ///
51        /// This iterator may yield entries with `value == EMPTY_WORD`. These are explicit
52        /// reversions of entries newly-set in newer versions, and so should be used. While they
53        /// technically should only ever correspond to a case where they _are_ reverting a
54        /// newly-set entry, care must be taken to remove them regardless if they do not match up
55        /// for some reason.
56        history_entries_iter: Peekable<Box<dyn Iterator<Item = TreeEntry> + 'forest>>,
57
58        /// The current state of the iteration state machine.
59        state: EntriesIteratorState,
60    },
61
62    /// An iterator over a tree in the forest that is simply an iterator over the full tree.
63    WithoutHistory {
64        /// The iterator over the entries in the full tree.
65        full_tree_iter: Box<dyn Iterator<Item = TreeEntry> + 'forest>,
66    },
67}
68
69impl<'forest> EntriesIterator<'forest> {
70    /// Constructs a new entries iterator pointing to the first item in the designated `tree` in the
71    /// `forest`, formed by combining a historical overlay with the current tree.
72    ///
73    /// Note that it _does not_ perform checks as to the correctness of the provided iterators. If
74    /// these are not an iterator over the full tree and the historical entries in turn, the results
75    /// the iterator yields will be invalid.
76    pub(super) fn new_with_history(
77        full_tree_iter: impl Iterator<Item = TreeEntry> + 'forest,
78        history_entries_iter: impl Iterator<Item = TreeEntry> + 'forest,
79    ) -> Self {
80        // This type gymnastics is unfortunately necessary to let us easily store the `Peekable`
81        // which we need to avoid carrying additional state in the state machine.
82        let full_tree_iter: Box<dyn Iterator<Item = _>> = Box::new(full_tree_iter);
83        let history_entries_iter: Box<dyn Iterator<Item = _>> = Box::new(history_entries_iter);
84
85        // We begin in `NotInLeaf`. This is implicitly `Start -> NotInLeaf`
86        Self::WithHistory {
87            full_tree_iter: full_tree_iter.peekable(),
88            history_entries_iter: history_entries_iter.peekable(),
89            state: EntriesIteratorState::NotInLeaf,
90        }
91    }
92
93    /// Constructs a new entries iterator pointing to the first item in the designated `tree` in the
94    /// `forest` without any associated history.
95    ///
96    /// Note that it _does not_ check whether `full_tree_iter` is actually an iterator over the
97    /// full tree. If it is not, the iterator will yield invalid results.
98    pub(super) fn new_without_history(
99        full_tree_iter: impl Iterator<Item = TreeEntry> + 'forest,
100    ) -> Self {
101        let full_tree_iter = Box::new(full_tree_iter);
102        Self::WithoutHistory { full_tree_iter }
103    }
104
105    /// Advances the iterator and returns the next value in the case where it is iterating over a
106    /// historical tree version.
107    ///
108    /// For the details of the state machine that this implements, please see the documentation for
109    /// the [`EntriesIteratorState`]. It explains the valid state transitions and the conditions
110    /// under which they occur. This implementation does not match them directly in order to
111    /// simplify the logic, but matches the intended semantics.
112    ///
113    /// # Panics
114    ///
115    /// - If the method is called with a `self` that is not in the [`Self::WithHistory`] variant.
116    #[inline(always)] // To help the optimizer eliminate the redundant check in Iterator::next()
117    fn next_with_history(&mut self) -> Option<TreeEntry> {
118        let EntriesIterator::WithHistory {
119            full_tree_iter,
120            history_entries_iter,
121            state,
122        } = self
123        else {
124            panic!("EntriesIterator::next_with_history called without history")
125        };
126
127        match state {
128            EntriesIteratorState::NotInLeaf => {
129                // Here we are (semantically) not pointing to any specific leaf, so we need to work
130                // out which of our possible outgoing transitions take place. This state does not
131                // actually return anything except in the `-> End` case.
132                match (full_tree_iter.peek(), history_entries_iter.peek()) {
133                    (None, None) => {
134                        // No more entries exist in either of the iterators. `NotInLeaf -> End`.
135                        None
136                    },
137                    (Some(_), None) => {
138                        // Entries only exist in the full tree iterator. `NotInLeaf -> TreeOnly`
139                        *state = EntriesIteratorState::TreeOnly;
140                        self.next_with_history()
141                    },
142                    (None, Some(_)) => {
143                        // Entries only exist in the full tree iterator. `NotInLeaf -> HistOnly`
144                        *state = EntriesIteratorState::HistOnly;
145                        self.next_with_history()
146                    },
147                    (Some(full), Some(hist)) => {
148                        // Entries exist in both, but the exact state transition has not yet been
149                        // determined. We have three other possible outgoing edges from `NotInLeaf`.
150                        let full_idx = LeafIndex::from(full.key);
151                        let hist_idx = LeafIndex::from(hist.key);
152
153                        if full_idx == hist_idx {
154                            // We are in the same leaf. `NotInLeaf -> InLeafShared`
155                            *state = EntriesIteratorState::InLeafShared;
156                        } else if full_idx < hist_idx {
157                            // We are in different leaves with full_idx coming first. `NotInLeaf ->
158                            // InLeafTreeOnly`
159                            *state = EntriesIteratorState::InLeafTreeOnly;
160                        } else {
161                            // We are in different leaves with hist_idx coming first. `NotInLeaf ->
162                            // InTreeHistOnly`.
163                            *state = EntriesIteratorState::InLeafHistOnly;
164                        }
165
166                        self.next_with_history()
167                    },
168                }
169            },
170            EntriesIteratorState::HistOnly => {
171                // In this state we simply can continue yielding the history entries iterator until
172                // it is empty. We just have to check that we're not yielding EMPTY_WORD entries
173                // directly as these should not be seen.
174                history_entries_iter.next().and_then(|e| {
175                    if e.value == EMPTY_WORD {
176                        self.next_with_history()
177                    } else {
178                        Some(e)
179                    }
180                })
181            },
182            EntriesIteratorState::TreeOnly => {
183                // In this state we can simply continue yielding the tree entries iterator until it
184                // is empty. When it returns `None` we have `TreeOnly -> End`
185                full_tree_iter.next()
186            },
187            EntriesIteratorState::InLeafHistOnly => {
188                // Here, we are in a leaf that is only in the history. We technically only want to
189                // transition out of this state once we have exhausted the leaf, but in actuality we
190                // can rely on the logic for `NotInLeaf` to do the right thing here. We only have to
191                // skip empty words as these should never be yielded.
192                *state = EntriesIteratorState::NotInLeaf;
193                history_entries_iter.next().and_then(|e| {
194                    if e.value == EMPTY_WORD {
195                        self.next_with_history()
196                    } else {
197                        Some(e)
198                    }
199                })
200            },
201            EntriesIteratorState::InLeafTreeOnly => {
202                // Here we are in a leaf that is only in the full tree. We technically only want to
203                // transition out of this state once we have exhausted the leaf, but in actuality we
204                // can rely on the logic for `NotInleaf` to do the right thing here.
205                *state = EntriesIteratorState::NotInLeaf;
206                full_tree_iter.next()
207            },
208            EntriesIteratorState::InLeafShared => {
209                // Here we have both iterators in the same LEAF but that does not mean they have the
210                // same item.
211                let hist_item =
212                    history_entries_iter.peek().expect("Entry already checked to exist");
213                let tree_item = full_tree_iter.peek().expect("Entry already checked to exist");
214
215                if hist_item.key == tree_item.key {
216                    *state = EntriesIteratorState::InLeafBothKeysEq;
217                } else if hist_item.key < tree_item.key {
218                    *state = EntriesIteratorState::InLeafBothHistPrio;
219                } else {
220                    *state = EntriesIteratorState::InLeafBothTreePrio;
221                }
222
223                self.next_with_history()
224            },
225            EntriesIteratorState::InLeafBothKeysEq => {
226                // If the keys are equal we want to pop both entries and only return the history's
227                // one. We can again rely on `NotInLeaf` to do our logic correctly.
228                *state = EntriesIteratorState::NotInLeaf;
229
230                // We can discard this entry entirely as it has been overwritten.
231                full_tree_iter.next();
232
233                // But this one may or may not need to be returned.
234                let hist_item =
235                    history_entries_iter.next().expect("Entry already checked to exist");
236                if hist_item.value == EMPTY_WORD {
237                    // We never want to yield empty items, so we skip them.
238                    self.next_with_history()
239                } else {
240                    // Otherwise the item is real and we want to yield it.
241                    Some(hist_item)
242                }
243            },
244            EntriesIteratorState::InLeafBothHistPrio => {
245                // Here we have a history item with a key < the full tree item, so we want to return
246                // that. We can again rely on `NotInLeaf` to do our logic correctly.
247                *state = EntriesIteratorState::NotInLeaf;
248                history_entries_iter.next()
249            },
250            EntriesIteratorState::InLeafBothTreePrio => {
251                // Here we have a full tree item with a key < the history item, so we want to return
252                // that. We can again rely on `NotInLeaf` to do our logic correctly.
253                *state = EntriesIteratorState::NotInLeaf;
254                full_tree_iter.next()
255            },
256        }
257    }
258
259    /// Advances the iterator and returns the next value in the case where it is iterating over the
260    /// current tree version.
261    ///
262    /// # Panics
263    ///
264    /// - If the method is called with a `self` that is not the [`Self::WithoutHistory`] variant.
265    #[inline(always)] // To help the optimizer eliminate the redundant check in Iterator::next()
266    fn next_without_history(&mut self) -> Option<TreeEntry> {
267        let EntriesIterator::WithoutHistory { full_tree_iter } = self else {
268            panic!("EntriesIterator::next_without_history called with history")
269        };
270
271        full_tree_iter.next()
272    }
273}

278impl Iterator for EntriesIterator<'_> {
279    type Item = TreeEntry;
280
281    fn next(&mut self) -> Option<Self::Item> {
282        match self {
283            EntriesIterator::WithHistory { .. } => self.next_with_history(),
284            EntriesIterator::WithoutHistory { .. } => self.next_without_history(),
285        }
286    }
287}

292/// The state machine that is the entries iterator for the forest.
293///
294/// We do not represent the ghost states of `Start` and `End`, so [`Self::NotInLeaf`] serves as the
295/// initial state of the machine in practice. A full diagram of the state machine's allowable
296/// transitions can be found below. See the individual variants for the conditions under which these
297/// transitions take place.
298///
299/// ```text
300///                                    ┌─────────┐
301///                                    │  Start  │
302///                                    └─────────┘
303///                                         │
304///                                         │
305///                                         ▼
306///                                   ┌───────────┐
307///        ┌─────────────┬────────────│           │◀──────────────┬──────────────────┐
308///        │             │            │ NotInLeaf │               │                  │
309///        │             │       ┌────│           │────────────┬──┼───────────────┐  │
310///        │             │       │    └───────────┘            │  │               │  │
311///        │             │       │         │  ▲                │  │               │  │
312///        │             │       │         │  │                │  │               │  │
313///        │             │       │         │  │                │  │               │  │
314///        │             │       │         │  │                │  │               │  │
315///        ▼             ▼       │         ▼  │                ▼  │               ▼  │
316///  ┌──────────┐  ┌──────────┐  │  ┌────────────────┐  ┌────────────────┐  ┌──────────────┐
317///  │ TreeOnly │  │ HistOnly │  │  │ InLeafHistOnly │  │ InLeafTreeOnly │  │ InLeafShared │◀─────────────┐
318///  └──────────┘  └──────────┘  │  └────────────────┘  └────────────────┘  └──────────────┘              │
319///        │             │       │                                                  │                     │
320///        │             │       │                                                  │                     │
321///        │             │       │            ┌──────────────────────┬──────────────┴────────┐            │
322///        │             │       │            │                      │                       │            │
323///        │             │       │            ▼                      ▼                       ▼            │
324///        │             │       │  ┌──────────────────┐  ┌────────────────────┐  ┌────────────────────┐  │
325///        └─────────────┴─────┐ │  │ InLeafBothKeysEq │  │ InLeafBothHistPrio │  │ InLeafBothTreePrio │  │
326///                            │ │  └──────────────────┘  └────────────────────┘  └────────────────────┘  │
327///                            │ │            │                      │                       │            │
328///                            │ │            │                      │                       │            │
329///                            │ │            └──────────────────────┴───────────────────────┴────────────┘
330///                            ▼ ▼
331///                        ┌─────────┐
332///                        │   End   │
333///                        └─────────┘
334/// ```
335///
336/// Note that this describes the _semantics_ of the transitions between states, and may not directly
337/// correspond to the implementation in [`EntriesIterator::next_with_history`] for reasons of
338/// performance and maintainability.
339pub(super) enum EntriesIteratorState {
340    /// The iterator is currently not in any leaf.
341    ///
342    /// This state should not advance the underlying iterators directly, and the iterator is not
343    /// intended to return a value for `next` while in this state.
344    ///
345    /// Incoming state transitions:
346    ///
347    /// - `Start -> NotInLeaf`: The state of the state machine.
348    /// - `InLeafHistOnly -> NotInLeaf`: Upon completing the leaf in the history.
349    /// - `InLeafTreeOnly -> NotInLeaf`: Upon completing the leaf in the tree.
350    /// - `InLeafShared -> NotInLeaf`: Upon completing the leaf that exists in both.
351    ///
352    /// Outgoing state transitions:
353    ///
354    /// - `NotInLeaf -> End`: If neither iterator has remaining entries.
355    /// - `NotInLeaf -> HistOnly`: If the tree entries iterator is empty.
356    /// - `NotInLeaf -> TreeOnly`: If the history entries iterator is empty.
357    /// - `NotInLeaf -> InLeafHistOnly`: If the next leaf is only in the history.
358    /// - `NotInLeaf -> InLeafTreeOnly`: If the next leaf is only in the tree.
359    /// - `NotInLeaf -> InLeafShared`: If the leaf exists in both iterators.
360    NotInLeaf,
361
362    /// The iterator over the full tree has no entries, so we can iterate only over the history
363    /// until completion.
364    ///
365    /// Incoming state transitions:
366    ///
367    /// - `NotInLeaf -> HistOnly`: The tree entries iterator is empty.
368    ///
369    /// Outgoing state transitions:
370    ///
371    /// - `HistOnly -> End`: The history entries iterator is empty.
372    HistOnly,
373
374    /// The iterator over the history has no entries, so we can iterate only over the full tree
375    /// until completion.
376    ///
377    /// Incoming state transitions:
378    ///
379    /// - `NotInLeaf -> TreeOnly`: The history entries iterator is empty.
380    ///
381    /// Outgoing state transitions:
382    ///
383    /// - `TreeOnly -> End`: The tree entries iterator is empty.
384    TreeOnly,
385
386    /// The iterator is operating over a leaf that only exists in the history iterator.
387    ///
388    /// Incoming state transitions:
389    ///
390    /// - `NotInLeaf -> InLeafHistOnly`: The tree entries iterator has items but the latest is not
391    ///   in the same leaf as the history's latest.
392    ///
393    /// Outgoing state transitions:
394    ///
395    /// - `InLeafHistOnly -> NotInLeaf`: Upon completing iteration through the current leaf.
396    InLeafHistOnly,
397
398    /// The iterator is operating over a leaf that only exists in the tree iterator.
399    ///
400    /// Incoming state transitions:
401    ///
402    /// - `NotInLeaf -> InLeafTreeOnly`: The history entries iterator has items but the latest is
403    ///   not in the same leaf as the tree's latest.
404    ///
405    /// Outgoing state transitions:
406    ///
407    /// - `InLeafTreeOnly -> NotInLeaf`: Upon completing iteration through the current leaf.
408    InLeafTreeOnly,
409
410    /// The iterator is operating over a leaf that exists in both iterators.
411    ///
412    /// Incoming state transitions:
413    ///
414    /// - `NotInLeaf -> InLeafShared`: Both iterators have their latest entry in the same leaf.
415    ///
416    /// Outgoing state transitions:
417    ///
418    /// - `InLeafShared -> InLeafBothKeysEq`: If the two keys in the shared leaf are equal.
419    /// - `InLeafShared -> InLeafBothKeysHistPrio`: If the key in the history < the key in the tree.
420    /// - `InLeafShared -> InLeafBothKeysTreePrio`: If the key in the tree < the key in the history.
421    /// - `InLeafShared -> NotInLeaf`: Upon completing iteration through the current leaf.
422    InLeafShared,
423
424    /// The iterator is operating over a leaf that exists in both iterators, and the current keys
425    /// are the same.
426    ///
427    /// Incoming state transitions:
428    ///
429    /// - `InLeafShared -> InLeafBothKeysEq`: If the key in each iterator is the same.
430    ///
431    /// Outgoing state transitions:
432    ///
433    /// - `InLeafBothKeysEq -> InLeafShared`: When needing to check the next element.
434    InLeafBothKeysEq,
435
436    /// The iterator is operating over a leaf that exists in both iterators, and the current key
437    /// in the history is less than the current key in the tree.
438    ///
439    /// Incoming state transitions:
440    ///
441    /// - `InLeafShared -> InLeafBothHistPrio`: If the key in the history iterator < the key in the
442    ///   tree iterator.
443    ///
444    /// Outgoing state transitions:
445    ///
446    /// - `InLeafBothHistPrio -> InLeafShared`: When needing to check the next element.
447    InLeafBothHistPrio,
448
449    /// The iterator is operating over a leaf that exists in both iterators, and the current key in
450    /// the tree is less than the current key in the history.
451    ///
452    /// Incoming state transitions:
453    ///
454    /// - `InLeafShared -> InLeafBothTreePrio`: If the key in the tree iterator < the key in the
455    ///   history iterator.
456    ///
457    /// Outgoing state transitions:
458    ///
459    /// - `InLeafBothTreePrio -> InLeafShared`: When needing to check the next element.
460    InLeafBothTreePrio,
461}
