1//! This module contains the implementation of the iterator over the entries of an arbitrary tree in
2//! the forest.
3//!
4//! # Performance
5//!
6//! The performance of this iterator has a significant dependency on the tree that it is running
7//! over. Due to the differing performance characteristics of backends, we cannot provide exact
8//! performance bounds, but the following general rules apply.
9//!
10//! - Iterating over the entries of the **latest tree in a lineage** is going to be **the fastest
11//!   possible query**. This depends only on the direct iteration performance of the backend in
12//!   question.
13//! - Iterating over the entries of **a historical tree is going to be slower**. This is because it
14//!   has to do work to merge the entries provided by the history with the entries of the full tree
15//!   in order to create a coherent picture of the historical tree.
16//!
17//! We highly recommend benchmarking the iteration behavior on the concrete workload(s) you are
18//! concerned about, rather than trying to statically reason about performance of this iterator.
19

30// ENTRIES ITERATOR
31// ================================================================================================
32
33/// An iterator over the entries of an arbitrary tree in the forest, yielding entries in an
34/// arbitrary order.
35///
36/// It is split into two variants for performance, as iterating over a full tree is significantly
37/// simpler than iterating over a historical tree. While it would be nice to be able to return one
38/// of two different iterators depending on the circumstances of construction, Rust's `impl Trait`
39/// bounds do not allow for this.
40///
41/// The iterator **must never transition between variants** during the process of iteration.
42///
43/// The order in which the results are yielded is undefined.
44pub(super) enum EntriesIterator<'forest> {
45    /// An iterator over a tree in the forest that is formed from a merger of the full tree and a
46    /// historical overlay.
47    WithHistory {
48        /// The iterator over the entries in the full tree.
49        ///
50        /// This iterator should never yield any entries where `value == EMPTY_WORD`.
51        full_tree_iter: Peekable<Box<dyn Iterator<Item = TreeEntry> + 'forest>>,
52
53        /// The view into the history at the correct point.
54        history_view: HistoryView<'forest>,
55
56        /// Tracks the keys in the history that are given values that have already been yielded.
57        yielded_history_keys: Set<Word>,
58
59        /// The current state of the iteration state machine.
60        state: EntriesIteratorState<'forest>,
61    },
62
63    /// An iterator over a tree in the forest that is simply an iterator over the full tree.
64    WithoutHistory {
65        /// The iterator over the entries in the full tree.
66        full_tree_iter: Box<dyn Iterator<Item = TreeEntry> + 'forest>,
67    },
68}
69
70impl<'forest> EntriesIterator<'forest> {
71    /// Constructs a new entries iterator pointing to the first item in the designated `tree` in the
72    /// `forest`, formed by combining a historical overlay with the current tree.
73    ///
74    /// Note that it _does not_ perform checks as to the correctness of the provided iterators.
75    pub(super) fn new_with_history(
76        full_tree_iter: impl Iterator<Item = TreeEntry> + 'forest,
77        history_view: HistoryView<'forest>,
78    ) -> Self {
79        // This type gymnastics is unfortunately necessary to let us easily store the `Peekable`
80        // which we need to avoid carrying additional state in the state machine.
81        let full_tree_iter: Box<dyn Iterator<Item = _>> = Box::new(full_tree_iter);
82
83        Self::WithHistory {
84            full_tree_iter: full_tree_iter.peekable(),
85            history_view,
86            yielded_history_keys: Set::new(),
87            state: EntriesIteratorState::Initial,
88        }
89    }
90
91    /// Constructs a new entries iterator pointing to the first item in the designated `tree` in the
92    /// `forest` without any associated history.
93    ///
94    /// Note that it _does not_ check whether `full_tree_iter` is actually an iterator over the
95    /// full tree. If it is not, this iterator will yield invalid results.
96    pub(super) fn new_without_history(
97        full_tree_iter: impl Iterator<Item = TreeEntry> + 'forest,
98    ) -> Self {
99        let full_tree_iter = Box::new(full_tree_iter);
100        Self::WithoutHistory { full_tree_iter }
101    }
102
103    /// Advances the iterator and returns the next value in the case where it is iterating over a
104    /// historical tree version.
105    ///
106    /// # Panics
107    ///
108    /// - If the method is called with a `self` that is not in the [`Self::WithHistory`] variant.
109    #[inline(always)] // To help the optimizer eliminate the redundant check in Iterator::next()
110    fn next_with_history(&mut self) -> Option<TreeEntry> {
111        let EntriesIterator::WithHistory {
112            full_tree_iter,
113            history_view,
114            yielded_history_keys,
115            state,
116        } = self
117        else {
118            panic!("EntriesIterator::next_with_history called without history")
119        };
120
121        loop {
122            match state {
123                EntriesIteratorState::Initial => {
124                    // In the initial state we need to advance to the appropriate next state.
125                    if full_tree_iter.peek().is_none() {
126                        // If there is nothing left to read from the full-tree iterator, we move
127                        // into the history iteration state.
128                        let iterator = Box::new(
129                            history_view
130                                .changed_keys()
131                                .map(|(key, value)| TreeEntry { key, value }),
132                        );
133                        *state = EntriesIteratorState::ReadingHistory { iterator };
134                    } else {
135                        // Otherwise we proceed with the full tree.
136                        *state = EntriesIteratorState::ReadingFullTree;
137                    }
138                },
139                EntriesIteratorState::ReadingFullTree => {
140                    // If we are reading the full tree's iterator, we need to keep advancing as long
141                    // as we can, but we have to remove entries using the
142                    // history.
143                    if full_tree_iter.peek().is_none() {
144                        // If we have nothing left to read, we transition into the reading history
145                        // state,
146                        let iterator = Box::new(
147                            history_view
148                                .changed_keys()
149                                .map(|(key, value)| TreeEntry { key, value }),
150                        );
151                        *state = EntriesIteratorState::ReadingHistory { iterator };
152                        return self.next_with_history();
153                    }
154
155                    let Some(entry) = full_tree_iter.next() else {
156                        unreachable!("The iterator is known to have another item available");
157                    };
158
159                    let value = if let Some(v) = history_view.value(&entry.key) {
160                        // If the history has a value for this key, then we need to return that
161                        // instead. We also store the key so we do not return it again by accident.
162                        yielded_history_keys.insert(entry.key);
163
164                        // If it is an empty value we don't want to return it so we continue the
165                        // outer loop.
166                        if v.is_empty() {
167                            continue;
168                        }
169
170                        TreeEntry { key: entry.key, value: v }
171                    } else {
172                        entry
173                    };
174
175                    return Some(value);
176                },
177                EntriesIteratorState::ReadingHistory { iterator } => {
178                    // This is a terminal state. We cannot transition to any other state from here,
179                    // and so the iterator ends when the iterator over the added
180                    // history items does.
181                    for entry in iterator.by_ref() {
182                        if !yielded_history_keys.contains(&entry.key) && !entry.value.is_empty() {
183                            return Some(entry);
184                        }
185
186                        // Here we have already returned this entry, or it is empty. In both cases
187                        // we want to skip it, so we let the loop continue.
188                    }
189                    return None;
190                },
191            }
192        }
193    }
194
195    /// Advances the iterator and returns the next value in the case where it is iterating over the
196    /// current tree version.
197    ///
198    /// # Panics
199    ///
200    /// - If the method is called with a `self` that is not the [`Self::WithoutHistory`] variant.
201    #[inline(always)] // To help the optimizer eliminate the redundant check in Iterator::next()
202    fn next_without_history(&mut self) -> Option<TreeEntry> {
203        let EntriesIterator::WithoutHistory { full_tree_iter } = self else {
204            panic!("EntriesIterator::next_without_history called with history")
205        };
206
207        full_tree_iter.next()
208    }
209}

211// ITERATOR TRAIT
212// ================================================================================================
213
214impl Iterator for EntriesIterator<'_> {
215    type Item = TreeEntry;
216
217    fn next(&mut self) -> Option<Self::Item> {
218        match self {
219            EntriesIterator::WithHistory { .. } => self.next_with_history(),
220            EntriesIterator::WithoutHistory { .. } => self.next_without_history(),
221        }
222    }
223}

225// ENTRIES ITERATOR STATE

228/// The state machine that is the entries iterator for the forest.
229pub(super) enum EntriesIteratorState<'forest> {
230    /// The initial state of the iterator, indicating that it has never advanced the underlying
231    /// iterator.
232    Initial,
233
234    /// Indicates that the iterator is currently reading from the full-tree's iterator, and yielding
235    /// results from that.
236    ReadingFullTree,
237
238    /// Indicates that the iterator is currently reading from the additional pairs added by the
239    /// history delta.
240    ReadingHistory {
241        /// The iterator over the entries that are _added_ by the history.
242        iterator: Box<dyn Iterator<Item = TreeEntry> + 'forest>,
243    },
244}
