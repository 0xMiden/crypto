9// MMR PEAKS
10// ================================================================================================
11
12#[derive(Debug, Clone, PartialEq, Eq)]
13#[cfg_attr(feature = "serde", derive(serde::Deserialize, serde::Serialize))]
14pub struct MmrPeaks {
15    /// The number of leaves (represented by [`Forest`]) is used to differentiate MMRs that have
16    /// the same number of peaks. This happens because the number of peaks goes up-and-down as
17    /// the structure is used causing existing trees to be merged and new ones to be created.
18    /// As an example, every time the MMR has a power-of-two number of leaves there is a single
19    /// peak.
20    ///
21    /// Every tree in the MMR forest has a distinct power-of-two size, this means only the right-
22    /// most tree can have an odd number of elements (i.e. `1`). Additionally this means that the
23    /// bits in `num_leaves` conveniently encode the size of each individual tree.
24    ///
25    /// Examples:
26    ///
27    /// - With 5 leaves, the binary `0b101`. The number of set bits is equal the number of peaks,
28    ///   in this case there are 2 peaks. The 0-indexed least-significant position of the bit
29    ///   determines the number of elements of a tree, so the rightmost tree has `2**0` elements
30    ///   and the left most has `2**2`.
31    /// - With 12 leaves, the binary is `0b1100`, this case also has 2 peaks, the leftmost tree has
32    ///   `2**3=8` elements, and the right most has `2**2=4` elements.
33    forest: Forest,
34
35    /// All the peaks of every tree in the MMR forest. The peaks are always ordered by number of
36    /// leaves, starting from the peak with most children, to the one with least.
37    ///
38    /// Invariant: The length of `peaks` must be equal to the number of true bits in `num_leaves`.
39    peaks: Vec<Word>,
40}
41
42impl Default for MmrPeaks {
43    /// Returns new [`MmrPeaks`] instantiated from an empty vector of peaks and 0 leaves.
44    fn default() -> Self {
45        Self {
46            forest: Forest::empty(),
47            peaks: Vec::new(),
48        }
49    }
50}
51
52impl MmrPeaks {
53    // CONSTRUCTOR
54    // --------------------------------------------------------------------------------------------
55
56    /// Returns new [MmrPeaks] instantiated from the provided vector of peaks and the number of
57    /// leaves in the underlying MMR.
58    ///
59    /// # Errors
60    /// Returns an error if the number of leaves and the number of peaks are inconsistent.
61    pub fn new(forest: Forest, peaks: Vec<Word>) -> Result<Self, MmrError> {
62        if forest.num_trees() != peaks.len() {
63            return Err(MmrError::InvalidPeaks(format!(
64                "number of one bits in leaves is {} which does not equal peak length {}",
65                forest.num_trees(),
66                peaks.len()
67            )));
68        }
69
70        Ok(Self { forest, peaks })
71    }
72
73    // ACCESSORS
74    // --------------------------------------------------------------------------------------------
75
76    /// Returns the underlying forest (a set of mountain range peaks).
77    pub fn forest(&self) -> Forest {
78        self.forest
79    }
80
81    /// Returns a count of leaves in the underlying MMR.
82    pub fn num_leaves(&self) -> usize {
83        self.forest.num_leaves()
84    }
85
86    /// Returns the number of peaks of the underlying MMR.
87    pub fn num_peaks(&self) -> usize {
88        self.peaks.len()
89    }
90
91    /// Returns the list of peaks of the underlying MMR.
92    pub fn peaks(&self) -> &[Word] {
93        &self.peaks
94    }
95
96    /// Returns the peak by the provided index.
97    ///
98    /// # Errors
99    /// Returns an error if the provided peak index is greater or equal to the current number of
100    /// peaks in the Mmr.
101    pub fn get_peak(&self, peak_idx: usize) -> Result<&Word, MmrError> {
102        self.peaks
103            .get(peak_idx)
104            .ok_or(MmrError::PeakOutOfBounds { peak_idx, peaks_len: self.peaks.len() })
105    }
106
107    /// Converts this [MmrPeaks] into its components: number of leaves (represented as a [`Forest`])
108    /// and a vector of peaks of the underlying MMR.
109    pub fn into_parts(self) -> (Forest, Vec<Word>) {
110        (self.forest, self.peaks)
111    }
112
113    /// Hashes the peaks.
114    ///
115    /// The procedure will:
116    /// - Flatten and pad the peaks to a vector of Felts.
117    /// - Hash the vector of Felts.
118    pub fn hash_peaks(&self) -> Word {
119        Poseidon2::hash_elements(&self.flatten_and_pad_peaks())
120    }
121
122    /// Verifies the Merkle opening proof.
123    ///
124    /// # Errors
125    /// Returns an error if:
126    /// - provided opening proof is invalid.
127    /// - Mmr root value computed using the provided leaf value differs from the actual one.
128    pub fn verify(&self, value: Word, opening: MmrProof) -> Result<(), MmrError> {
129        let root = self.get_peak(opening.peak_index())?;
130        opening
131            .path()
132            .merkle_path()
133            .verify(opening.relative_pos() as u64, value, root)
134            .map_err(MmrError::InvalidMerklePath)
135    }
136
137    /// Flattens and pads the peaks to make hashing inside of the Miden VM easier.
138    ///
139    /// The procedure will:
140    /// - Flatten the vector of Words into a vector of Felts.
141    /// - Pad the peaks with ZERO to an even number of words, this removes the need to handle
142    ///   Poseidon2 padding.
143    /// - Pad the peaks to a minimum length of 16 words, which reduces the constant cost of hashing.
144    pub fn flatten_and_pad_peaks(&self) -> Vec<Felt> {
145        let num_peaks = self.peaks.len();
146
147        // To achieve the padding rules above we calculate the length of the final vector.
148        // This is calculated as the number of field elements. Each peak is 4 field elements.
149        // The length is calculated as follows:
150        // - If there are less than 16 peaks, the data is padded to 16 peaks and as such requires 64
151        //   field elements.
152        // - If there are more than 16 peaks and the number of peaks is odd, the data is padded to
153        //   an even number of peaks and as such requires `(num_peaks + 1) * 4` field elements.
154        // - If there are more than 16 peaks and the number of peaks is even, the data is not padded
155        //   and as such requires `num_peaks * 4` field elements.
156        let len = if num_peaks < 16 {
157            64
158        } else if num_peaks % 2 == 1 {
159            (num_peaks + 1) * 4
160        } else {
161            num_peaks * 4
162        };
163
164        let mut elements = Vec::with_capacity(len);
165        elements.extend_from_slice(Word::words_as_elements(&self.peaks));
166        elements.resize(len, ZERO);
167        elements
168    }
169}
170
171impl From<MmrPeaks> for Vec<Word> {
172    fn from(peaks: MmrPeaks) -> Self {
173        peaks.peaks
174    }
