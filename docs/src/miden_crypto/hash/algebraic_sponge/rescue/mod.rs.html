33#[inline(always)]
34fn apply_sbox(state: &mut [Felt; STATE_WIDTH]) {
35    state[0] = state[0].exp_const_u64::<7>();
36    state[1] = state[1].exp_const_u64::<7>();
37    state[2] = state[2].exp_const_u64::<7>();
38    state[3] = state[3].exp_const_u64::<7>();
39    state[4] = state[4].exp_const_u64::<7>();
40    state[5] = state[5].exp_const_u64::<7>();
41    state[6] = state[6].exp_const_u64::<7>();
42    state[7] = state[7].exp_const_u64::<7>();
43    state[8] = state[8].exp_const_u64::<7>();
44    state[9] = state[9].exp_const_u64::<7>();
45    state[10] = state[10].exp_const_u64::<7>();
46    state[11] = state[11].exp_const_u64::<7>();
47}

49// INVERSE SBOX FUNCTION
50// ================================================================================================
51
52#[inline(always)]
53fn apply_inv_sbox(state: &mut [Felt; STATE_WIDTH]) {
54    // compute base^10540996611094048183 using 72 multiplications per array element
55    // 10540996611094048183 = b1001001001001001001001001001000110110110110110110110110110110111
56
57    // compute base^10
58    let mut t1 = *state;
59    t1.iter_mut().for_each(|t| *t = t.square());
60
61    // compute base^100
62    let mut t2 = t1;
63    t2.iter_mut().for_each(|t| *t = t.square());
64
65    // compute base^100100
66    let t3 = exp_acc::<Felt, STATE_WIDTH, 3>(t2, t2);
67
68    // compute base^100100100100
69    let t4 = exp_acc::<Felt, STATE_WIDTH, 6>(t3, t3);
70
71    // compute base^100100100100100100100100
72    let t5 = exp_acc::<Felt, STATE_WIDTH, 12>(t4, t4);
73
74    // compute base^100100100100100100100100100100
75    let t6 = exp_acc::<Felt, STATE_WIDTH, 6>(t5, t3);
76
77    // compute base^1001001001001001001001001001000100100100100100100100100100100
78    let t7 = exp_acc::<Felt, STATE_WIDTH, 31>(t6, t6);
79
80    // compute base^1001001001001001001001001001000110110110110110110110110110110111
81    for (i, s) in state.iter_mut().enumerate() {
82        let a = (t7[i].square() * t6[i]).square().square();
83        let b = t1[i] * t2[i] * *s;
84        *s = a * b;
85    }
86
87    #[inline(always)]
88    fn exp_acc<B: Field, const N: usize, const M: usize>(base: [B; N], tail: [B; N]) -> [B; N] {
89        let mut result = base;
90        for _ in 0..M {
91            result.iter_mut().for_each(|r| *r = r.square());
92        }
93        result.iter_mut().zip(tail).for_each(|(r, t)| *r *= t);
94        result
95    }
96}
