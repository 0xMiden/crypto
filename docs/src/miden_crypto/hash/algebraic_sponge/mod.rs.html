1//! Algebraic sponge-based hash functions.
2//!
3//! These are hash functions based on the sponge construction, which itself is defined from
4//! a cryptographic permutation function and a padding rule.
5//!
6//! Throughout the module, the padding rule used is the one in <https://eprint.iacr.org/2023/1045>.
7//! The core of the definition of an algebraic sponge-based hash function is then the definition
8//! of its cryptographic permutation function. This can be done by implementing the trait
9//! `[AlgebraicSponge]` which boils down to implementing the `apply_permutation` method.
10//!
11//! There are currently three algebraic sponge-based hash functions implemented in the module, RPO
12//! and RPX hash functions, both of which belong to the Rescue family of hash functions, and
13//! Poseidon2 hash function.
14

25// CONSTANTS
26// ================================================================================================
27
28/// Sponge state is set to 12 field elements or 96 bytes; 8 elements are reserved for the rate and
29/// the remaining 4 elements are reserved for the capacity.
30pub(crate) const STATE_WIDTH: usize = 12;
31
32/// The rate portion of the state is located in elements 0 through 7.
33pub(crate) const RATE_RANGE: Range<usize> = 0..8;
34pub(crate) const RATE_WIDTH: usize = RATE_RANGE.end - RATE_RANGE.start;
35
36/// The first and second 4-element words of the rate portion.
37pub(crate) const RATE0_RANGE: Range<usize> = 0..4;
38pub(crate) const RATE1_RANGE: Range<usize> = 4..8;
39
40/// The capacity portion of the state is located in elements 8, 9, 10, and 11.
41pub(crate) const CAPACITY_RANGE: Range<usize> = 8..12;
42
43/// The output of the hash function is a digest which consists of 4 field elements or 32 bytes,
44/// taken from the first word of the rate portion of the state.
45pub(crate) const DIGEST_RANGE: Range<usize> = 0..4;
46
47/// The number of byte chunks defining a field element when hashing a sequence of bytes
48const BINARY_CHUNK_SIZE: usize = 7;
49
50// ALGEBRAIC SPONGE
51// ================================================================================================
52
53pub(crate) trait AlgebraicSponge {
54    fn apply_permutation(state: &mut [Felt; STATE_WIDTH]);
55
56    /// Returns a hash of the provided field elements.
57    fn hash_elements<E>(elements: &[E]) -> Word
58    where
59        E: BasedVectorSpace<Felt>,
60    {
61        // Count total number of base field elements without collecting
62        let total_len = elements
63            .iter()
64            .map(|elem| E::as_basis_coefficients_slice(elem).len())
65            .sum::<usize>();
66
67        // initialize state to all zeros, except for the first element of the capacity part, which
68        // is set to `total_len % RATE_WIDTH`.
69        let mut state = [ZERO; STATE_WIDTH];
70        state[CAPACITY_RANGE.start] = Felt::from_u8((total_len % RATE_WIDTH) as u8);
71
72        // absorb elements into the state one by one until the rate portion of the state is filled
73        // up; then apply the permutation and start absorbing again; repeat until all
74        // elements have been absorbed
75        let mut i = 0;
76        for elem in elements.iter() {
77            for &felt in E::as_basis_coefficients_slice(elem) {
78                state[RATE_RANGE.start + i] = felt;
79                i += 1;
80                if i.is_multiple_of(RATE_WIDTH) {
81                    Self::apply_permutation(&mut state);
82                    i = 0;
83                }
84            }
85        }
86
87        // if we absorbed some elements but didn't apply a permutation to them (would happen when
88        // the number of elements is not a multiple of RATE_WIDTH), apply the permutation after
89        // padding by as many 0 as necessary to make the input length a multiple of the RATE_WIDTH.
90        if i > 0 {
91            while i != RATE_WIDTH {
92                state[RATE_RANGE.start + i] = ZERO;
93                i += 1;
94            }
95            Self::apply_permutation(&mut state);
96        }
97
98        // return the digest portion of the state as hash result
99        Word::new(state[DIGEST_RANGE].try_into().unwrap())
100    }
101
102    /// Returns a hash of the provided sequence of bytes.
103    fn hash(bytes: &[u8]) -> Word {
104        // initialize the state with zeroes
105        let mut state = [ZERO; STATE_WIDTH];
106
107        // determine the number of field elements needed to encode `bytes` when each field element
108        // represents at most 7 bytes.
109        let num_field_elem = bytes.len().div_ceil(BINARY_CHUNK_SIZE);
110
111        // set the first capacity element to `RATE_WIDTH + (num_field_elem % RATE_WIDTH)`. We do
112        // this to achieve:
113        // 1. Domain separating hashing of `[u8]` from hashing of `[Felt]`.
114        // 2. Avoiding collisions at the `[Felt]` representation of the encoded bytes.
115        state[CAPACITY_RANGE.start] =
116            Felt::from_u8((RATE_WIDTH + (num_field_elem % RATE_WIDTH)) as u8);
117
118        // initialize a buffer to receive the little-endian elements.
119        let mut buf = [0_u8; 8];
120
121        // iterate the chunks of bytes, creating a field element from each chunk and copying it
122        // into the state.
123        //
124        // every time the rate range is filled, a permutation is performed. if the final value of
125        // `rate_pos` is not zero, then the chunks count wasn't enough to fill the state range,
126        // and an additional permutation must be performed.
127        let mut current_chunk_idx = 0_usize;
128        // handle the case of an empty `bytes`
129        let last_chunk_idx = if num_field_elem == 0 {
130            current_chunk_idx
131        } else {
132            num_field_elem - 1
133        };
134        let rate_pos = bytes.chunks(BINARY_CHUNK_SIZE).fold(0, |rate_pos, chunk| {
135            // copy the chunk into the buffer
136            if current_chunk_idx != last_chunk_idx {
137                buf[..BINARY_CHUNK_SIZE].copy_from_slice(chunk);
138            } else {
139                // on the last iteration, we pad `buf` with a 1 followed by as many 0's as are
140                // needed to fill it
141                buf.fill(0);
142                buf[..chunk.len()].copy_from_slice(chunk);
143                buf[chunk.len()] = 1;
144            }
145            current_chunk_idx += 1;
146
147            // set the current rate element to the input. since we take at most 7 bytes, we are
148            // guaranteed that the inputs data will fit into a single field element.
149            state[RATE_RANGE.start + rate_pos] = Felt::new(u64::from_le_bytes(buf));
150
151            // proceed filling the range. if it's full, then we apply a permutation and reset the
152            // counter to the beginning of the range.
153            if rate_pos == RATE_WIDTH - 1 {
154                Self::apply_permutation(&mut state);
155                0
156            } else {
157                rate_pos + 1
158            }
159        });
160
161        // if we absorbed some elements but didn't apply a permutation to them (would happen when
162        // the number of elements is not a multiple of RATE_WIDTH), apply the permutation. we
163        // don't need to apply any extra padding because the first capacity element contains a
164        // flag indicating the number of field elements constituting the last block when the latter
165        // is not divisible by `RATE_WIDTH`.
166        if rate_pos != 0 {
167            state[RATE_RANGE.start + rate_pos..RATE_RANGE.end].fill(ZERO);
168            Self::apply_permutation(&mut state);
169        }
170
171        // return the digest portion of the rate as hash result.
172        Word::new(state[DIGEST_RANGE].try_into().unwrap())
173    }
174
175    /// Returns a hash of two digests. This method is intended for use in construction of
176    /// Merkle trees and verification of Merkle paths.
177    fn merge(values: &[Word; 2]) -> Word {
178        // initialize the state by copying the digest elements into the rate portion of the state
179        // (8 total elements), and set the capacity elements to 0.
180        let mut state = [ZERO; STATE_WIDTH];
181        let it = Word::words_as_elements_iter(values.iter());
182        for (i, v) in it.enumerate() {
183            state[RATE_RANGE.start + i] = *v;
184        }
185
186        // apply the permutation and return the digest portion of the state
187        Self::apply_permutation(&mut state);
188        Word::new(state[DIGEST_RANGE].try_into().unwrap())
189    }
190
191    /// Returns a hash of many digests.
192    fn merge_many(values: &[Word]) -> Word {
193        let elements = Word::words_as_elements(values);
194        Self::hash_elements(elements)
195    }
196
197    /// Returns hash(`seed` || `value`). This method is intended for use in PRNG and PoW contexts.
198    fn merge_with_int(seed: Word, value: u64) -> Word {
199        // initialize the state as follows:
200        // - seed is copied into the first 4 elements of the rate portion of the state.
201        // - if the value fits into a single field element, copy it into the fifth rate element and
202        //   set the first capacity element to 5.
203        // - if the value doesn't fit into a single field element, split it into two field elements,
204        //   copy them into rate elements 5 and 6 and set the first capacity element to 6.
205        let mut state = [ZERO; STATE_WIDTH];
206        state[RATE0_RANGE].copy_from_slice(seed.as_elements());
207        state[RATE1_RANGE.start] = Felt::new(value);
208        if value < Felt::ORDER_U64 {
209            state[CAPACITY_RANGE.start] = Felt::from_u8(5_u8);
210        } else {
211            state[RATE1_RANGE.start + 1] = Felt::new(value / Felt::ORDER_U64);
212            state[CAPACITY_RANGE.start] = Felt::from_u8(6_u8);
213        }
214
215        // apply the permutation and return the digest portion of the rate
216        Self::apply_permutation(&mut state);
217        Word::new(state[DIGEST_RANGE].try_into().unwrap())
218    }
219
220    // DOMAIN IDENTIFIER HASHING
221    // --------------------------------------------------------------------------------------------
222
223    /// Returns a hash of two digests and a domain identifier.
224    fn merge_in_domain(values: &[Word; 2], domain: Felt) -> Word {
225        // initialize the state by copying the digest elements into the rate portion of the state
226        // (8 total elements), and set the capacity elements to 0.
227        let mut state = [ZERO; STATE_WIDTH];
228        let it = Word::words_as_elements_iter(values.iter());
229        for (i, v) in it.enumerate() {
230            state[RATE_RANGE.start + i] = *v;
231        }
232
233        // set the second capacity element to the domain value. The first capacity element is used
234        // for padding purposes.
235        state[CAPACITY_RANGE.start + 1] = domain;
236
237        // apply the permutation and return the first four elements of the state
238        Self::apply_permutation(&mut state);
239        Word::new(state[DIGEST_RANGE].try_into().unwrap())
240    }
241}
