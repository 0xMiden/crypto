25/// Recomposes the quotient polynomial from its chunks evaluated at a point.
26///
27/// Given quotient chunks and their domains, this computes the Lagrange
28/// interpolation coefficients (zps) and reconstructs quotient(zeta).
29pub fn recompose_quotient_from_chunks<SC>(
30    quotient_chunks_domains: &[Domain<SC>],
31    quotient_chunks: &[Vec<SC::Challenge>],
32    zeta: SC::Challenge,
33) -> SC::Challenge
34where
35    SC: StarkGenericConfig,
36{
37    let zps = quotient_chunks_domains
38        .iter()
39        .enumerate()
40        .map(|(i, domain)| {
41            quotient_chunks_domains
42                .iter()
43                .enumerate()
44                .filter(|(j, _)| *j != i)
45                .map(|(_, other_domain)| {
46                    other_domain.vanishing_poly_at_point(zeta)
47                        * other_domain
48                            .vanishing_poly_at_point(domain.first_point())
49                            .inverse()
50                })
51                .product::<SC::Challenge>()
52        })
53        .collect_vec();
54
55    quotient_chunks
56        .iter()
57        .enumerate()
58        .map(|(ch_i, ch)| {
59            // We checked in valid_shape the length of "ch" is equal to
60            // <SC::Challenge as BasedVectorSpace<Val<SC>>>::DIMENSION. Hence
61            // the unwrap() will never panic.
62            zps[ch_i]
63                * ch.iter()
64                    .enumerate()
65                    .map(|(e_i, &c)| SC::Challenge::ith_basis_element(e_i).unwrap() * c)
66                    .sum::<SC::Challenge>()
67        })
68        .sum::<SC::Challenge>()
69}

71/// Verifies that the folded constraints match the quotient polynomial at zeta.
72///
73/// This evaluates the AIR constraints at the out-of-domain point and checks
74/// that constraints(zeta) / Z_H(zeta) = quotient(zeta).
75#[allow(clippy::too_many_arguments)]
76pub fn verify_constraints<SC, A, PcsErr>(
77    air: &A,
78    trace_local: &[SC::Challenge],
79    trace_next: &[SC::Challenge],
80    preprocessed_local: Option<&[SC::Challenge]>,
81    preprocessed_next: Option<&[SC::Challenge]>,
82    aux_local: Option<&[SC::Challenge]>,
83    aux_next: Option<&[SC::Challenge]>,
84    randomness: &[SC::Challenge],
85    aux_bus_boundary_values: &[SC::Challenge],
86    public_values: &[Val<SC>],
87    trace_domain: Domain<SC>,
88    zeta: SC::Challenge,
89    alpha: SC::Challenge,
90    quotient: SC::Challenge,
91) -> Result<(), VerificationError<PcsErr>>
92where
93    SC: StarkGenericConfig + Sync,
94    A: MidenAir<Val<SC>, SC::Challenge>,
95    Val<SC>: TwoAdicField,
96{
97    let sels = trace_domain.selectors_at_point(zeta);
98
99    // =====================================
100    // Periodic entries
101    // =====================================
102    let periodic_values: Vec<SC::Challenge> = evaluate_periodic_at_point::<Val<SC>, SC::Challenge>(
103        air.periodic_table(),
104        trace_domain,
105        zeta,
106    );
107
108    // =====================================
109    // Main trace
110    // =====================================
111    let main = VerticalPair::new(
112        RowMajorMatrixView::new_row(trace_local),
113        RowMajorMatrixView::new_row(trace_next),
114    );
115
116    // =====================================
117    // Preprocessed trace
118    // =====================================
119    let preprocessed = match (preprocessed_local, preprocessed_next) {
120        (Some(local), Some(next)) => Some(VerticalPair::new(
121            RowMajorMatrixView::new_row(local),
122            RowMajorMatrixView::new_row(next),
123        )),
124        _ => None,
125    };
126
127    // =====================================
128    // Aux trace
129    // =====================================
130    // Aux trace is committed as flattened base limbs. Recompose into EF values.
131    let aux_local_ext;
132    let aux_next_ext;
133    let aux = match (aux_local, aux_next) {
134        (Some(local), Some(next)) => {
135            aux_local_ext = verifier_row_to_ext::<Val<SC>, SC::Challenge>(local)
136                .ok_or(VerificationError::InvalidProofShape)?;
137            aux_next_ext = verifier_row_to_ext::<Val<SC>, SC::Challenge>(next)
138                .ok_or(VerificationError::InvalidProofShape)?;
139
140            VerticalPair::new(
141                RowMajorMatrixView::new_row(&aux_local_ext),
142                RowMajorMatrixView::new_row(&aux_next_ext),
143            )
144        }
145        _ => {
146            // Create an empty ViewPair with zero width
147            let empty: &[SC::Challenge] = &[];
148            VerticalPair::new(
149                RowMajorMatrixView::new_row(empty),
150                RowMajorMatrixView::new_row(empty),
151            )
152        }
153    };
154    let mut folder: VerifierConstraintFolder<'_, SC> = VerifierConstraintFolder {
155        main,
156        aux,
157        randomness,
158        aux_bus_boundary_values,
159        preprocessed,
160        public_values,
161        periodic_values: &periodic_values,
162        is_first_row: sels.is_first_row,
163        is_last_row: sels.is_last_row,
164        is_transition: sels.is_transition,
165        alpha,
166        accumulator: SC::Challenge::ZERO,
167    };
168    air.eval(&mut folder);
169    let folded_constraints = folder.accumulator;
170
171    // Check that constraints(zeta) / Z_H(zeta) = quotient(zeta)
172    if folded_constraints * sels.inv_vanishing != quotient {
173        return Err(VerificationError::OodEvaluationMismatch { index: None });
174    }
175
176    Ok(())
177}

179/// Validates and commits the preprocessed trace if present.
180/// Returns the preprocessed width and its commitment hash (available iff width > 0).
181#[allow(clippy::type_complexity)]
182fn process_preprocessed_trace<SC, A>(
183    air: &A,
184    opened_values: &crate::proof::OpenedValues<SC::Challenge>,
185    pcs: &SC::Pcs,
186    trace_domain: <SC::Pcs as Pcs<SC::Challenge, SC::Challenger>>::Domain,
187    is_zk: usize,
188) -> Result<
189    (
190        usize,
191        Option<<SC::Pcs as Pcs<SC::Challenge, SC::Challenger>>::Commitment>,
192    ),
193    VerificationError<PcsError<SC>>,
194>
195where
196    SC: StarkGenericConfig,
197    A: MidenAir<Val<SC>, SC::Challenge>,
198    Val<SC>: TwoAdicField,
199{
200    // If verifier asked for preprocessed trace, then proof should have it
201    let preprocessed = air.preprocessed_trace();
202    let preprocessed_width = preprocessed.as_ref().map(|m| m.width).unwrap_or(0);
203    let preprocessed_local_len = opened_values
204        .preprocessed_local
205        .as_ref()
206        .map_or(0, |v| v.len());
207    let preprocessed_next_len = opened_values
208        .preprocessed_next
209        .as_ref()
210        .map_or(0, |v| v.len());
211    if preprocessed_width != preprocessed_local_len || preprocessed_width != preprocessed_next_len {
212        // Verifier expects preprocessed trace while proof does not have it, or vice versa
213        return Err(VerificationError::InvalidProofShape);
214    }
215
216    if preprocessed_width > 0 {
217        assert_eq!(is_zk, 0); // TODO: preprocessed columns not supported in zk mode
218        let height = preprocessed.as_ref().unwrap().values.len() / preprocessed_width;
219        assert_eq!(
220            height,
221            trace_domain.size(),
222            "Verifier's preprocessed trace height must be equal to trace domain size"
223        );
224        let (preprocessed_commit, _) = debug_span!("process preprocessed trace")
225            .in_scope(|| pcs.commit([(trace_domain, preprocessed.unwrap())]));
226        Ok((preprocessed_width, Some(preprocessed_commit)))
227    } else {
228        Ok((preprocessed_width, None))
229    }
230}

232#[instrument(skip_all)]
233pub fn verify<SC, A>(
234    config: &SC,
235    air: &A,
236    proof: &Proof<SC>,
237    public_values: &[Val<SC>],
238    var_length_public_inputs: &[&[&[Val<SC>]]],
239) -> Result<(), VerificationError<PcsError<SC>>>
240where
241    SC: StarkGenericConfig + Sync,
242    A: MidenAir<Val<SC>, SC::Challenge>,
243    Val<SC>: TwoAdicField,
244{
245    let air = &AirWithBoundaryConstraints {
246        inner: air,
247        phantom: PhantomData::<SC>,
248    };
249
250    let Proof {
251        commitments,
252        opened_values,
253        opening_proof,
254        aux_finals,
255        degree_bits,
256    } = proof;
257
258    let pcs = config.pcs();
259    let degree = 1 << degree_bits;
260    let aux_width = air.aux_width();
261    let num_randomness = air.num_randomness();
262
263    let trace_domain = pcs.natural_domain_for_degree(degree);
264    // TODO: allow moving preprocessed commitment to preprocess time, if known in advance
265    let (preprocessed_width, preprocessed_commit) =
266        process_preprocessed_trace::<SC, _>(air, opened_values, pcs, trace_domain, config.is_zk())?;
267
268    let log_quotient_degree = get_log_quotient_degree::<Val<SC>, SC::Challenge, _>(
269        air,
270        preprocessed_width,
271        public_values.len(),
272        config.is_zk(),
273        aux_width,
274        num_randomness,
275    );
276    let quotient_degree = 1 << (log_quotient_degree + config.is_zk());
277    let mut challenger = config.initialise_challenger();
278    let init_trace_domain = pcs.natural_domain_for_degree(degree >> (config.is_zk()));
279
280    let quotient_domain =
281        trace_domain.create_disjoint_domain(1 << (degree_bits + log_quotient_degree));
282    let quotient_chunks_domains = quotient_domain.split_domains(quotient_degree);
283
284    let randomized_quotient_chunks_domains = quotient_chunks_domains
285        .iter()
286        .map(|domain| pcs.natural_domain_for_degree(domain.size() << (config.is_zk())))
287        .collect_vec();
288    // Check that the random commitments are/are not present depending on the ZK setting.
289    // - If ZK is enabled, the prover should have random commitments.
290    // - If ZK is not enabled, the prover should not have random commitments.
291    if (opened_values.random.is_some() != SC::Pcs::ZK)
292        || (commitments.random.is_some() != SC::Pcs::ZK)
293    {
294        return Err(VerificationError::RandomizationError);
295    }
296
297    // Observe the instance.
298    challenger.observe(Val::<SC>::from_usize(proof.degree_bits));
299    challenger.observe(Val::<SC>::from_usize(proof.degree_bits - config.is_zk()));
300    challenger.observe(Val::<SC>::from_usize(preprocessed_width));
301    // TODO: Might be best practice to include other instance data here in the transcript, like some
302    // encoding of the AIR. This protects against transcript collisions between distinct instances.
303    // Practically speaking though, the only related known attack is from failing to include public
304    // values. It's not clear if failing to include other instance data could enable a transcript
305    // collision, since most such changes would completely change the set of satisfying witnesses.
306
307    challenger.observe(commitments.trace.clone());
308    if preprocessed_width > 0 {
309        challenger.observe(preprocessed_commit.as_ref().unwrap().clone());
310    }
311    challenger.observe_slice(public_values);
312
313    // begin processing aux trace (optional)
314    let num_randomness = air.num_randomness();
315
316    let air_width = air.width();
317    let bus_types = air.bus_types();
318    let valid_shape = opened_values.trace_local.len() == air_width
319        && opened_values.trace_next.len() == air_width
320        && opened_values.quotient_chunks.len() == quotient_degree
321        && opened_values
322            .quotient_chunks
323            .iter()
324            .all(|qc| qc.len() == SC::Challenge::DIMENSION)
325        // We've already checked that opened_values.random is present if and only if ZK is enabled.
326        && opened_values.random.as_ref().is_none_or(|r_comm| r_comm.len() == SC::Challenge::DIMENSION)
327        // Check aux trace shape
328        && if num_randomness > 0 {
329            let aux_width_base = aux_width * SC::Challenge::DIMENSION;
330            // Note: bus_types length is not matched against aux_width, to allow for more generic aux traces.
331            match (&opened_values.aux_trace_local, &opened_values.aux_trace_next) {
332                (Some(l), Some(n)) => l.len() == aux_width_base
333                    && n.len() == aux_width_base
334                    && aux_finals.len() == aux_width,
335                _ => false,
336            }
337        } else {
338            opened_values.aux_trace_local.is_none() && opened_values.aux_trace_next.is_none() && aux_finals.is_empty()
339        };
340    if !valid_shape {
341        return Err(VerificationError::InvalidProofShape);
342    }
343    let randomness = if num_randomness != 0 {
344        let randomness: Vec<SC::Challenge> = (0..num_randomness)
345            .map(|_| challenger.sample_algebra_element())
346            .collect();
347
348        if let Some(aux_commit) = &commitments.aux {
349            challenger.observe(aux_commit.clone());
350        } else {
351            return Err(VerificationError::InvalidProofShape);
352        }
353        for aux_final in aux_finals {
354            challenger.observe_algebra_element(*aux_final);
355        }
356
357        randomness
358    } else {
359        // No aux trace expected
360        if commitments.aux.is_some() {
361            return Err(VerificationError::InvalidProofShape);
362        }
363        vec![]
364    };
365
366    // Get the first Fiat Shamir challenge which will be used to combine all constraint polynomials
367    // into a single polynomial.
368    //
369    // Soundness Error: n/|EF| where n is the number of constraints.
370    let alpha = challenger.sample_algebra_element();
371
372    challenger.observe(commitments.quotient_chunks.clone());
373
374    // We've already checked that commitments.random is present if and only if ZK is enabled.
375    // Observe the random commitment if it is present.
376    if let Some(r_commit) = commitments.random.clone() {
377        challenger.observe(r_commit);
378    }
379
380    // Get an out-of-domain point to open our values at.
381    //
382    // Soundness Error: dN/|EF| where `N` is the trace length and our constraint polynomial has degree `d`.
383    let zeta = challenger.sample_algebra_element();
384    let zeta_next = init_trace_domain
385        .next_point(zeta)
386        .ok_or(VerificationError::NextPointUnavailable)?;
387
388    // We've already checked that commitments.random and opened_values.random are present if and only if ZK is enabled.
389    let mut coms_to_verify = if let Some(random_commit) = &commitments.random {
390        let random_values = opened_values
391            .random
392            .as_ref()
393            .ok_or(VerificationError::RandomizationError)?;
394        vec![(
395            random_commit.clone(),
396            vec![(trace_domain, vec![(zeta, random_values.clone())])],
397        )]
398    } else {
399        vec![]
400    };
401    coms_to_verify.extend(vec![
402        (
403            commitments.trace.clone(),
404            vec![(
405                trace_domain,
406                vec![
407                    (zeta, opened_values.trace_local.clone()),
408                    (zeta_next, opened_values.trace_next.clone()),
409                ],
410            )],
411        ),
412        (
413            commitments.quotient_chunks.clone(),
414            // Check the commitment on the randomized domains.
415            zip_eq(
416                randomized_quotient_chunks_domains.iter(),
417                &opened_values.quotient_chunks,
418                VerificationError::InvalidProofShape,
419            )?
420            .map(|(domain, values)| (*domain, vec![(zeta, values.clone())]))
421            .collect_vec(),
422        ),
423    ]);
424
425    // Add aux trace verification if present
426    if let Some(aux_commit) = &commitments.aux {
427        let aux_local = opened_values
428            .aux_trace_local
429            .as_ref()
430            .ok_or(VerificationError::InvalidProofShape)?;
431        let aux_next = opened_values
432            .aux_trace_next
433            .as_ref()
434            .ok_or(VerificationError::InvalidProofShape)?;
435        coms_to_verify.push((
436            aux_commit.clone(),
437            vec![(
438                trace_domain,
439                vec![(zeta, aux_local.clone()), (zeta_next, aux_next.clone())],
440            )],
441        ));
442    }
443
444    // Add preprocessed commitment verification if present
445    if preprocessed_width > 0 {
446        let preprocessed_local = opened_values
447            .preprocessed_local
448            .as_ref()
449            .ok_or(VerificationError::InvalidProofShape)?;
450        let preprocessed_next = opened_values
451            .preprocessed_next
452            .as_ref()
453            .ok_or(VerificationError::InvalidProofShape)?;
454
455        coms_to_verify.push((
456            preprocessed_commit.unwrap(),
457            vec![(
458                trace_domain,
459                vec![
460                    (zeta, preprocessed_local.clone()),
461                    (zeta_next, preprocessed_next.clone()),
462                ],
463            )],
464        ));
465    }
466
467    pcs.verify(coms_to_verify, opening_proof, &mut challenger)
468        .map_err(VerificationError::InvalidOpeningArgument)?;
469
470    let quotient = recompose_quotient_from_chunks::<SC>(
471        &quotient_chunks_domains,
472        &opened_values.quotient_chunks,
473        zeta,
474    );
475
476    // Verify the aux trace final values match the expected values if the aux trace contains buses (one bus per aux column)
477    // Note: if no buses are defined (bus_types.is_empty), the boundary values of the aux_trace are not checked against the provided variable-length public inputs.
478    for (idx, (bus_type, aux_final)) in bus_types.iter().zip(aux_finals).enumerate() {
479        let public_inputs_for_bus = *var_length_public_inputs
480            .get(idx)
481            .ok_or(VerificationError::InvalidProofShape)?;
482        let expected_final = match bus_type {
483            BusType::Multiset => bus_multiset_boundary_varlen::<_, SC>(
484                &randomness,
485                public_inputs_for_bus.iter().copied(),
486            ),
487            BusType::Logup => bus_logup_boundary_varlen::<_, SC>(
488                &randomness,
489                public_inputs_for_bus.iter().copied(),
490            ),
491        };
492
493        if *aux_final != expected_final {
494            return Err(VerificationError::InvalidBusBoundaryValues);
495        }
496    }
497
498    verify_constraints::<SC, _, PcsError<SC>>(
499        air,
500        &opened_values.trace_local,
501        &opened_values.trace_next,
502        opened_values.preprocessed_local.as_deref(),
503        opened_values.preprocessed_next.as_deref(),
504        opened_values.aux_trace_local.as_deref(),
505        opened_values.aux_trace_next.as_deref(),
506        &randomness,
507        aux_finals,
508        public_values,
509        init_trace_domain,
510        zeta,
511        alpha,
512        quotient,
513    )?;
514
515    Ok(())
516}

518/// Computes the final value for a multiset bus given variable-length public inputs.
519pub fn bus_multiset_boundary_varlen<
520    'a,
521    I: IntoIterator<Item = &'a [Val<SC>]>,
522    SC: StarkGenericConfig,
523>(
524    randomness: &[SC::Challenge],
525    public_inputs: I,
526) -> SC::Challenge {
527    let mut bus_p_last = SC::Challenge::ONE;
528    let rand = randomness;
529    for row in public_inputs {
530        let mut p_last = rand[0];
531        for (c, p_i) in row.iter().enumerate() {
532            p_last += SC::Challenge::from(*p_i) * rand[c + 1];
533        }
534        bus_p_last *= p_last;
535    }
536    bus_p_last
537}

539/// Computes the final value for a logup bus boundary constraint given variable-length public inputs.
540pub fn bus_logup_boundary_varlen<
541    'a,
542    I: IntoIterator<Item = &'a [Val<SC>]>,
543    SC: StarkGenericConfig,
544>(
545    randomness: &[SC::Challenge],
546    public_inputs: I,
547) -> SC::Challenge {
548    let mut bus_q_last = SC::Challenge::ZERO;
549    let rand = randomness;
550    for row in public_inputs {
551        let mut q_last = rand[0];
552        for (c, p_i) in row.iter().enumerate() {
553            let p_i = *p_i;
554            q_last += SC::Challenge::from(p_i) * rand[c + 1];
555        }
556        bus_q_last += q_last.inverse();
557    }
558    bus_q_last
559}

561#[derive(Debug)]
562pub enum VerificationError<PcsErr> {
563    InvalidProofShape,
564    /// An error occurred while verifying the claimed openings.
565    InvalidOpeningArgument(PcsErr),
566    /// Out-of-domain evaluation mismatch, i.e. `constraints(zeta)` did not match
567    /// `quotient(zeta) Z_H(zeta)`.
568    OodEvaluationMismatch {
569        index: Option<usize>,
570    },
571    /// The FRI batch randomization does not correspond to the ZK setting.
572    RandomizationError,
573    /// The domain does not support computing the next point algebraically.
574    NextPointUnavailable,
575    /// The expected bus boundary final values do not match the opened values.
576    InvalidBusBoundaryValues,
577}
