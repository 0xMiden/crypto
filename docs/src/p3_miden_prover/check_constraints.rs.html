19/// Runs constraint checks using a given AIR definition and trace matrix.
20///
21/// Iterates over every row in `main`, providing both the current and next row
22/// (with wraparound) to the AIR logic. Also injects public values into the builder
23/// for first/last row assertions.
24///
25/// # Arguments
26/// - `air`: The AIR logic to run
27/// - `main`: The trace matrix (rows of witness values)
28/// - `aux`: The aux trace matrix (if 2 phase proving)
29/// - `aux_randomness`: The randomness values that are used to generate `aux` trace
30/// - `public_values`: Public values provided to the builder
31#[cfg(debug_assertions)]
32#[instrument(name = "check constraints", skip_all)]
33pub(crate) fn check_constraints<F, EF, A>(
34    air: &A,
35    main: &RowMajorMatrix<F>,
36    aux_trace: &Option<RowMajorMatrix<F>>,
37    aux_randomness: &[EF],
38    public_values: &Vec<F>,
39) where
40    F: Field,
41    EF: ExtensionField<F> + BasedVectorSpace<F>,
42    A: MidenAir<F, EF>,
43{
44    let height = main.height();
45    let preprocessed = air.preprocessed_trace();
46
47    (0..height).for_each(|row_index| {
48        let row_index_next = (row_index + 1) % height;
49
50        // row_index < height so we can used unchecked indexing.
51        let local = unsafe { main.row_slice_unchecked(row_index) };
52        // row_index_next < height so we can used unchecked indexing.
53        let next = unsafe { main.row_slice_unchecked(row_index_next) };
54        let main = ViewPair::new(
55            RowMajorMatrixView::new_row(&*local),
56            RowMajorMatrixView::new_row(&*next),
57        );
58
59        // Keep these Vecs in the outer scope so their backing memory lives
60        // long enough for the `RowMajorMatrixView` references stored in `aux`.
61        let aux_local_ext;
62        let aux_next_ext;
63
64        #[allow(clippy::option_if_let_else)]
65        let aux = if let Some(aux_matrix) = aux_trace.as_ref() {
66            let aux_local = unsafe { aux_matrix.row_slice_unchecked(row_index) };
67            aux_local_ext = prover_row_to_ext::<F, EF>(&aux_local);
68
69            let aux_next = unsafe { aux_matrix.row_slice_unchecked(row_index_next) };
70            aux_next_ext = prover_row_to_ext::<F, EF>(&aux_next);
71
72            p3_matrix::stack::VerticalPair::new(
73                RowMajorMatrixView::new_row(&aux_local_ext),
74                RowMajorMatrixView::new_row(&aux_next_ext),
75            )
76        } else {
77            // Create an empty ViewPair with zero width
78            let empty: &[EF] = &[];
79            p3_matrix::stack::VerticalPair::new(
80                RowMajorMatrixView::new_row(empty),
81                RowMajorMatrixView::new_row(empty),
82            )
83        };
84
85        let aux_bus_boundary_values;
86        if let Some(aux_matrix) = aux_trace.as_ref() {
87            let aux_bus_boundary_values_base =
88                unsafe { aux_matrix.row_slice_unchecked(height - 1) };
89            aux_bus_boundary_values = prover_row_to_ext::<F, EF>(&aux_bus_boundary_values_base);
90        } else {
91            aux_bus_boundary_values = vec![];
92        };
93
94        let preprocessed_pair = preprocessed.as_ref().map(|preprocessed_matrix| {
95            let preprocessed_local = preprocessed_matrix
96                .values
97                .chunks(preprocessed_matrix.width)
98                .nth(row_index)
99                .unwrap();
100            let preprocessed_next = preprocessed_matrix
101                .values
102                .chunks(preprocessed_matrix.width)
103                .nth(row_index_next)
104                .unwrap();
105            ViewPair::new(
106                RowMajorMatrixView::new_row(preprocessed_local),
107                RowMajorMatrixView::new_row(preprocessed_next),
108            )
109        });
110
111        // Compute periodic values for the current row
112        let periodic_table = air.periodic_table();
113        let periodic_values: Vec<EF> = periodic_table
114            .iter()
115            .map(|col| {
116                if col.is_empty() {
117                    EF::ZERO
118                } else {
119                    // Use modulo to get the repeating value
120                    // Convert from base field to extension field using From trait
121                    EF::from(col[row_index % col.len()])
122                }
123            })
124            .collect();
125
126        let mut builder = DebugConstraintBuilder {
127            row_index,
128            main,
129            aux,
130            aux_randomness,
131            aux_bus_boundary_values: &aux_bus_boundary_values,
132            preprocessed: preprocessed_pair,
133            public_values,
134            periodic_values,
135            is_first_row: F::from_bool(row_index == 0),
136            is_last_row: F::from_bool(row_index == height - 1),
137            is_transition: F::from_bool(row_index != height - 1),
138        };
139
140        air.eval(&mut builder);
141    });
142}

144/// A builder that runs constraint assertions during testing.
145///
146/// Used in conjunction with `check_constraints` to simulate
147/// an execution trace and verify that the AIR logic enforces all constraints.
148#[cfg(debug_assertions)]
149#[derive(Debug)]
150pub struct DebugConstraintBuilder<'a, F: Field, EF: ExtensionField<F>> {
151    /// The index of the row currently being evaluated.
152    row_index: usize,
153    /// A view of the current and next row as a vertical pair.
154    main: ViewPair<'a, F>,
155    /// A view of the current and next aux row as a vertical pair.
156    aux: ViewPair<'a, EF>,
157    /// randomness that is used to compute aux trace
158    aux_randomness: &'a [EF],
159    /// Aux bus boundary values (against the last row)
160    aux_bus_boundary_values: &'a [EF],
161    /// A view of the preprocessed current and next row as a vertical pair (if present).
162    preprocessed: Option<ViewPair<'a, F>>,
163    /// The public values provided for constraint validation (e.g. inputs or outputs).
164    public_values: &'a [F],
165    /// Periodic column values (computed for the current row)
166    periodic_values: Vec<EF>,
167    /// A flag indicating whether this is the first row.
168    is_first_row: F,
169    /// A flag indicating whether this is the last row.
170    is_last_row: F,
171    /// A flag indicating whether this is a transition row (not the last row).
172    is_transition: F,
173}

175#[cfg(debug_assertions)]
176impl<'a, F, EF> MidenAirBuilder for DebugConstraintBuilder<'a, F, EF>
177where
178    F: Field,
179    EF: ExtensionField<F>,
180{
181    type F = F;
182    type Expr = F;
183    type Var = F;
184    type M = ViewPair<'a, F>;
185    type PublicVar = F;
186    type EF = EF;
187    type ExprEF = EF;
188    type VarEF = EF;
189    type MP = ViewPair<'a, EF>;
190    type RandomVar = EF;
191    type PeriodicVal = EF;
192
193    fn main(&self) -> Self::M {
194        self.main
195    }
196
197    fn is_first_row(&self) -> Self::Expr {
198        self.is_first_row
199    }
200
201    fn is_last_row(&self) -> Self::Expr {
202        self.is_last_row
203    }
204
205    fn is_transition_window(&self, size: usize) -> Self::Expr {
206        if size == 2 {
207            self.is_transition
208        } else {
209            panic!("DebugConstraintBuilder only supports transition window of size 2");
210        }
211    }
212
213    fn assert_zero<I: Into<Self::Expr>>(&mut self, x: I) {
214        let value = x.into();
215        assert!(
216            value == F::ZERO,
217            "Constraint failed at row {}: expected zero, got {:?}",
218            self.row_index,
219            value
220        );
221    }
222
223    fn public_values(&self) -> &[Self::PublicVar] {
224        self.public_values
225    }
226
227    fn preprocessed(&self) -> Self::M {
228        self.preprocessed.unwrap_or_else(|| {
229            // Return an empty ViewPair if there are no preprocessed columns
230            let empty: &[F] = &[];
231            ViewPair::new(
232                RowMajorMatrixView::new_row(empty),
233                RowMajorMatrixView::new_row(empty),
234            )
235        })
236    }
237
238    fn assert_zero_ext<I>(&mut self, x: I)
239    where
240        I: Into<Self::ExprEF>,
241    {
242        let value = x.into();
243        assert!(
244            value == EF::ZERO,
245            "Extension field constraint failed at row {}: expected zero, got {:?}",
246            self.row_index,
247            value
248        );
249    }
250
251    fn permutation(&self) -> Self::MP {
252        self.aux
253    }
254
255    fn permutation_randomness(&self) -> &[Self::RandomVar] {
256        self.aux_randomness
257    }
258
259    fn aux_bus_boundary_values(&self) -> &[Self::VarEF] {
260        self.aux_bus_boundary_values
261    }
262
263    fn periodic_evals(&self) -> &[Self::PeriodicVal] {
264        &self.periodic_values
265    }
266}

279    /// A test AIR that enforces a simple linear transition logic:
280    /// - Each cell in the next row must equal the current cell plus 1 (i.e., `next = current + 1`)
281    /// - On the last row, the current row must match the provided public values.
282    ///
283    /// This is useful for validating constraint evaluation, transition logic,
284    /// and row condition flags (first/last/transition).
285    #[derive(Debug)]
286    struct RowLogicAir {
287        with_aux: bool,
288    }
289
290    impl<F, EF> MidenAir<F, EF> for RowLogicAir
291    where
292        F: Field,
293        EF: ExtensionField<F>,
294    {
295        fn width(&self) -> usize {
296            2
297        }
298
299        fn aux_width(&self) -> usize {
300            if self.with_aux { 3 } else { 0 }
301        }
302
303        fn num_randomness(&self) -> usize {
304            if self.with_aux { 1 } else { 0 }
305        }
306
307        fn eval<AB: MidenAirBuilder<F = F>>(&self, builder: &mut AB) {
308            let main = builder.main();
309            let aux = builder.permutation();
310
311            // ======================
312            // main trace
313            // ======================
314            // | main1             | main2            |
315            // | row[i]            | perm(main1)[i]   |
316            // | row[i+1]=row[i]+1 | perm(main1)[i+1] |
317
318            let a = main.get(0, 0).unwrap();
319            let b = main.get(1, 0).unwrap();
320
321            // New logic: enforce row[i+1] = row[i] + 1, only on transitions
322            builder.when_transition().assert_eq(b, a + F::ONE);
323
324            // ======================
325            // aux trace
326            // ======================
327            if <Self as MidenAir<F, EF>>::num_randomness(self) != 0 {
328                // Note: For now this is hard coded with LogUp
329                // To show that {x_i} and {y_i} are permutations of each other
330                // We compute
331                // |    aux1           |    aux2           |   aux3                          |
332                // | t_i = 1/(r - x_i) | w_i = 1/(r - y_i) | aux3[i] = aux3[i-1] + t_i - w_i |
333                //
334                // - r is the input randomness
335                // - in practice x_i and y_i should be copied from corresponding main trace (with selectors)
336                //
337                // ZZ note:
338                // This is practically LogUp with univariate. This requires 3 extension columns = 12 base columns.
339                // It is better than checking \prod(r-xi) == \prod(r-yi) which requires 4 extension columns (the last two store the running product)
340
341                // aux row computation is correct
342                let xi = main.get(0, 0).unwrap();
343                let yi = main.get(0, 1).unwrap();
344
345                let r = builder.permutation_randomness()[0];
346
347                // current row EF elements
348                let t_i = aux.get(0, 0).unwrap();
349                let w_i = aux.get(0, 1).unwrap();
350                let s_i = aux.get(0, 2).unwrap();
351                // next row EF elements
352                let t_next = aux.get(1, 0).unwrap();
353                let w_next = aux.get(1, 1).unwrap();
354                let s_next = aux.get(1, 2).unwrap();
355
356                // t * (r - x_i) == 1  and  w * (r - y_i) == 1
357                // Convert xi and yi to ExprEF by going through the Into trait
358                let r_expr = r.into();
359                let xi_ef: AB::ExprEF = AB::ExprEF::from(xi.into());
360                let yi_ef: AB::ExprEF = AB::ExprEF::from(yi.into());
361                builder.assert_eq_ext(t_i.into() * (r_expr.clone() - xi_ef), AB::ExprEF::ONE);
362                builder.assert_eq_ext(w_i.into() * (r_expr - yi_ef), AB::ExprEF::ONE);
363
364                // transition is correct: s' = s + t' - w'
365                builder
366                    .when_transition()
367                    .assert_eq_ext(s_next.into(), s_i.into() + t_next.into() - w_next.into());
368
369                // a3[last] = Î£(t - w) == 0 if multisets match
370                builder.when_last_row().assert_zero_ext(s_i);
371            }
372
373            // ======================
374            // public input
375            // ======================
376            // Add public value equality on last row for extra coverage
377            let public_values = builder.public_values();
378            let pv0 = public_values[0];
379            let pv1 = public_values[1];
380
381            let mut when_last = builder.when_last_row();
382            when_last.assert_eq(main.get(0, 0).unwrap(), pv0);
383            when_last.assert_eq(main.get(0, 1).unwrap(), pv1);
384        }
385    }

387    // A very simple permutation
388    fn permute<F: Field>(x: &[F]) -> Vec<F> {
389        x.iter().rev().cloned().collect::<Vec<F>>()
390    }
391
392    // Generate a main trace.
393    // The first column is incremental
394    // The second column is the rev of the first column
395    fn gen_main(main_col: &[Goldilocks]) -> RowMajorMatrix<Goldilocks> {
396        let main_rev = permute(main_col);
397        let main_values = main_col
398            .iter()
399            .zip(main_rev.iter())
400            .flat_map(|(a, b)| vec![a, b])
401            .cloned()
402            .collect();
403        RowMajorMatrix::new(main_values, 2)
404    }

406    // Generate the aux trace for logup arguments (flattened for storage).
407    fn gen_aux(
408        main_col: &[Goldilocks],
409        aux_randomness: &BinomialExtensionField<Goldilocks, 2>,
410    ) -> RowMajorMatrix<Goldilocks> {
411        use p3_matrix::dense::DenseMatrix;
412        // Build a DenseMatrix main trace with width 2
413        let main_rev = permute(main_col);
414        let main_values = main_col
415            .iter()
416            .zip(main_rev.iter())
417            .flat_map(|(a, b)| vec![*a, *b])
418            .collect();
419        let main = DenseMatrix::new(main_values, 2);
420        // Use the library generator and return the flattened aux
421        super::super::generate_logup_trace::<BinomialExtensionField<Goldilocks, 2>, _>(
422            &main,
423            aux_randomness,
424        )
425    }

449    fn test_permuted_incremental_rows_with_last_row_check() {
450        let len = 100;
451
452        // Each row = previous + 1, with 4 rows total, 2 columns.
453        // Last row must match public values [4, 1]
454        // randomness = 5 + 10x + 15x^2 + 20x^3
455        // | m1 | m2 | a1      | a2      | a3 |
456        // | 1  | 4  | 1/(r-1) | 1/(r-4) | .. |
457        // | 2  | 3  | 1/(r-2) | 1/(r-3) | .. |
458        // | 3  | 2  | 1/(r-3) | 1/(r-2) | .. |
459        // | 4  | 1  | 1/(r-4) | 1/(r-1) | .. |
460        let air = RowLogicAir { with_aux: true };
461        let main_col: Vec<_> = (1..=len).map(Goldilocks::from_u64).collect();
462        let main = gen_main(&main_col);
463
464        let aux_randomness =
465            BinomialExtensionField::<Goldilocks, 2>::from_basis_coefficients_slice(
466                [Goldilocks::from_u64(1005), Goldilocks::from_u64(10010)].as_ref(),
467            )
468            .unwrap();
469
470        let aux = gen_aux(&main_col, &aux_randomness);
471
472        check_constraints::<Goldilocks, BinomialExtensionField<Goldilocks, 2>, _>(
473            &air,
474            &main,
475            &Some(aux),
476            aux_randomness.as_basis_coefficients_slice(),
477            &vec![Goldilocks::from_u64(len), Goldilocks::from_u64(1)],
