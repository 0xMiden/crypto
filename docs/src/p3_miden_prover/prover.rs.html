27/// Commits the preprocessed trace if present.
28/// Returns the commitment hash and prover data (available iff preprocessed is Some).
29#[allow(clippy::type_complexity)]
30fn commit_preprocessed_trace<SC>(
31    preprocessed: RowMajorMatrix<Val<SC>>,
32    pcs: &SC::Pcs,
33    trace_domain: <SC::Pcs as Pcs<SC::Challenge, SC::Challenger>>::Domain,
34) -> (
35    <SC::Pcs as Pcs<SC::Challenge, SC::Challenger>>::Commitment,
36    <SC::Pcs as Pcs<SC::Challenge, SC::Challenger>>::ProverData,
37)
38where
39    SC: StarkGenericConfig,
40{
41    debug_span!("commit to preprocessed trace")
42        .in_scope(|| pcs.commit([(trace_domain, preprocessed)]))
43}

45#[instrument(skip_all)]
46#[allow(clippy::multiple_bound_locations, clippy::type_repetition_in_bounds)] // cfg not supported in where clauses?
47pub fn prove<SC, A>(
48    config: &SC,
49    air: &A,
50    trace: &RowMajorMatrix<Val<SC>>,
51    public_values: &[Val<SC>],
52) -> Proof<SC>
53where
54    SC: StarkGenericConfig + Sync,
55    A: MidenAir<Val<SC>, SC::Challenge>,
56    Val<SC>: TwoAdicField,
57{
58    let air = &AirWithBoundaryConstraints {
59        inner: air,
60        phantom: PhantomData::<SC>,
61    };
62
63    // Compute the height `N = 2^n` and `log_2(height)`, `n`, of the trace.
64    let degree = trace.height();
65    let log_degree = log2_strict_usize(degree);
66    let log_ext_degree = log_degree + config.is_zk();
67
68    // Get preprocessed trace and its width for symbolic constraint evaluation
69    let preprocessed_trace = air.preprocessed_trace();
70    let preprocessed_width = preprocessed_trace.as_ref().map(|m| m.width).unwrap_or(0);
71
72    // Compute the constraint polynomials as vectors of symbolic expressions.
73    let aux_width = air.aux_width();
74    let num_randomness = air.num_randomness();
75    let symbolic_constraints = get_symbolic_constraints(
76        air,
77        preprocessed_width,
78        public_values.len(),
79        aux_width,
80        num_randomness,
81    );
82
83    // Count the number of constraints that we have.
84    let constraint_count = symbolic_constraints.len();
85
86    // Each constraint polynomial looks like `C_j(X_1, ..., X_w, Y_1, ..., Y_w, Z_1, ..., Z_j)`.
87    // When evaluated on a given row, the X_i's will be the `i`'th element of the that row, the
88    // Y_i's will be the `i`'th element of the next row and the Z_i's will be evaluations of
89    // selector polynomials on the given row index.
90    //
91    // When we convert to working with polynomials, the `X_i`'s and `Y_i`'s will be replaced by the
92    // degree `N - 1` polynomials `T_i(x)` and `T_i(hx)` respectively. The selector polynomials are
93    //  a little more complicated however.
94    //
95    // In our our case, the selector polynomials are `S_1(x) = is_first_row`, `S_2(x) = is_last_row`
96    // and `S_3(x) = is_transition`. Both `S_1(x)` and `S_2(x)` are polynomials of degree `N - 1`
97    // as they must be non zero only at a single location in the initial domain. However, `is_transition`
98    // is a polynomial of degree `1` as it simply need to be `0` on the last row.
99    //
100    // The constraint degree (`deg(C)`) is the linear factor of `N` in the constraint polynomial. In other
101    // words, it is roughly the total degree of `C` however, we treat `Z_3` as a constant term which does
102    // not contribute to the degree.
103    //
104    // E.g. `C_j = Z_1 * (X_1^3 - X_2 * X_3 * X_4)` would have degree `4`.
105    //      `C_j = Z_3 * (X_1^3 - X_2 * X_3 * X_4)` would have degree `3`.
106    //
107    // The point of all this is that, defining:
108    //          C(x) = C(T_1(x), ..., T_w(x), T_1(hx), ... T_w(hx), S_1(x), S_2(x), S_3(x))
109    // We get the constraint bound:
110    //          deg(C(x)) <= deg(C) * (N - 1) + 1
111    // The `+1` is due to the `is_transition` selector which is not accounted for in `deg(C)`. Note
112    // that S_i^2 should never appear in a constraint as it should just be replaced by `S_i`.
113    //
114    // For now in comments we assume that `deg(C) = 3` meaning `deg(C(x)) <= 3N - 2`
115
116    // From the degree of the constraint polynomial, compute the number
117    // of quotient polynomials we will split Q(x) into. This is chosen to
118    // always be a power of 2.
119    let log_quotient_degree = get_log_quotient_degree::<Val<SC>, SC::Challenge, _>(
120        air,
121        preprocessed_width,
122        public_values.len(),
123        config.is_zk(),
124        aux_width,
125        num_randomness,
126    );
127    let quotient_degree = 1 << (log_quotient_degree + config.is_zk());
128
129    // Initialize the PCS and the Challenger.
130    let pcs = config.pcs();
131    let mut challenger = config.initialise_challenger();
132
133    // Get the subgroup `H` of size `N`. We treat each column `T_i` of
134    // the trace as an evaluation vector of polynomials `T_i(x)` over `H`.
135    // (In the Circle STARK case `H` is instead a standard position twin coset of size `N`)
136    let trace_domain = pcs.natural_domain_for_degree(degree);
137
138    // When ZK is enabled, we need to use an extended domain of size `2N` as we will
139    // add random values to the trace.
140    let ext_trace_domain = pcs.natural_domain_for_degree(degree * (config.is_zk() + 1));
141
142    // Let `g` denote a generator of the multiplicative group of `F` and `H'` the unique
143    // subgroup of `F` of size `N << (pcs.config.log_blowup + config.is_zk())`.
144    // If `zk` is enabled, we double the trace length by adding random values.
145    //
146    // For each trace column `T_i`, we compute the evaluation vector of `T_i(x)` over `H'`. This
147    // new extended trace `ET` is hashed into a Merkle tree with its rows bit-reversed.
148    //      trace_commit contains the root of the tree
149    //      trace_data contains the entire tree.
150    //          - trace_data.leaves is the matrix containing `ET`.
151    // Note: commit() automatically uses the optimized single-matrix path when given a single matrix
152    let (trace_commit, trace_data) = info_span!("commit to trace data")
153        .in_scope(|| pcs.commit([(ext_trace_domain, trace.clone())]));
154
155    let (preprocessed_commit, preprocessed_data) = preprocessed_trace.map_or_else(
156        || (None, None),
157        |preprocessed| {
158            let (commit, data) =
159                commit_preprocessed_trace::<SC>(preprocessed, pcs, ext_trace_domain);
160            #[cfg(debug_assertions)]
161            assert_eq!(config.is_zk(), 0); // TODO: preprocessed columns not supported in zk mode
162            (Some(commit), Some(data))
163        },
164    );
165
166    // Observe the instance.
167    // degree < 2^255 so we can safely cast log_degree to a u8.
168    challenger.observe(Val::<SC>::from_u8(log_ext_degree as u8));
169    challenger.observe(Val::<SC>::from_u8(log_degree as u8));
170    challenger.observe(Val::<SC>::from_usize(preprocessed_width));
171    // TODO: Might be best practice to include other instance data here; see verifier comment.
172
173    // Observe the Merkle root of the trace commitment.
174    challenger.observe(trace_commit.clone());
175    if preprocessed_width > 0 {
176        challenger.observe(preprocessed_commit.as_ref().unwrap().clone());
177    }
178
179    // Observe the public input values.
180    challenger.observe_slice(public_values);
181
182    // begin aux trace generation (optional)
183    let num_randomness = air.num_randomness();
184
185    let (aux_trace_commit_opt, _aux_trace_opt, aux_trace_data_opt, randomness, aux_finals) =
186        if num_randomness > 0 {
187            let randomness: Vec<SC::Challenge> = (0..num_randomness)
188                .map(|_| challenger.sample_algebra_element())
189                .collect();
190
191            // Ask config (VM) to build the aux trace if available.
192            let aux_trace_opt = air.build_aux_trace(trace, &randomness);
193
194            // At the moment, it panics if the aux trace is not available.
195            // In a future PR, we will introduce LogUp based permutation as a fall back if aux trace is not available.
196            let aux_trace = aux_trace_opt
197                .expect("aux_challenges > 0 but no aux trace was provided or generated");
198
199            let aux_finals_base = aux_trace
200                .last_row()
201                .expect("aux_challenges > 0 but aux trace was empty")
202                .into_iter()
203                .collect_vec();
204            let aux_finals = prover_row_to_ext(&aux_finals_base);
205
206            let (aux_trace_commit, aux_trace_data) = info_span!("commit to aux trace data")
207                .in_scope(|| pcs.commit([(ext_trace_domain, aux_trace.clone().flatten_to_base())]));
208
209            challenger.observe(aux_trace_commit.clone());
210            for aux_final in &aux_finals {
211                challenger.observe_algebra_element(*aux_final);
212            }
213
214            (
215                Some(aux_trace_commit),
216                Some(aux_trace),
217                Some(aux_trace_data),
218                randomness,
219                aux_finals,
220            )
221        } else {
222            (None, None, None, vec![], vec![])
223        };
224
225    #[cfg(debug_assertions)]
226    crate::check_constraints::<Val<SC>, SC::Challenge, _>(
227        air,
228        trace,
229        &_aux_trace_opt,
230        &randomness,
231        &public_values.to_vec(),
232    );
233
234    // Get the first Fiat Shamir challenge which will be used to combine all constraint polynomials
235    // into a single polynomial.
236    //
237    // Soundness Error:
238    // If a prover is malicious, we can find a row `i` such that some of the constraints
239    // C_0, ..., C_n are non 0 on this row. The malicious prover "wins" if the random challenge
240    // alpha is such that:
241    // (1): C_0(i) + alpha * C_1(i) + ... + alpha^n * C_n(i) = 0
242    // This is a polynomial of degree n, so it has at most n roots. Thus the probability of this
243    // occurring for a given trace and set of constraints is n/|EF|.
244    //
245    // Currently, we do not observe data about the constraint polynomials directly. In particular
246    // a prover could take a trace and fiddle around with the AIR it claims to satisfy without
247    // changing this sample alpha.
248    //
249    // In particular this means that a malicious prover could create a custom AIR for a given trace
250    // such that equation (1) holds. However, such AIRs would need to be very specific and
251    // so such tampering should be obvious to spot. The verifier needs to check the AIR anyway to
252    // confirm that satisfying it indeed proves what the prover claims. Hence this should not be
253    // a soundness issue.
254    let alpha: SC::Challenge = challenger.sample_algebra_element();
255
256    // A domain large enough to uniquely identify the quotient polynomial.
257    // This domain must be contained in the domain over which `trace_data` is defined.
258    // Explicitly it should be equal to `gK` for some subgroup `K` contained in `H'`.
259    let quotient_domain =
260        ext_trace_domain.create_disjoint_domain(1 << (log_ext_degree + log_quotient_degree));
261
262    // Return a the subset of the extended trace `ET` corresponding to the rows giving evaluations
263    // over the quotient domain.
264    //
265    // This only works if the trace domain is `gH'` and the quotient domain is `gK` for some subgroup `K` contained in `H'`.
266    // TODO: Make this explicit in `get_evaluations_on_domain` or otherwise fix this.
267    let trace_on_quotient_domain = pcs.get_evaluations_on_domain(&trace_data, 0, quotient_domain);
268    let aux_trace_on_quotient_domain = aux_trace_data_opt
269        .as_ref()
270        .map(|data| pcs.get_evaluations_on_domain(data, 0, quotient_domain));
271
272    let preprocessed_on_quotient_domain = preprocessed_data
273        .as_ref()
274        .map(|data| pcs.get_evaluations_on_domain(data, 0, quotient_domain));
275
276    // Compute the quotient polynomial `Q(x)` by evaluating
277    //          `C(T_1(x), ..., T_w(x), T_1(hx), ..., T_w(hx), selectors(x)) / Z_H(x)`
278    // at every point in the quotient domain. The degree of `Q(x)` is `<= deg(C(x)) - N = 2N - 2` in the case
279    // where `deg(C) = 3`. (See the discussion above constraint_degree for more details.)
280    let quotient_values: Vec<SC::Challenge> = quotient_values::<SC, _, _>(
281        air,
282        public_values,
283        trace_domain,
284        quotient_domain,
285        &trace_on_quotient_domain,
286        aux_trace_on_quotient_domain.as_ref(),
287        &randomness,
288        &aux_finals,
289        preprocessed_on_quotient_domain.as_ref(),
290        alpha,
291        constraint_count,
292    );
293
294    // Due to `alpha`, evaluations of `Q` all lie in the extension field `E`.
295    // We flatten this into a matrix of `F` values by treating `E` as an `F`
296    // vector space and so separating each element of `E` into `e + 1 = [E: F]` elements of `F`.
297    //
298    // This is valid to do because our domain lies in the base field `F`. Hence we can split
299    // `Q(x)` into `e + 1` polynomials `Q_0(x), ... , Q_e(x)` each contained in `F`.
300    // such that `Q(x) = [Q_0(x), ... ,Q_e(x)]` holds for all `x` in `F`.
301    let quotient_flat = RowMajorMatrix::new_col(quotient_values).flatten_to_base();
302
303    // Currently each polynomial `Q_i(x)` is of degree `<= 2(N - 1)` and
304    // we have it's evaluations over a the coset `gK of size `2N`. Let `k` be the chosen
305    // generator of `K` which satisfies `k^2 = h`.
306    //
307    // We can split this coset into the sub-cosets `gH` and `gkH` each of size `N`.
308    // Define:  L_g(x)    = (x^N - (gk)^N)/(g^N - (gk)^N) = (x^N + g^N)/2g^N
309    //          L_{gk}(x) = (x^N - g^N)/(g^N - (gk)^N)    = -(x^N - g^N)/2g^N.
310    // Then `L_g` is equal to `1` on `gH` and `0` on `gkH` and `L_{gk}` is equal to `1` on `gkH` and `0` on `gH`.
311    //
312    // Thus we can decompose `Q_i(x) = L_{g}(x)q_{i0}(x) + L_{gk}(x)q_{i1}(x)` (Or an randomized version of this in the zk case)
313    // where `q_{i0}(x)` and `q_{i1}(x)` are polynomials of degree `<= N - 1`.
314    // Moreover the evaluations of `q_{i0}(x), q_{i1}(x)` on `gH` and `gkH` respectively are
315    // exactly the evaluations of `Q_i(x)` on `gH` and `gkH`.
316    // For each polynomial `q_{ij}`, compute the evaluation vector of `q_{ij}(x)` over `gH'`. We bit
317    // reverse the rows and hash the resulting matrix into a merkle tree.
318    //      quotient_commit contains the root of the tree
319    //      quotient_data contains the entire tree.
320    //          - quotient_data.leaves is a pair of matrices containing the `q_i0(x)` and `q_i1(x)`.
321    let (quotient_commit, quotient_data) = info_span!("commit to quotient poly chunks")
322        .in_scope(|| pcs.commit_quotient(quotient_domain, quotient_flat, quotient_degree));
323    challenger.observe(quotient_commit.clone());
324
325    // If zk is enabled, we generate random extension field values of the size of the randomized trace. If `n` is the degree of the initial trace,
326    // then the randomized trace has degree `2n`. To randomize the FRI batch polynomial, we then need an extension field random polynomial of degree `2n -1`.
327    // So we can generate a random polynomial  of degree `2n`, and provide it to `open` as is.
328    // Then the method will add `(R(X) - R(z)) / (X - z)` (which is of the desired degree `2n - 1`), to the batch of polynomials.
329    // Since we need a random polynomial defined over the extension field, and the `commit` method is over the base field,
330    // we actually need to commit to `SC::CHallenge::D` base field random polynomials.
331    // This is similar to what is done for the quotient polynomials.
332    // TODO: This approach is only statistically zk. To make it perfectly zk, `R` would have to truly be an extension field polynomial.
333    let (opt_r_commit, opt_r_data) = if SC::Pcs::ZK {
334        let (r_commit, r_data) = pcs
335            .get_opt_randomization_poly_commitment([ext_trace_domain])
336            .expect("ZK is enabled, so we should have randomization commitments");
337        (Some(r_commit), Some(r_data))
338    } else {
339        (None, None)
340    };
341
342    // Combine our commitments to the trace and quotient polynomials into a single object which
343    // will be passed to the verifier.
344    let commitments = Commitments {
345        trace: trace_commit,
346        aux: aux_trace_commit_opt,
347        quotient_chunks: quotient_commit,
348        random: opt_r_commit.clone(),
349    };
350
351    if let Some(r_commit) = opt_r_commit {
352        challenger.observe(r_commit);
353    }
354
355    // Get an out-of-domain point to open our values at.
356    //
357    // Soundness Error:
358    // This sample will be used to check the equality: `C(X) = ZH(X)Q(X)`. If a prover is malicious
359    // and this equality is false, the probability that it is true at the point `zeta` will be
360    // deg(C(X))/|EF| = dN/|EF| where `N` is the trace length and our constraints have degree `d`.
361    //
362    // Completeness Error:
363    // If zeta happens to lie in the domain `gK`, then when opening at zeta we will run into division
364    // by zero errors. This doesn't lead to a soundness issue as the verifier will just reject in those
365    // cases but it is a completeness issue and contributes a completeness error of |gK| = 2N/|EF|.
366    let zeta: SC::Challenge = challenger.sample_algebra_element();
367    let zeta_next = trace_domain
368        .next_point(zeta)
369        .expect("domain should support next_point operation");
370
371    let is_random = opt_r_data.is_some();
372    let (opened_values, opening_proof) = info_span!("open").in_scope(|| {
373        let mut rounds = vec![];
374        if let Some(r_data) = opt_r_data.as_ref() {
375            rounds.push((r_data, vec![vec![zeta]]));
376        }
377        rounds.push((&trace_data, vec![vec![zeta, zeta_next]]));
378        rounds.push((&quotient_data, vec![vec![zeta]; quotient_degree])); // open every chunk at zeta
379        if let Some(aux_data) = aux_trace_data_opt.as_ref() {
380            rounds.push((aux_data, vec![vec![zeta, zeta_next]]));
381        }
382        if let Some(preprocessed_data) = preprocessed_data.as_ref() {
383            rounds.push((preprocessed_data, vec![vec![zeta, zeta_next]]));
384        }
385
386        pcs.open(rounds, &mut challenger)
387    });
388
389    let random = if is_random {
390        Some(opened_values[0][0][0].clone())
391    } else {
392        None
393    };
394
395    let mut cur_index = SC::Pcs::TRACE_IDX;
396    let trace_local = opened_values[cur_index][0][0].clone();
397    let trace_next = opened_values[cur_index][0][1].clone();
398
399    cur_index = SC::Pcs::QUOTIENT_IDX;
400    let quotient_chunks = opened_values[cur_index]
401        .iter()
402        .map(|v| v[0].clone())
403        .collect_vec();
404    cur_index += 1;
405
406    let (aux_trace_local, aux_trace_next) = if aux_trace_data_opt.is_some() {
407        let aux_local = opened_values[cur_index][0][0].clone();
408        let aux_next = opened_values[cur_index][0][1].clone();
409        cur_index += 1;
410        (Some(aux_local), Some(aux_next))
411    } else {
412        (None, None)
413    };
414
415    let (preprocessed_local, preprocessed_next) = if preprocessed_width > 0 {
416        (
417            Some(opened_values[cur_index][0][0].clone()),
418            Some(opened_values[cur_index][0][1].clone()),
419        )
420    } else {
421        (None, None)
422    };
423
424    let opened_values = OpenedValues {
425        trace_local,
426        trace_next,
427        aux_trace_local,
428        aux_trace_next,
429        preprocessed_local,
430        preprocessed_next,
431        quotient_chunks,
432        random,
433    };
434    Proof {
435        commitments,
436        opened_values,
437        opening_proof,
438        aux_finals,
439        degree_bits: log_ext_degree,
440    }
441}

443// TODO: Group some arguments to remove the `allow`?
444#[instrument(name = "compute quotient polynomial", skip_all)]
445#[allow(clippy::too_many_arguments)]
446pub fn quotient_values<SC, A, Mat>(
447    air: &A,
448    public_values: &[Val<SC>],
449    trace_domain: Domain<SC>,
450    quotient_domain: Domain<SC>,
451    trace_on_quotient_domain: &Mat,
452    aux_trace_on_quotient_domain: Option<&Mat>,
453    randomness: &[SC::Challenge],
454    aux_bus_boundary_values: &[SC::Challenge],
455    preprocessed_on_quotient_domain: Option<&Mat>,
456    alpha: SC::Challenge,
457    constraint_count: usize,
458) -> Vec<SC::Challenge>
459where
460    SC: StarkGenericConfig + Sync,
461    A: MidenAir<Val<SC>, SC::Challenge>,
462    Mat: Matrix<Val<SC>> + Sync,
463    Val<SC>: TwoAdicField,
464{
465    let quotient_size = quotient_domain.size();
466    let width = trace_on_quotient_domain.width();
467    let mut sels = debug_span!("Compute Selectors")
468        .in_scope(|| trace_domain.selectors_on_coset(quotient_domain));
469
470    let qdb = log2_strict_usize(quotient_domain.size()) - log2_strict_usize(trace_domain.size());
471    let next_step = 1 << qdb;
472
473    // =====================================
474    // Periodic entries section
475    // =====================================
476    // Get periodic table from AIR. Periodic columns are derived solely from
477    // `periodic_table()` (never committed) and behave like degree-0 constants
478    // shared by prover and verifier.
479    let periodic_table = air.periodic_table();
480
481    // Precompute the quotient-domain points for easy lookups.
482    let quotient_points: Vec<SC::Challenge> = {
483        let mut pts = Vec::with_capacity(quotient_size);
484        let mut point = SC::Challenge::from(quotient_domain.first_point());
485        pts.push(point);
486        for _ in 1..quotient_size {
487            point = quotient_domain
488                .next_point(point)
489                .expect("quotient_domain should yield enough points");
490            pts.push(point);
491        }
492        pts
493    };
494
495    let periodic_on_quotient = compute_periodic_on_quotient_eval_domain::<Val<SC>, SC::Challenge>(
496        periodic_table,
497        trace_domain,
498        &quotient_points,
499    );
500
501    // =====================================
502    // normal eval section
503    // =====================================
504    // We take PackedVal::<SC>::WIDTH worth of values at a time from a quotient_size slice, so we need to
505    // pad with default values in the case where quotient_size is smaller than PackedVal::<SC>::WIDTH.
506    for _ in quotient_size..PackedVal::<SC>::WIDTH {
507        sels.is_first_row.push(Val::<SC>::default());
508        sels.is_last_row.push(Val::<SC>::default());
509        sels.is_transition.push(Val::<SC>::default());
510        sels.inv_vanishing.push(Val::<SC>::default());
511    }
512
513    let mut alpha_powers = alpha.powers().collect_n(constraint_count);
514    alpha_powers.reverse();
515    // alpha powers looks like Vec<EF> ~ Vec<[F; D]>
516    // It's useful to also have access to the transpose of this of form [Vec<F>; D].
517    let decomposed_alpha_powers: Vec<_> = (0..SC::Challenge::DIMENSION)
518        .map(|i| {
519            alpha_powers
520                .iter()
521                .map(|x| x.as_basis_coefficients_slice()[i])
522                .collect()
523        })
524        .collect();
525
526    // Pack aux bus boundary values
527    let packed_aux_bus_boundary_values: Vec<PackedChallenge<SC>> = aux_bus_boundary_values
528        .iter()
529        .copied()
530        .map(Into::into)
531        .collect();
532
533    (0..quotient_size)
534        .into_par_iter()
535        .step_by(PackedVal::<SC>::WIDTH)
536        .flat_map_iter(|i_start| {
537            let i_range = i_start..i_start + PackedVal::<SC>::WIDTH;
538
539            let is_first_row = *PackedVal::<SC>::from_slice(&sels.is_first_row[i_range.clone()]);
540            let is_last_row = *PackedVal::<SC>::from_slice(&sels.is_last_row[i_range.clone()]);
541            let is_transition = *PackedVal::<SC>::from_slice(&sels.is_transition[i_range.clone()]);
542            let inv_vanishing = *PackedVal::<SC>::from_slice(&sels.inv_vanishing[i_range.clone()]);
543
544            let main = RowMajorMatrix::new(
545                trace_on_quotient_domain
546                    .vertically_packed_row_pair::<PackedVal<SC>>(i_start, next_step),
547                width,
548            );
549            let aux = aux_trace_on_quotient_domain.as_ref().map_or_else(
550                || RowMajorMatrix::new(vec![], 0),
551                |aux_trace| {
552                    // Aux trace is stored in flattened base field format (each EF element = D base elements)
553                    // We need to convert it to packed extension field format
554                    let d = <SC::Challenge as BasedVectorSpace<Val<SC>>>::DIMENSION;
555                    let base_packed: Vec<PackedVal<SC>> =
556                        aux_trace.vertically_packed_row_pair(i_start, next_step);
557                    let ef_width = aux_trace.width() / d;
558
559                    // Convert from packed base field to packed extension field
560                    // Each EF element is formed from D consecutive base field elements
561                    let ef_packed: Vec<PackedChallenge<SC>> = (0..ef_width * 2)
562                        .map(|i| {
563                            PackedChallenge::<SC>::from_basis_coefficients_fn(|j| {
564                                base_packed[i * d + j]
565                            })
566                        })
567                        .collect();
568
569                    RowMajorMatrix::new(ef_packed, ef_width)
570                },
571            );
572
573            let preprocessed = preprocessed_on_quotient_domain.map(|preprocessed| {
574                let preprocessed_width = preprocessed.width();
575                RowMajorMatrix::new(
576                    preprocessed.vertically_packed_row_pair::<PackedVal<SC>>(i_start, next_step),
577                    preprocessed_width,
578                )
579            });
580
581            // Pack challenges for constraint evaluation
582            let packed_randomness: Vec<PackedChallenge<SC>> =
583                randomness.iter().copied().map(Into::into).collect();
584
585            // Grab precomputed periodic evaluations for this packed chunk.
586            let periodic_values: Vec<PackedChallenge<SC>> = periodic_on_quotient
587                .as_ref()
588                .map(|cols| {
589                    cols.iter()
590                        .map(|values| {
591                            let slice = &values[i_range.clone()];
592                            PackedChallenge::<SC>::from_ext_slice(slice)
593                        })
594                        .collect()
595                })
596                .unwrap_or_default();
597
598            let accumulator = PackedChallenge::<SC>::ZERO;
599            let mut folder: ProverConstraintFolder<'_, SC> = ProverConstraintFolder {
600                main: main.as_view(),
601                aux: aux.as_view(),
602                preprocessed: preprocessed.as_ref().map(|m| m.as_view()),
603                public_values,
604                periodic_values: &periodic_values,
605                is_first_row,
606                is_last_row,
607                is_transition,
608                alpha_powers: &alpha_powers,
609                decomposed_alpha_powers: &decomposed_alpha_powers,
610                accumulator,
611                constraint_index: 0,
612                packed_randomness,
613                aux_bus_boundary_values: &packed_aux_bus_boundary_values,
614            };
615
616            air.eval(&mut folder);
617
618            // quotient(x) = constraints(x) / Z_H(x)
619            let quotient = folder.accumulator * inv_vanishing;
620
621            // "Transpose" D packed base coefficients into WIDTH scalar extension coefficients.
622            (0..core::cmp::min(quotient_size, PackedVal::<SC>::WIDTH)).map(move |idx_in_packing| {
623                SC::Challenge::from_basis_coefficients_fn(|coeff_idx| {
624                    quotient.as_basis_coefficients_slice()[coeff_idx].as_slice()[idx_in_packing]
625                })
626            })
627        })
628        .collect()
629}
