19/// Create a proof that an opening `f(zeta)` is correct by proving that the
20/// function `(f(x) - f(zeta))/(x - zeta)` is low degree.
21///
22/// This further supports proving a batch of these claims for a collection of polynomials of shrinking degrees.
23/// Polynomials of equal degree can be combined using randomness before calling this function.
24///
25/// The Soundness error from prove_fri comes from the paper:
26/// Proximity Gaps for Reed-Solomon Codes (<https://eprint.iacr.org/2020/654>)
27/// and is either `rate^{num_queries}` or `rate^{num_queries/2}` depending on if you rely on conjectured or
28/// proven soundness. Particularly safety conscious users may want to set `num_queries` slightly higher than
29/// this to account for the fact that most implementations batch inputs using a single random challenge
30/// instead of one challenge for each polynomial and due to the birthday paradox,
31/// there is a non trivial chance that two queried indices will be equal.
32///
33/// Arguments:
34/// - `folding`: The FRI folding scheme to use.
35/// - `params`: The parameters for the specific FRI protocol instance.
36/// - `inputs`: The evaluation vectors of all polynomials we are applying FRI to. The function assumes that
37///   commitments to these vectors have been produced and observed by the challenger earlier in the protocol.
38/// - `challenger`: The Fiat-Shamir challenger to use for sampling challenges.
39/// - `log_global_max_height`: The log of the maximum height of the input matrices.
40/// - `prover_data_with_opening_points`: A list of pairs of a batch commitment to a collection
41///   of matrices and a list of points to open those matrices at.
42#[instrument(name = "FRI prover", skip_all)]
43pub fn prove_fri<Folding, Val, Challenge, InputMmcs, FriMmcs, Challenger>(
44    folding: &Folding,
45    params: &FriParameters<FriMmcs>,
46    inputs: Vec<Vec<Challenge>>,
47    challenger: &mut Challenger,
48    log_global_max_height: usize,
49    prover_data_with_opening_points: &[ProverDataWithOpeningPoints<
50        '_,
51        Challenge,
52        InputMmcs::ProverData<RowMajorMatrix<Val>>,
53    >],
54    input_mmcs: &InputMmcs,
55) -> FriProof<Challenge, FriMmcs, Challenger::Witness, Folding::InputProof>
56where
57    Val: TwoAdicField,
58    Challenge: ExtensionField<Val>,
59    InputMmcs: Mmcs<Val>,
60    FriMmcs: Mmcs<Challenge>,
61    Challenger: FieldChallenger<Val> + GrindingChallenger + CanObserve<FriMmcs::Commitment>,
62    Folding: FriFoldingStrategy<Val, Challenge, InputProof = Vec<BatchOpening<Val, InputMmcs>>>,
63{
64    assert!(!inputs.is_empty());
65    assert!(
66        inputs
67            .iter()
68            .tuple_windows()
69            .all(|(l, r)| l.len() >= r.len()),
70        "Inputs are not sorted in descending order of length."
71    );
72
73    let log_max_height = log2_strict_usize(inputs[0].len());
74    let log_min_height = log2_strict_usize(inputs.last().unwrap().len());
75    if params.log_final_poly_len > 0 {
76        // Final_poly_degree must be less than or equal to the degree of the smallest polynomial.
77        assert!(log_min_height > params.log_final_poly_len + params.log_blowup);
78    }
79
80    // Continually fold the inputs down until the polynomial degree reaches final_poly_degree.
81    // Returns a vector of commitments to the intermediate stage polynomials, the intermediate stage polynomials
82    // themselves and the final polynomial.
83    // Note that the challenger observes the commitments and the final polynomial inside this function so we don't
84    // need to observe the output of this function here.
85    let commit_phase_result = commit_phase(folding, params, inputs, challenger);
86
87    // Produce a proof of work witness before receiving any query challenges.
88    // This helps to prevent grinding attacks.
89    let pow_witness = challenger.grind(params.proof_of_work_bits);
90
91    let query_proofs = info_span!("query phase").in_scope(|| {
92        // Sample num_queries indexes to check.
93        // The probability that no two FRI indices are equal (ignoring extra query index bits) is:
94        // (Grabbed this from wikipedia page on the birthday problem)
95        // N!/(N^{num_queries} * (N - num_queries)!) ~ (1 - 1/N)^{num_queries * (num_queries - 1)/2}
96        //                                           ~ (1 - num_queries^2/2N)
97        // Here N = 2^log_max_height.
98        // With num_queries = 100, N = 2^20, this is 0.995 so there is a .5% chance of a collision.
99        // Due to this, security conscious users may want to set num_queries a little higher than the
100        // theoretical minimum.
101        iter::repeat_with(|| {
102            let index = challenger.sample_bits(log_max_height + folding.extra_query_index_bits());
103            // For each index, create a proof that the folding operations along the chain:
104            // round 0: index, round 1: index >> 1, round 2: index >> 2, ... are correct.
105            QueryProof {
106                input_proof: open_input(
107                    log_global_max_height,
108                    index,
109                    prover_data_with_opening_points,
110                    input_mmcs,
111                ),
112                commit_phase_openings: answer_query(
113                    params,
114                    &commit_phase_result.data,
115                    index >> folding.extra_query_index_bits(),
116                ),
117            }
118        })
119        .take(params.num_queries)
120        .collect()
121    });
122
123    FriProof {
124        commit_phase_commits: commit_phase_result.commits,
125        query_proofs,
126        final_poly: commit_phase_result.final_poly,
127        pow_witness,
128    }
129}

131struct CommitPhaseResult<F: Field, M: Mmcs<F>> {
132    commits: Vec<M::Commitment>,
133    data: Vec<M::ProverData<RowMajorMatrix<F>>>,
134    final_poly: Vec<F>,
135}

137/// Perform the commit phase of the FRI protocol.
138///
139/// In each round we reduce our evaluations over `H` to evaluations over `H^2` by defining
140/// ```text
141///     f_{i + 1}(x^2) = (f_i(x) + f_i(-x))/2 + beta_i (f_i(x) - f_i(-x))/2x
142/// ```
143/// We then commit to the evaluation vector of `f_{i + 1}` over `H^2`.
144///
145/// Once the degree of our polynomial falls below `final_poly_degree`, we compute the coefficients of our
146/// polynomial and return them along with all intermediate evaluations and corresponding commitments.
147///
148/// Arguments:
149/// - `folding`: The FRI folding scheme used by the prover.
150/// - `params`: The parameters for the specific FRI protocol instance.
151/// - `inputs`: The evaluation vectors of the polynomials. These must be sorted in descending order of length and each
152///   evaluation vector must be in bit reversed order. This function assumes that commitments to these vectors
153///   have already been produced and observed by the challenger.
154/// - `challenger`: The Fiat-Shamir challenger to use for sampling challenges.
155#[instrument(name = "commit phase", skip_all)]
156fn commit_phase<Folding, Val, Challenge, M, Challenger>(
157    folding: &Folding,
158    params: &FriParameters<M>,
159    inputs: Vec<Vec<Challenge>>,
160    challenger: &mut Challenger,
161) -> CommitPhaseResult<Challenge, M>
162where
163    Val: TwoAdicField,
164    Challenge: ExtensionField<Val>,
165    M: Mmcs<Challenge>,
166    Challenger: FieldChallenger<Val> + CanObserve<M::Commitment>,
167    Folding: FriFoldingStrategy<Val, Challenge>,
168{
169    let mut inputs_iter = inputs.into_iter().peekable();
170    let mut folded = inputs_iter.next().unwrap();
171    let mut commits = vec![];
172    let mut data = vec![];
173
174    while folded.len() > params.blowup() * params.final_poly_len() {
175        // As folded is in bit reversed order, it looks like:
176        //      `[f_i(h^0), f_i(h^{N/2}), f_i(h^{N/4}), f_i(h^{3N/4}), ...] = [f_i(1), f_i(-1), f_i(h^{N/4}), f_i(-h^{N/4}), ...]`
177        // so the relevant evaluations are adjacent and we can just reinterpret the vector as a matrix with width = folding_factor.
178        let folding_factor = 1 << folding.log_folding_factor();
179        let leaves = RowMajorMatrix::new(folded, folding_factor);
180
181        // Commit to these evaluations and observe the commitment.
182        let (commit, prover_data) = params.mmcs.commit_matrix(leaves);
183        challenger.observe(commit.clone());
184        commits.push(commit);
185
186        // Get the Fiat-Shamir challenge for this round.
187        let beta: Challenge = challenger.sample_algebra_element();
188
189        // We passed ownership of `leaves` to the MMCS, so get a reference to it
190        let leaves = params.mmcs.get_matrices(&prover_data).pop().unwrap();
191        // Do the folding operation:
192        // if the folding factor = 2 then
193        //      `f_{i + 1}'(x^2) = (f_i(x) + f_i(-x))/2 + beta_i (f_i(x) - f_i(-x))/2x`
194        // note the folding factor can be an arbitrary power of 2
195        folded = folding.fold_matrix(beta, leaves.as_view());
196
197        data.push(prover_data);
198
199        // If we have reached the size of the next input vector, we can add it to the current vector.
200        if let Some(v) = inputs_iter.next_if(|v| v.len() == folded.len()) {
201            // Each element of `inputs_iter` is a reduced opening polynomial, which is itself a
202            // random linear combination `f_{i, 0} + alpha f_{i, 1} + ...`, when we add it
203            // to the current folded polynomial, we need to multiply by a random factor.
204            izip!(&mut folded, v).for_each(|(c, x)| *c += beta.square() * x);
205        }
206    }
207
208    // Now we need to get the coefficients of the final polynomial. As we know that the degree
209    // is `<= params.final_poly_len()` and the evaluations are stored in bit-reversed order,
210    // we can just truncate the folded vector, bit-reverse again and run an IDFT.
211    folded.truncate(params.final_poly_len());
212    reverse_slice_index_bits(&mut folded);
213    let final_poly = debug_span!("idft final poly")
214        .in_scope(|| Radix2DFTSmallBatch::default().idft_algebra(folded));
215
216    // Observe all coefficients of the final polynomial.
217    for &x in &final_poly {
218        challenger.observe_algebra_element(x);
219    }
220
221    CommitPhaseResult {
222        commits,
223        data,
224        final_poly,
225    }
226}

228/// Given an `index` produce a proof that the chain of folds at `index, index >> log_folding_factor, ... ` are correct.
229/// This is the prover's complement to the verifier's [`verify_query`] function.
230///
231/// In addition to the output of this function, the prover must also supply the verifier with the input values
232/// (with associated opening proofs). These are produced by the `open_input` function passed into `prove_fri`.
233///
234/// For each `i` in `[0, ..., num_folds)` this returns the values of all siblings of `index >> (i * log_folding_factor)`
235/// in round `i` along with an opening proof. The verifier can then use these values along with possibly an input
236/// value to compute the value at `index >> ((i + 1) * log_folding_factor)` in round `i + 1`.
237///
238/// We repeat until we reach the final round where the verifier can check the value against the
239/// polynomial they were sent.
240///
241/// Arguments:
242/// - `params`: The parameters for the specific FRI protocol instance.
243/// - `folded_polynomial_commits`: A slice of commitments to the intermediate stage polynomials.
244/// - `start_index`: The opening index for the unfolded polynomial. For folded polynomials,
245///   we use this index right shifted by the number of folds.
246#[inline]
247fn answer_query<F, M>(
248    config: &FriParameters<M>,
249    folded_polynomial_commits: &[M::ProverData<RowMajorMatrix<F>>],
250    start_index: usize,
251) -> Vec<CommitPhaseProofStep<F, M>>
252where
253    F: Field,
254    M: Mmcs<F>,
255{
256    let log_folding_factor = config.log_folding_factor;
257    let folding_factor = config.folding_factor();
258
259    folded_polynomial_commits
260        .iter()
261        .enumerate()
262        .map(|(i, commit)| {
263            // After i folding rounds, the current index we are looking at is `index >> (i * log_folding_factor)`.
264            let index_i = start_index >> (i * log_folding_factor);
265            // Get the index within the folding group
266            let index_in_group = index_i % folding_factor;
267            // Get the index of the folding group (row in the committed matrix)
268            let group_index = index_i >> log_folding_factor;
269
270            // Get a proof that the entire row (all siblings in the folding group) is correct.
271            let (mut opened_rows, opening_proof) =
272                config.mmcs.open_batch(group_index, commit).unpack();
273
274            // opened_rows should contain just the row at group_index.
275            assert_eq!(opened_rows.len(), 1);
276            let opened_row = opened_rows.pop().unwrap();
277            assert_eq!(
278                opened_row.len(),
279                folding_factor,
280                "Committed data should have width equal to folding factor"
281            );
282
283            // We need to send all values in the row EXCEPT the one at index_in_group,
284            // since the verifier already knows that value (it's the folded evaluation from the previous round).
285            let sibling_values: Vec<F> = opened_row
286                .into_iter()
287                .enumerate()
288                .filter_map(|(idx, val)| {
289                    if idx != index_in_group {
290                        Some(val)
291                    } else {
292                        None
293                    }
294                })
295                .collect();
296
297            CommitPhaseProofStep {
298                sibling_values,
299                opening_proof,
300            }
301        })
302        .collect()
303}

305/// Given an index, produce batch opening proofs for each collection of matrices
306/// combined into a single mmcs commitment.
307///
308/// In cases where the maximum height of a batch of matrices is smaller than the
309/// global max height, shift the index down to compensate.
310///
311/// Arguments:
312/// - `log_global_max_height`: The log of the maximum height of the input matrices.
313/// - `index`: The index to open the matrices at.
314/// - `prover_data_with_opening_points`: A list of pairs of a batch commitment to a collection
315///   of matrices and a list of points to open those matrices at.
316/// - `mmcs`: The mixed matrix commitment scheme used to produce the batch commitments.
317#[inline]
318fn open_input<Val, Challenge, InputMmcs>(
319    log_global_max_height: usize,
320    index: usize,
321    prover_data_with_opening_points: &[ProverDataWithOpeningPoints<
322        '_,
323        Challenge,
324        InputMmcs::ProverData<RowMajorMatrix<Val>>,
325    >],
326    mmcs: &InputMmcs,
327) -> Vec<BatchOpening<Val, InputMmcs>>
328where
329    Val: TwoAdicField,
330    Challenge: ExtensionField<Val>,
331    InputMmcs: Mmcs<Val>,
332{
333    // This gives the verifier access to evaluations `f(x)` from which it can compute
334    // `(f(zeta) - f(x))/(zeta - x)` and then combine them together and roll into FRI
335    // as appropriate.
336    prover_data_with_opening_points
337        .iter()
338        .map(|(data, _)| {
339            let log_max_height = log2_strict_usize(mmcs.get_max_height(data));
340            let bits_reduced = log_global_max_height - log_max_height;
341            // If a matrix is smaller than global max height, we roll it into
342            // fri in a later round.
343            let reduced_index = index >> bits_reduced;
344            mmcs.open_batch(reduced_index, data)
345        })
346        .collect()
