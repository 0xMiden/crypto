1//! The FRI PCS protocol over two-adic fields.
2//!
3//! The following implements a slight variant of the usual FRI protocol. As usual we start
4//! with a polynomial `F(x)` of degree `n` given as evaluations over the coset `gH` with `|H| = 2^n`.
5//!
6//! Now consider the polynomial `G(x) = F(gx)`. Note that `G(x)` has the same degree as `F(x)` and
7//! the evaluations of `F(x)` over `gH` are identical to the evaluations of `G(x)` over `H`.
8//!
9//! Hence we can reinterpret our vector of evaluations as evaluations of `G(x)` over `H` and apply
10//! the standard FRI protocol to this evaluation vector. This makes it easier to apply FRI to a collection
11//! of polynomials defined over different cosets as we don't need to keep track of the coset shifts. We
12//! can just assume that every polynomial is defined over the subgroup of the relevant size.
13//!
14//! If we changed our domain construction (e.g., using multiple cosets), we would need to carefully reconsider these assumptions.
15

41/// A polynomial commitment scheme using FRI to generate opening proofs.
42///
43/// We commit to a polynomial `f` via its evaluation vectors over a coset
44/// `gH` where `|H| >= 2 * deg(f)`. A value `f(z)` is opened by using a FRI
45/// proof to show that the evaluations of `(f(x) - f(z))/(x - z)` over
46/// `gH` are low degree.
47#[derive(Debug)]
48pub struct TwoAdicFriPcs<Val, Dft, InputMmcs, FriMmcs> {
49    pub(crate) dft: Dft,
50    pub(crate) mmcs: InputMmcs,
51    pub(crate) fri: FriParameters<FriMmcs>,
52    _phantom: PhantomData<Val>,
53}

123        &self,

142            self.fold_matrix_arbitrary(beta, &m, folding_factor)
143        }

147impl<InputProof, InputError: Debug> TwoAdicFriFolding<InputProof, InputError> {
148    pub fn new(log_folding_factor: usize) -> Self {
149        assert!(log_folding_factor > 0);
150        Self {
151            log_folding_factor,
152            _phantom: Default::default(),
153        }
154    }

156    fn fold_row_2<EF, F>(
157        &self,
158        index: usize,
159        log_height: usize,
160        beta: EF,
161        evals: impl Iterator<Item = EF>,
162    ) -> EF
163    where
164        EF: ExtensionField<F>,
165        F: TwoAdicField,
166    {
167        let arity = 2;
168        let log_arity = 1;
169        let (e0, e1) = evals
170            .collect_tuple()
171            .expect("TwoAdicFriFolder only supports arity=2");
172        // If performance critical, make this API stateful to avoid this
173        // This is a bit more math than is necessary, but leaving it here
174        // in case we want higher arity in the future.
175        let subgroup_start = F::two_adic_generator(log_height + log_arity)
176            .exp_u64(reverse_bits_len(index, log_height) as u64);
177        let mut xs = F::two_adic_generator(log_arity)
178            .shifted_powers(subgroup_start)
179            .collect_n(arity);
180        reverse_slice_index_bits(&mut xs);
181        assert_eq!(log_arity, 1, "can only interpolate two points for now");
182        // interpolate and evaluate at beta
183        e0 + (beta - xs[0]) * (e1 - e0) * (xs[1] - xs[0]).inverse()
184        // Currently Algebra<F> does not include division so we do it manually.
185        // Note we do not want to do an EF division as that is far more expensive.
186    }
187
188    fn fold_row_arbitrary<EF, F>(
189        &self,
190        index: usize,
191        log_height: usize,
192        beta: EF,
193        evals: impl Iterator<Item = EF>,
194        folding_factor: usize,
195    ) -> EF
196    where
197        EF: ExtensionField<F>,
198        F: TwoAdicField,
199    {
200        let arity = folding_factor;
201        let log_arity = log2_strict_usize(arity); // panics if arity is not a power of 2
202
203        let evals: Vec<EF> = evals.collect();
204        assert_eq!(
205            evals.len(),
206            folding_factor,
207            "Expected {} evaluations, got {}",
208            folding_factor,
209            evals.len()
210        );
211
212        // If performance critical, make this API stateful to avoid this
213        // This is a bit more math than is necessary, but leaving it here
214        // in case we want higher arity in the future.
215        let subgroup_start = F::two_adic_generator(log_height + log_arity)
216            .exp_u64(reverse_bits_len(index, log_height) as u64);
217        let mut xs = F::two_adic_generator(log_arity)
218            .shifted_powers(subgroup_start)
219            .collect_n(arity);
220        reverse_slice_index_bits(&mut xs);
221
222        // Perform Lagrange interpolation at beta
223        // We optimize by keeping denominators in base field F
224        lagrange_interpolate_ext(&xs, &evals, beta)
225    }
226
227    fn fold_matrix_2<EF, F, M: Matrix<EF>>(&self, beta: EF, m: &M) -> Vec<EF>
228    where
229        EF: ExtensionField<F>,
230        F: TwoAdicField,
231    {
232        // We use the fact that
233        //     p_e(x^2) = (p(x) + p(-x)) / 2
234        //     p_o(x^2) = (p(x) - p(-x)) / (2 x)
235        // that is,
236        //     p_e(g^(2i)) = (p(g^i) + p(g^(n/2 + i))) / 2
237        //     p_o(g^(2i)) = (p(g^i) - p(g^(n/2 + i))) / (2 g^i)
238        // so
239        //     result(g^(2i)) = p_e(g^(2i)) + beta p_o(g^(2i))
240        //
241        // As p_e, p_o will be in the extension field we want to find ways to avoid extension multiplications.
242        // We should only need a single one (namely multiplication by beta).
243        let g_inv = F::two_adic_generator(log2_strict_usize(m.height()) + 1).inverse();
244
245        // TODO: vectorize this (after we have packed extension fields)
246
247        // As beta is in the extension field, we want to avoid multiplying by it
248        // for as long as possible. Here we precompute the powers  `g_inv^i / 2` in the base field.
249        let mut halve_inv_powers = g_inv.shifted_powers(F::ONE.halve()).collect_n(m.height());
250        reverse_slice_index_bits(&mut halve_inv_powers);
251
252        m.par_rows()
253            .zip(halve_inv_powers)
254            .map(|(mut row, halve_inv_power)| {
255                let (lo, hi) = row.next_tuple().unwrap();
256                (lo + hi).halve() + (lo - hi) * beta * halve_inv_power
257            })
258            .collect()
259    }
260
261    /// Same as applying fold_row to every row, possibly faster.
262    fn fold_matrix_arbitrary<EF, F, M: Matrix<EF>>(
263        &self,
264        beta: EF,
265        m: &M,
266        folding_factor: usize,
267    ) -> Vec<EF>
268    where
269        EF: ExtensionField<F>,
270        F: TwoAdicField,
271    {
272        let log_arity = log2_strict_usize(folding_factor);
273        let log_height = log2_strict_usize(m.height());
274
275        // Precompute domain points for each row
276        let g = F::two_adic_generator(log_height + log_arity);
277
278        // For each row index, compute its domain points
279        let domain_points: Vec<Vec<F>> = (0..m.height())
280            .map(|index| {
281                let subgroup_start = g.exp_u64(reverse_bits_len(index, log_height) as u64);
282                let mut xs = F::two_adic_generator(log_arity)
283                    .shifted_powers(subgroup_start)
284                    .collect_n(folding_factor);
285                reverse_slice_index_bits(&mut xs);
286                xs
287            })
288            .collect();
289
290        // Process each row in parallel
291        m.par_rows()
292            .zip(domain_points)
293            .map(|(row, xs)| {
294                let evals: Vec<EF> = row.collect();
295                assert_eq!(evals.len(), folding_factor);
296
297                // Perform Lagrange interpolation at beta
298                lagrange_interpolate_ext(&xs, &evals, beta)
299            })
300            .collect()
301    }
302}

304impl<Val, Dft, InputMmcs, FriMmcs, Challenge, Challenger> Pcs<Challenge, Challenger>
305    for TwoAdicFriPcs<Val, Dft, InputMmcs, FriMmcs>
306where
307    Val: TwoAdicField,
308    Dft: TwoAdicSubgroupDft<Val>,
309    InputMmcs: Mmcs<Val>,
310    FriMmcs: Mmcs<Challenge>,
311    Challenge: ExtensionField<Val>,
312    Challenger:
313        FieldChallenger<Val> + CanObserve<FriMmcs::Commitment> + GrindingChallenger<Witness = Val>,
314{
315    type Domain = TwoAdicMultiplicativeCoset<Val>;
316    type Commitment = InputMmcs::Commitment;
317    type ProverData = InputMmcs::ProverData<RowMajorMatrix<Val>>;
318    type EvaluationsOnDomain<'a> = BitReversedMatrixView<RowMajorMatrixView<'a, Val>>;
319    type Proof = FriProof<Challenge, FriMmcs, Val, Vec<BatchOpening<Val, InputMmcs>>>;
320    type Error = FriError<FriMmcs::Error, InputMmcs::Error>;
321    const ZK: bool = false;

323    /// Get the unique subgroup `H` of size `|H| = degree`.
324    ///
325    /// # Panics:
326    /// This function will panic if `degree` is not a power of 2 or `degree > (1 << Val::TWO_ADICITY)`.
327    fn natural_domain_for_degree(&self, degree: usize) -> Self::Domain {
328        TwoAdicMultiplicativeCoset::new(Val::ONE, log2_strict_usize(degree)).unwrap()
329    }
330
331    /// Commit to a collection of evaluation matrices.
332    ///
333    /// Each element of `evaluations` contains a coset `shift * H` and a matrix `mat` with `mat.height() = |H|`.
334    /// Interpreting each column of `mat` as the evaluations of a polynomial `p_i(x)` over `shift * H`,
335    /// this computes the evaluations of `p_i` over `gK` where `g` is the chosen generator of the multiplicative group
336    /// of `Val` and `K` is the unique subgroup of order `|H| << self.fri.log_blowup`.
337    ///
338    /// This then outputs a Merkle commitment to these evaluations.
339    fn commit(
340        &self,
341        evaluations: impl IntoIterator<Item = (Self::Domain, RowMajorMatrix<Val>)>,
342    ) -> (Self::Commitment, Self::ProverData) {
343        let ldes: Vec<_> = evaluations
344            .into_iter()
345            .map(|(domain, evals)| {
346                assert_eq!(domain.size(), evals.height());
347                // coset_lde_batch converts from evaluations over `xH` to evaluations over `shift * x * K`.
348                // Hence, letting `shift = g/x` the output will be evaluations over `gK` as desired.
349                // When `x = g`, we could just use the standard LDE but currently this doesn't seem
350                // to give a meaningful performance boost.
351                let shift = Val::GENERATOR / domain.shift();
352                // Compute the LDE with blowup factor fri.log_blowup.
353                // We bit reverse as this is required by our implementation of the FRI protocol.
354                self.dft
355                    .coset_lde_batch(evals, self.fri.log_blowup, shift)
356                    .bit_reverse_rows()
357                    .to_row_major_matrix()
358            })
359            .collect();
360
361        // Commit to the bit-reversed LDEs.
362        self.mmcs.commit(ldes)
363    }

393    /// Given the evaluations on a domain `gH`, return the evaluations on a different domain `g'K`.
394    ///
395    /// Arguments:
396    /// - `prover_data`: The prover data containing all committed evaluation matrices.
397    /// - `idx`: The index of the matrix containing the evaluations we want. These evaluations
398    ///   are assumed to be over the coset `gH` where `g = Val::GENERATOR`.
399    /// - `domain`: The domain `g'K` on which to get evaluations on. Currently, this assumes that
400    ///   `g' = g` and `K` is a subgroup of `H` and panics if this is not the case.
401    fn get_evaluations_on_domain<'a>(
402        &self,
403        prover_data: &'a Self::ProverData,
404        idx: usize,
405        domain: Self::Domain,
406    ) -> Self::EvaluationsOnDomain<'a> {
407        // todo: handle extrapolation for LDEs we don't have
408        assert_eq!(domain.shift(), Val::GENERATOR);
409        let lde = self.mmcs.get_matrices(prover_data)[idx];
410        assert!(lde.height() >= domain.size());
411        lde.split_rows(domain.size()).0.bit_reverse_rows()
412    }

414    /// Open a batch of matrices at a collection of points.
415    ///
416    /// Returns the opened values along with a proof.
417    ///
418    /// This function assumes that all matrices correspond to evaluations over the
419    /// coset `gH` where `g = Val::GENERATOR` and `H` is a subgroup of appropriate size depending on the
420    /// matrix.
421    fn open(
422        &self,
423        // For each multi-matrix commitment,
424        commitment_data_with_opening_points: Vec<(
425            // The matrices and auxiliary prover data
426            &Self::ProverData,
427            // for each matrix,
428            Vec<
429                // points to open
430                Vec<Challenge>,
431            >,
432        )>,
433        challenger: &mut Challenger,
434    ) -> (OpenedValues<Challenge>, Self::Proof) {
435        /*
436
437        A quick rundown of the optimizations in this function:
438        We are trying to compute sum_i alpha^i * (p(X) - y)/(X - z),
439        for each z an opening point, y = p(z). Each p(X) is given as evaluations in bit-reversed order
440        in the columns of the matrices. y is computed by barycentric interpolation.
441        X and p(X) are in the base field; alpha, y and z are in the extension.
442        The primary goal is to minimize extension multiplications.
443
444        - Instead of computing all alpha^i, we just compute alpha^i for i up to the largest width
445        of a matrix, then multiply by an "alpha offset" when accumulating.
446              a^0 x0 + a^1 x1 + a^2 x2 + a^3 x3 + ...
447            = ( a^0 x0 + a^1 x1 ) + a^2 ( a^0 x2 + a^1 x3 ) + ...
448            (see `alpha_pows`, `alpha_pow_offset`, `num_reduced`)
449
450        - For each unique point z, we precompute 1/(X-z) for the largest subgroup opened at this point.
451        Since we compute it in bit-reversed order, smaller subgroups can simply truncate the vector.
452            (see `inv_denoms`)
453
454        - Then, for each matrix (with columns p_i) and opening point z, we want:
455            for each row (corresponding to subgroup element X):
456                reduced[X] += alpha_offset * sum_i [ alpha^i * inv_denom[X] * (p_i[X] - y[i]) ]
457
458            We can factor out inv_denom, and expand what's left:
459                reduced[X] += alpha_offset * inv_denom[X] * sum_i [ alpha^i * p_i[X] - alpha^i * y[i] ]
460
461            And separate the sum:
462                reduced[X] += alpha_offset * inv_denom[X] * [ sum_i [ alpha^i * p_i[X] ] - sum_i [ alpha^i * y[i] ] ]
463
464            And now the last sum doesn't depend on X, so we can precompute that for the matrix, too.
465            So the hot loop (that depends on both X and i) is just:
466                sum_i [ alpha^i * p_i[X] ]
467
468            with alpha^i an extension, p_i[X] a base
469
470        */
471
472        // Contained in each `Self::ProverData` is a list of matrices which have been committed to.
473        // We extract those matrices to be able to refer to them directly.
474        let mats_and_points = commitment_data_with_opening_points
475            .iter()
476            .map(|(data, points)| {
477                let mats = self
478                    .mmcs
479                    .get_matrices(data)
480                    .into_iter()
481                    .map(|m| m.as_view())
482                    .collect_vec();
483                debug_assert_eq!(
484                    mats.len(),
485                    points.len(),
486                    "each matrix should have a corresponding set of evaluation points"
487                );
488                (mats, points)
489            })
490            .collect_vec();
491
492        // Find the maximum height and the maximum width of matrices in the batch.
493        // These do not need to correspond to the same matrix.
494        let (global_max_height, global_max_width) = mats_and_points
495            .iter()
496            .flat_map(|(mats, _)| mats.iter().map(|m| (m.height(), m.width())))
497            .reduce(|(hmax, wmax), (h, w)| (hmax.max(h), wmax.max(w)))
498            .expect("No Matrices Supplied?");
499        let log_global_max_height = log2_strict_usize(global_max_height);
500
501        // Get all values of the coset `gH` for the largest necessary subgroup `H`.
502        // We also bit reverse which means that coset has the nice property that
503        // `coset[..2^i]` contains the values of `gK` for `|K| = 2^i`.
504        let coset = {
505            let coset =
506                TwoAdicMultiplicativeCoset::new(Val::GENERATOR, log_global_max_height).unwrap();
507            let mut coset_points = coset.iter().collect();
508            reverse_slice_index_bits(&mut coset_points);
509            coset_points
510        };
511
512        // For each unique opening point z, we will find the largest degree bound
513        // for that point, and precompute 1/(z - X) for the largest subgroup (in bitrev order).
514        let inv_denoms = compute_inverse_denominators(&mats_and_points, &coset);
515
516        // Evaluate coset representations and write openings to the challenger
517        let all_opened_values = mats_and_points
518            .iter()
519            .map(|(mats, points)| {
520                // For each collection of matrices
521                izip!(mats.iter(), points.iter())
522                    .map(|(mat, points_for_mat)| {
523                        // TODO: This assumes that every input matrix has a blowup of at least self.fri.log_blowup.
524                        // If the blow_up factor is smaller than self.fri.log_blowup, this will lead to errors.
525                        // If it is bigger, we shouldn't get any errors but it will be slightly slower.
526                        // Ideally, polynomials could be passed in with their blow_up factors known.
527
528                        // The point of this correction is that each column of the matrix corresponds to a low degree polynomial.
529                        // Hence we can save time by restricting the height of the matrix to be the minimal height which
530                        // uniquely identifies the polynomial.
531                        let h = mat.height() >> self.fri.log_blowup;
532
533                        // `subgroup` and `mat` are both in bit-reversed order, so we can truncate.
534                        let (low_coset, _) = mat.split_rows(h);
535                        let coset_h = &coset[..h];
536
537                        points_for_mat
538                            .iter()
539                            .map(|&point| {
540                                let _guard =
541                                    info_span!("evaluate matrix", dims = %mat.dimensions())
542                                        .entered();
543
544                                // Use Barycentric interpolation to evaluate each column of the matrix at the given point.
545                                let ys =
546                                    info_span!("compute opened values with Lagrange interpolation")
547                                        .in_scope(|| {
548                                            // Get the relevant inverse denominators for this point and use these to
549                                            // interpolate to get the evaluation of each polynomial in the matrix
550                                            // at the desired point.
551                                            let inv_denoms = &inv_denoms.get(&point).unwrap()[..h];
552                                            interpolate_coset_with_precomputation(
553                                                &low_coset,
554                                                Val::GENERATOR,
555                                                point,
556                                                coset_h,
557                                                inv_denoms,
558                                            )
559                                        });
560                                ys.iter()
561                                    .for_each(|&y| challenger.observe_algebra_element(y));
562                                ys
563                            })
564                            .collect_vec()
565                    })
566                    .collect_vec()
567            })
568            .collect_vec();
569
570        // Batch combination challenge
571
572        // Soundness Error:
573        // See the discussion in the doc comment of [`prove_fri`]. Essentially, the soundness error
574        // for this sample is tightly tied to the soundness error of the FRI protocol.
575        // Roughly speaking, at a minimum is it k/|EF| where `k` is the sum of, for each function, the number of
576        // points it needs to be opened at. This comes from the fact that we are taking a large linear combination
577        // of `(f(zeta) - f(x))/(zeta - x)` for each function `f` and all of `f`'s opening points.
578        // In our setup, k is two times the trace width plus the number of quotient polynomials.
579        let alpha: Challenge = challenger.sample_algebra_element();
580
581        // We precompute powers of alpha as we need the same powers for each matrix.
582        // We compute both a vector of unpacked powers and a vector of packed powers.
583        // TODO: It should be possible to refactor this to only use the packed powers but
584        // this is not a bottleneck so is not a priority.
585        let packed_alpha_powers =
586            Challenge::ExtensionPacking::packed_ext_powers_capped(alpha, global_max_width)
587                .collect_vec();
588        let alpha_powers =
589            Challenge::ExtensionPacking::to_ext_iter(packed_alpha_powers.iter().copied())
590                .collect_vec();
591
592        // Now that we have sent the openings to the verifier, it remains to prove
593        // that those openings are correct.
594
595        // Given a low degree polynomial `f(x)` with claimed evaluation `f(zeta)`, we can check
596        // that `f(zeta)` is correct by doing a low degree test on `(f(zeta) - f(x))/(zeta - x)`.
597        // We will use `alpha` to batch together both different claimed openings `zeta` and
598        // different polynomials `f` whose evaluation vectors have the same height.
599
600        // TODO: If we allow different polynomials to have different blow_up factors
601        // we may need to revisit this and to ensure it is safe to batch them together.
602
603        // num_reduced records the number of (function, opening point) pairs for each `log_height`.
604        // TODO: This should really be `[0; Val::TWO_ADICITY]` but that runs into issues with generics.
605        let mut num_reduced = [0; 32];
606
607        // For each `log_height` from 2^1 -> 2^32, reduced_openings will contain either `None`
608        // if there are no matrices of that height, or `Some(vec)` where `vec` is equal to
609        // a weighted sum of `(f(zeta) - f(x))/(zeta - x)` over all `f`'s of that height and
610        // for each `f`, all opening points `zeta`. The sum is weighted by powers of the challenge alpha.
611        let mut reduced_openings: [_; 32] = core::array::from_fn(|_| None);
612
613        for ((mats, points), openings_for_round) in
614            mats_and_points.iter().zip(all_opened_values.iter())
615        {
616            for (mat, points_for_mat, openings_for_mat) in
617                izip!(mats.iter(), points.iter(), openings_for_round.iter())
618            {
619                let _guard =
620                    info_span!("reduce matrix quotient", dims = %mat.dimensions()).entered();
621
622                let log_height = log2_strict_usize(mat.height());
623
624                // If this is our first matrix at this height, initialise reduced_openings to zero.
625                // Otherwise, get a mutable reference to it.
626                let reduced_opening_for_log_height = reduced_openings[log_height]
627                    .get_or_insert_with(|| vec![Challenge::ZERO; mat.height()]);
628                debug_assert_eq!(reduced_opening_for_log_height.len(), mat.height());
629
630                // Treating our matrix M as the evaluations of functions f_0, f_1, ...
631                // Compute the evaluations of `Mred(x) = f_0(x) + alpha*f_1(x) + ...`
632                let mat_compressed = info_span!("compress mat").in_scope(|| {
633                    // This will be reused for all points z which M is opened at so we collect into a vector.
634                    mat.rowwise_packed_dot_product::<Challenge>(&packed_alpha_powers)
635                        .collect::<Vec<_>>()
636                });
637
638                for (&point, openings) in points_for_mat.iter().zip(openings_for_mat) {
639                    // If we have multiple matrices at the same height, we need to scale alpha to combine them.
640                    // This means that reduced_openings will contain:
641                    // Mred_0(x) + alpha^{M_0.width()}Mred_1(x) + alpha^{M_0.width() + M_1.width()}Mred_2(x) + ...
642                    // Where M_0, M_1, ... are the matrices of the same height.
643                    let alpha_pow_offset = alpha.exp_u64(num_reduced[log_height] as u64);
644
645                    // As we have all the openings `f_i(z)`, we can combine them using `alpha`
646                    // in an identical way to before to compute `Mred(z)`.
647                    let reduced_openings: Challenge =
648                        dot_product(alpha_powers.iter().copied(), openings.iter().copied());
649
650                    mat_compressed
651                        .par_iter()
652                        .zip(reduced_opening_for_log_height.par_iter_mut())
653                        // inv_denoms contains `1/(z - x)` for `x` in a coset `gK`.
654                        // If `|K| =/= mat.height()` we actually want a subset of this
655                        // corresponding to the evaluations over `gH` for `|H| = mat.height()`.
656                        // As inv_denoms is bit reversed, the evaluations over `gH` are exactly
657                        // the evaluations over `gK` at the indices `0..mat.height()`.
658                        // So zip will truncate to the desired smaller length.
659                        .zip(inv_denoms.get(&point).unwrap().par_iter())
660                        // Map the function `Mred(x) -> (Mred(z) - Mred(x))/(z - x)`
661                        // across the evaluation vector of `Mred(x)`. Adjust by alpha_pow_offset
662                        // as needed.
663                        .for_each(|((&reduced_row, ro), &inv_denom)| {
664                            *ro += alpha_pow_offset * (reduced_openings - reduced_row) * inv_denom;
665                        });
666                    num_reduced[log_height] += mat.width();
667                }
668            }
669        }
670
671        // It remains to prove that all evaluation vectors in reduced_openings correspond to
672        // low degree functions.
673        let fri_input = reduced_openings.into_iter().rev().flatten().collect_vec();
674
675        let folding = TwoAdicFriFoldingForMmcs::<Val, InputMmcs>::new(self.fri.log_folding_factor);
676
677        // Produce the FRI proof.
678        let fri_proof = prover::prove_fri(
679            &folding,
680            &self.fri,
681            fri_input,
682            challenger,
683            log_global_max_height,
684            &commitment_data_with_opening_points,
685            &self.mmcs,
686        );
687
688        (all_opened_values, fri_proof)
689    }
690
691    fn verify(
692        &self,
693        // For each commitment:
694        commitments_with_opening_points: Vec<
695            CommitmentWithOpeningPoints<Challenge, Self::Commitment, Self::Domain>,
696        >,
697        proof: &Self::Proof,
698        challenger: &mut Challenger,
699    ) -> Result<(), Self::Error> {
700        // Write all evaluations to challenger.
701        // Need to ensure to do this in the same order as the prover.
702        for (_, round) in &commitments_with_opening_points {
703            for (_, mat) in round {
704                for (_, point) in mat {
705                    point
706                        .iter()
707                        .for_each(|&opening| challenger.observe_algebra_element(opening));
708                }
709            }
710        }
711
712        let folding = TwoAdicFriFoldingForMmcs::<Val, InputMmcs>::new(self.fri.log_folding_factor);
713
714        verifier::verify_fri(
715            &folding,
716            &self.fri,
717            proof,
718            challenger,
719            &commitments_with_opening_points,
720            &self.mmcs,
721        )?;
722
723        Ok(())
724    }
725}

727/// Compute vectors of inverse denominators for each unique opening point.
728///
729/// Arguments:
730/// - `mats_and_points` is a list of matrices and for each matrix a list of points. We assume that
731///    the total number of distinct points is very small as several methods contained herein are `O(n^2)`
732///    in the number of points.
733/// - `coset` is the set of points `gH` where `H` a two-adic subgroup such that `|H|` is greater
734///     than or equal to the largest height of any matrix in `mats_and_points`. The values
735///     in `coset` must be in bit-reversed order.
736///
737/// For each point `z`, let `M` be the matrix of largest height which opens at `z`.
738/// let `H_z` be the unique subgroup of order `M.height()`. Compute the vector of
739/// `1/(z - x)` for `x` in `gH_z`.
740///
741/// Return a LinearMap which allows us to recover the computed vectors for each `z`.
742#[instrument(skip_all)]
743fn compute_inverse_denominators<F: TwoAdicField, EF: ExtensionField<F>, M: Matrix<F>>(
744    mats_and_points: &[(Vec<M>, &Vec<Vec<EF>>)],
745    coset: &[F],
746) -> LinearMap<EF, Vec<EF>> {
747    // For each `z`, find the maximal height of any matrix which we need to
748    // open at `z`.
749    let mut max_log_height_for_point: LinearMap<EF, usize> = LinearMap::new();
750    for (mats, points) in mats_and_points {
751        for (mat, points_for_mat) in izip!(mats, *points) {
752            let log_height = log2_strict_usize(mat.height());
753            for &z in points_for_mat {
754                if let Some(lh) = max_log_height_for_point.get_mut(&z) {
755                    *lh = core::cmp::max(*lh, log_height);
756                } else {
757                    max_log_height_for_point.insert(z, log_height);
758                }
759            }
760        }
761    }
762
763    // Compute the inverse denominators for each point `z`.
764    max_log_height_for_point
765        .into_iter()
766        .map(|(z, log_height)| {
767            (
768                z,
769                batch_multiplicative_inverse(
770                    // As coset is stored in bit-reversed order,
771                    // we can just take the first `2^log_height` elements.
772                    &coset[..(1 << log_height)]
773                        .iter()
774                        .map(|&x| z - x)
775                        .collect_vec(),
776                ),
777            )
778        })
779        .collect()
780}

782/// Lagrange interpolation over an extension field.
783///
784/// Given base field points `xs` and extension field evaluations `ys`,
785/// compute the interpolated value at the extension field point `point`.
786fn lagrange_interpolate_ext<F, EF>(xs: &[F], ys: &[EF], point: EF) -> EF
787where
788    F: TwoAdicField,
789    EF: ExtensionField<F>,
790{
791    let n = xs.len();
792    debug_assert_eq!(xs.len(), ys.len(), "xs and ys must have the same length");
793
794    let mut numerators = Vec::with_capacity(n);
795    let mut denominators = Vec::with_capacity(n);
796
797    for i in 0..n {
798        let mut numerator = EF::ONE;
799        let mut denominator = F::ONE;
800
801        for j in 0..n {
802            if i != j {
803                numerator *= point - xs[j]; // Numerator: (point - xs[j]) in EF
804                denominator *= xs[i] - xs[j]; // Denominator: (xs[i] - xs[j]) in F
805            }
806        }
807
808        numerators.push(numerator);
809        denominators.push(denominator);
810    }
811
812    // Batch invert all denominators
813    let inv_denominators = batch_multiplicative_inverse(&denominators);
814
815    // Compute the final result: sum of ys[i] * numerator[i] / denominator[i]
816    let mut result = EF::ZERO;
817    for i in 0..n {
818        result += ys[i] * numerators[i] * inv_denominators[i];
819    }
820
821    result
822}
